[בֹּקֶר אוֹר אָשִׁיר עֻזֶּךָ] (רְשׁוּת לְנִשְׁמַת)

מתתיה בן יצחק מבולוניה

[כנראה נולד בעיר לוּצִ֗ינְיָנוֹ (Lucignano) בפלך טוֹסְקָנָה וחי בבּוֹלוֹנְיָה במחצית השניה של המאה הי"ד]

רְשׁוּת לְנִשְׁמַת שֶׁל שַׁבַּת זָכֹר

בֹּקֶר אוֹר אָשִׁיר עֻזֶּךָ / בִּקְהַל קֹדֶשׁ אַקְדִּישֶׁךָ

אַתָּה אֵל תַּגְדֵּל חַסְדֶּךָ / עַל־עָם מִתְפָּאֵר בִּשְׁמֶךָ.

מָעוֹן אַתָּה הַמִּתְנַשֵּׂא / עַל־רוּם מַעְלָה עוֹטֶה אוֹרָה

רוֹכֵב שָׁמַיִם וּמַחְסֶה / מַגְדִּיל חֶסֶד עִם־גְּבוּרָה

תִּפְאֶרֶת עֹז אֶל בּוֹ חוֹסֶה / נֶצַח הוֹד מַלְבִּישׁ עִם־אוֹרָה

שִׂים שָׁלוֹם יֶשַׁע וּבְרָכָה / עַל־עָם יוֹדֵעַ סוֹדֶךָ

עַל־זֹאת כָּל־חָסִיד אֵלֶיךָ / יִתֵּן מַהְלָל וּמַמְלָכָה.

תָּמִים נֹהֵג כַּצֹּאן יוֹסֵף / וּמְנוֹסֵס עַמְּךָ לִסְגֻלָּה

עֵת בֹּא שׂוֹטֵן לִשְׁקֹל כֶּסֶף / עַל־יַד עוֹשֶׂה בָעָם כָּלָּה

לָבוֹז בַּז כָּתַב עִם כֹּסֶף / אֶל־עָם וּמְדִינָה בִּמְגִלָּה

בֶּן־יָאִיר קוֹרֵא בִּשְׁמֶךָ / צוֹעֵק מָרָה עַל־בָּנֶיךָ

יָבִיא כֶסֶף כִּפּוּרֶיךָ / כֹּפֶר נַפְשָׁם אֶל־גִּנְזֶךָ.

תָּגֶל נַפְשִׁי בֵּאלֹהֵי כִּי / הִלְבִּישַׁנִי בִּגְדֵי־יֶשַׁע

עֵת קָם אִישׁ צַר סוֹרֵר דַּרְכִּי / אוֹיֵב הוּא הָמָן הָרָשָׁע

וַיִּתֵּן אֵל תָּמִים דַּרְכִּי / וַיְהִי יִשְׂרָאֵל עַם נוֹשָׁע

חָשַׁב עַל־צֹאן מַרְעִיתֶךָ / הַשְׁמִיד נַעַר וּזְקֵנֶיךָ

הֶעְתִּירוּ הוֹשַׁע צֹאנֶךָ / מוֹשִׁיעַ אַיִן בִּלְתֶּךָ.

יָחִיד נִחַם עַל־הָרָעָה / מִיַּד זוֹלֵל הוֹצִיא יָקָר

לִתְפִלַּת בֶּן־יָאִיר שָׁעָה / הֶעְטִיר עַמּוֹ שָׂשׂוֹן וִיקָר

אַתָּה אֵלִי לִי לִישׁוּעָה / הָיִית וַיְהִי צַר נֶעְקָר

כָּלוּ כֶעָשָׁן אוֹיְבֶיךָ / וַתִּלְבַּשׁ אֶסְתֵּר מַמְלָכָה

פָּנִים הָאֵר אֶל־עַבְדֶּךָ / הַיְשֵׁר אֶל־פָּנַי דַּרְכֶּךָ.

הַיֹּצֵר יַחַד לִבֵּנוּ / אֵלֶיךָ נָשָׂאתִי עֵינָי

חַי שִׂים בַּחַיִּים נַפְשֵׁנוּ / אָנֹכִי אֶשְׂמַח בַּיְיָ

זִכְרָךְ יִתְפָּאֵר מַלְכֵּנוּ / תִּתְהַלֵּל נַפְשִׁי בַּיְיָ

קוּמָה קַבֵּץ נִדָּחֶיךָ / נָא אַל־יַבשׁוּ קֹוֶיךָ

הוֹשִׁיעָה נָּא אֶת־עַמֶּךָ / נִשְׁמַת כָּל־חַי תּוֹדֶה שְׁמֶךָ.

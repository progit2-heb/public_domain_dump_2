נדוד הסיר אוני

אברהם אבן עזרא

א

נְדוֹד הֵסִיר אוֹנִי

וְהִבְהִיל רַעְיוֹנִי

וְשָׂם פִּי וּלְשׁוֹנִי

אֲסוּרִים בַּזִּקִּים

ב

לְפָנִים בַּנְּעוּרִים

הֲכִינוֹתִי שִׁירִים

בְּצַוַּאר הָעִבְרִים

נְתַתִּים כַּעֲנָקִים.

ג

בְּכָל מָקוֹם גַּרְתִּי

סְפָרִים חִבַּרְתִּי

וְסוֹדוֹת בֵּאַרְתִּי

וְכִרְאִי מוּצָקִים.

ד

שְׂפָתַי אִם אָשִׂים

בֶּהָרִים וּרְכָסִים

יְעוּפוּן בָּם סוּסִים

יְרוּצוּן כִּבְרָקִים.

ה

וְעַתָּה נָפַלְתִּי

לְעָפָר שָׁפַלְתִּי

וְתַחֲנוּן הִפַּלְתִּי

לְאֵל מִמַּעֲמַקִּים.

ו

אֲכָלוּנִי עוֹפוֹת

עֲדֵי הַקִּיפוֹפוֹת

וְעַל רֹאש אַשְׁפּוֹת

בְּנֵי עַוְלָה זוֹרְקִים.

ז

וּבִרְבוֹת טוֹבָתָם

יְכַחֲשׁוּ בַאֲבוֹתָם

וְעַל זַךְ חַרְבוֹתָם

מְנִיפִים וּמְרִיקִים.

ח

וְאַנְשֵׁי לֵב מָטוּ

וְהַיֹּום נִמְעָטוּ

אֲבָל שָׁמְנוּ בָעֲטוּ

אֲשֶׁר צִיָּה עוֹרְקִים

ט

יְקָר רָאֲתָה עֵינָם

וְאִם גֵּרִים הִנָּם

מְזָוֵיהֶם חִנָּם

מְלֵאִים וּמְפִיקִים.

י

וְאָנֹכִי מוּסָר

לְעֵינֵי כָל בָּשָׂר

כְּבוֹד הוֹדִי סָר

וְשֵׁן עָלַי חוֹרְקִים.

יא

וְגָבַר מַחֲשָׁבִי

לְאָרְכִּי וּלְרָחְבִּי

וּפֹה לֹא אֶפְתַּח פִּי

שְׁדֵי דַּעַת צוֹמְקִים

יב

וְחַיָּתִי סָפָה

בְּעַם עִמְקֵי שָׂפָה

וְנַפְשִׁי נִכְסָפָה

לְרַבִּי אֶלְיָקִים

יג

גְּבִיר בְּמִשְׁפָחוֹת

גְּבִיר דֹּבֵר צַחוֹת

גְּבִיר מֵשִׁיב רוּחוֹת

תְּעוּדָתוֹ מֵקִים.

יד

וְעַל פִּיו נִשְׁמָעוֹת

צְפוּנוֹת כָּל דֵּעוֹת

בְּדַת אֵל, וִירִיעוֹת

תְּעוּדָתוֹ מֵקִים

טו

שְׂפָתָיו סַפִּירִים

אֲמָרָיו יַקִּירִים

וּמִפַּחְדּוֹ שָׂרִים

יְבֹאוּן בִּנְקִיקִים.

טז

אֱמֶת רֵעַ אֵין לוֹ

בְּדַעְתּוֹ וּבְשִׂכְלוֹ

וּמַחֲמַדִּים כֻּלּוֹ

וְחִכּוֹ מַמְתַּקִּים.

יז

שְׁמָעֵנִי אָחִי

שְׁמַע נָא אֶל שִׂיחִי

אֲהֵבַתְךָ רוּחִי

וְעֵדִי שְׁחָקִים.

יח

וְלָמָּה נִתְעַבֵּר

בְּזֵד כִּי הִתְגַּבֵּר

וּמַה נּוּכַל דַּבֵּר

בְּאֶרֶץ מֶרְחַקִּים.

יט

אֲנַחְנוּ עַם דַּלּוּת

וְקַלּוּת עִם סִכְלוּת

בְּגָלוּת תּוֹךְ גָּלוּת

בְּכָל עֵת מַעְתִּיקִים.

כ

בְּיַד אֵל לוּ מַתְנוּ

בְּעֵת שֶׁיָשַׁבְנוּ

בְּמֶרְחָב, כִּי שַׁבְנוּ

לְחוּצִים וּדְחוּקִים.

כא

וּבְאֶדוֹם אֵין הָדָר

לְכָל חָכָם הוּא דָּר

בְּאַדְמַת בֶּן קֵדָר

וְעָלֵינוּ שׁוֹרְקִים

כב

וְאִלּוּ בָּא חָגָב

יְוָנִי בָּם שֻׂגַּב

וְרָכַב עַל כָּל גַּב

וְנֶחְשָׁב כַּעֲנָקִים

כג

וְחֶרְפָּה וָבשֶׁת

לְנֶפֶשׁ נִקְדֶּשֶׁת

אֲשֶׁר לֵבָב תָּשֵׁת

לְאַחַד הָרֵקִים

כד

הֲלֹא הַמּתְהוֹלֵל

בְּרִית קֹדֶשׁ חִלֵּל

עֲזֹב לוֹ וִיקַלֵּל

אֲנָשִׁים צַדִּיקִים

כה

כְּמוֹ שִׁמְעִי, כִּשְׁמוֹ

וְהַיַּחַשׂ עִמּוֹ

וְחֶשְׁבּוֹן עִם טַעֲמוֹ

דְּבָרִים עַתִּיקִים

כו

יְוָנִי בָּהֶם שִׂים

יְהִי שִׁקּוּץ מַעֲשִׂים

אֲשׂוֹחַח בִּיחָשִׂים

בְּהֵא הַדִּקְדּוּקִים.

כז

וְעֵינוֹ עָנְתָה בוֹ

עֲלֵי רֹעַ לִבּוֹ

בְּקִרְבּוֹ שָׂם אָרְבּוֹ

וְעֹשֵׂה תַמְרוּקִים.

כח

בְּהִשְׁתַּחֲוָיָתוֹ

לְכָל רֹאֶה אוֹתוֹ

מְלֹא כָל קֹמָתוֹ

וּמַשְׁפִּיל חַמּוּקִּים.

כט

וְחֵרֵף בֶּן תּוֹרָה

וְכָל יוֹדְעֵי מִקְרָא

לְזֹאת אַנְשֵׁי מוֹרָא

מְכֻסִּים בְּשַׂקִּים.

ל

וְאֵיךְ יִקְרָא מִינִים

לְבָנִים נֶאֱמָנִים

חֲסִידִים וּגְאוֹנִים

וְכָל שֹׁמְעָיו שֹׁתְקִים

לא

וְאַיֵּה קִנְאַת רָב

שְׁלֹמֹה נֵר מַעֲרָב

וְלָמָּה זֶה לֹא רָב

לַמֵּתִים וְלַעֲשׁוּקִים

לב

וְרַבֵּנוֹ יִצְחָק

יְבֹרַךְ מִשַּׁחַק

אֲמָרִים אֵלָיו חַק

וְהַיּוֹם נִמְחָקִים.

לג

וְאָמַר לַתּוֹעִים

אֲנִי מָלֵא דֵּעִים

בְּמוֹעֵד וּזְרָעִים

וְנָשִׁים וּנְזִיקִים.

לד

וְצַעֲקוֹתָיו רַבּוֹת

וְנָגְעוּ עַד עָבוֹת

וְלֹא יֵדַע אָבוֹת

וּבַמֶּה מַדְלִיקִים.

לה

כְּלִמַּת הַמַּמְרֶה

צְעִיר יָמִים יוֹרֶה

בְּטָהֳרוֹת הוּא קוֹרֵא

וְטֹעֶה בַּפְּסוּקִים.

לו

וְרֵאשִית פָּרָשָׁה

עֲלֵי פִיהוּ קָשָׁה

אֲשֶׁר הִיא נִדְרָשָׁה

בְּפִי הַתִּינוֹקִים.

לז

חֲשַׁבְנוּהוּ אָרוֹן

וְהוּא אַף גַּם חָרוֹן

וּבַבַּיִת חֶסְרוֹן

וְעוֹד בּוֹ מַחֲזִיקִים.

לח

לְמֵבִין נִסְתָּרִים

שְׁתֵּי עֵינַיִם אָרִים

לְמַלֵּט נוֹתָרִים

חֲמוּדִים וַחֲשׁוּקִים.

לט

וּבִנְיָמִין צָעִיר

קְנָאוֹת בִי הֵעִיר

וְכָאֵשׁ שֶׁהִבְעִיר

קְרָבַי נִשָּׁקִים.

מ

אֲהָהּ עַל זִכְרֶךָ

הֲכִי מַאֲמָרֶיךָ

בְּלִבּוֹת הוֹרֶיךָ

שְׁלָטִים נִמְרָקִים.

מא

תְּהִלּוֹת הִשְׁמַעְתָּ

וְתוֹרָה יָדַעְתָּ

מְהֵרָה נָגַעְתָּ

לְמִקְרָא וּפְרָקִים.

מב

הֲיָמִים שָׂבַעְתָּ

וּמַה זֶּה פָשַׁעְתָּ

וְלָמָּה גָוַעְתָּ

כְּמוֹ הַנֶּחֱנָקִים.

מג

בְּבֵיתְךָ שָׂטָן בָּא

וְרָעָתְךָ רַבָּה

לְזֹאת נֵרְךָ נִכְבָּה

וְנַדְתָּ לִמְצוּקִים.

מד

וְאָב מָלֵא רַחֲמִים

יְשַׁלֵּם נִחוּמִים

לְאָבוֹת נִדְהָמִים

שְׁנֵי הַוָּתִיקִים.

מה

וּמַשְׁחִית וּמְחַבֵּל

יְנֻדֶּה מִתֵּבֵל

וּמִנַּחֲלַת חֶבֶל

יְסוּרוּן מַזִּיקִים.

מו

וּפֹה תַמּוּ שִׁירִים

עֲלֵי פִי נִדְבָּרִים

וְאֵלֶּה הָאֲחֵרִים

לְבָנִים וּשְׁרוּקִים.

מז

אֲני לֹא חִבַּרְתִּם

בְיַד חָכָם שַׁרְתִּם

וְעִם שִׁירַי שַׂמְתִּים

וְהָיוּ נִדְבָּקִים.

מח

נְדִיבֵי עָם עוּשׁוּ

וְאַל נָא תַחֲרִישׁוּ

פְּנוּ פֶן תֵּעָנְשׁוּ

וְתִהְיוּ כַּף סוֹפְקִים.

מט

הֲסָרָה צִדְקַתְכֶם

הֲסָרָה חָכְמַתְכֶם

זָר יִמְשֹׁל בָּכֶם

וְשַׁבְתֶּם נֶחֱלָקִים

נ

תְּנוּ לֵב לַתְּלָאָה

בְּמַלְכוּת חַטָּאָה

אֲשֶׁר אֵין בָּהֶם יִרְאָה

וּבָשָׂר לֹא בֹּדְקִים.

נא

וְאֵיךְ יִזְבַּח בָּשָׂר

יְוָנִי כֹּל חָסֵר

וּמִדֶּרֶךְ אֵל סָר

וּמַשְׁלִיךּ חַלֻּקִים.

נב

יְהוּדָה הֶעָנָיו

לְעֵת בִּרְאוֹת עֵינָיו

חֲזִיר בֵּינוֹת שִׁנָּיו

וְאֹכְלוֹ בִּשְׁוָקִים.

נג

נְאֻם זֶה אֶלְעָזָר

וּמשֶׁה הַנֶּעֱזָר

וּמוֹטוֹת כֹּהֵן זָר

עֲדֶנָּה נִתָּקִים

נד

בְּבֵית לֵוִי תּוֹשָׁב

וְכֶעָרֵל נֶחְשָׁב

וּמִי יֵדַע אִם שָׁב

וְאִם אֵין מַסְפִּיקִים

נה

בְּנֵי אֵל עִזְבוּהוּ

הֲכִי אֵין כָּמֹהוּ

וּבַמֶּה נֶחְשָׁב הוּא

בְּמִי אַתֶּם עוֹסְקִים

נו

בְּיָדוֹ יִפֹּלוּ

מְתִים שֶׁהִסְכִּילוּ

אֲחֵרִים יַכְשִׁילוּ

וְלָשׁוֹן מַחֲלִיקִים.

נז

מְרֵעָיו בּוֹ שִׂמְחוּ

וְיֶדְכֶם לוֹ פִתְחוּ

וּמַתָּנוֹת שִׁלְחוּ

וּמַלְּאוּ מִזְרָקִים.

נח

וְאִם בּוֹ תַאֲמִינוּ

נְדִיבוֹתָיו בִּינוּ

וּמִטָּה הָכִינוּ

וּמַלְבּוּשִׁים נְקִיִּם

נט

תְּנוּ לוֹ שֵׁם יַלְדָּה

לְאִשָּׁה בַּת חֶמְדָּה

בְּנוּ לוֹ בֵית מִדָּה

עֲשׂוּ שָׁם אַתִּיקִים.

ס

וּמַאֲכָל גַּם מַשְׁקֶה

תְּנוּ אֶל רַב שָׁקֵה

אֲשֶׁר לֹא יִנָּקֶה

כְּמוֹ שַׂר הַמַּשְׁקִים.

סא

וְטוֹב הוּא שֶׁיִּלְחַם

בְּכָל יֹום לֶחֶם חַם

בְּמֶלֶךְ אוֹ פֶחָם

כְּמוֹ בֶן תַּפְנוּקִים.

סב

עֲשׂוּ עוֹף רִאשוֹנָה

וְדָג בָּאַחֲרוֹנָה

צְלוּ עוֹד בֶּן יוֹנָה

קְנוּ לוֹ צִמּוּקִים.

סג

לְהַצְהִיל הַפָּנִים

תְּנוּ עוֹד עַרְמוֹנִים

אֱגוֹזִים בִּתְאֵנִים

שְׁבוּרִים וּמְנֻקִּים

סד

וּבֶאֱלוֹהַּ שָׁמֵן

לְבַבְכֶם יֵאָמֵן

אֲבָל עַם אֵל אָמָּן

בְּפָנָיו הֵם יוֹרְקִים.

סה

בְּצִלּוֹ תֶחֱסָיוּ

מְשַׁרְתָיו אָז תִּחְיוּ

עֲדֵי עַד אִם תִּהְיוּ

עֲפַר רַגְלָיו נוֹשְׁקִים.

סו

וְעִבְדוּ זֶה הָאֵל

וְלָתֵת חֵן יוֹאֵל

וְעַתָּה יִשְׂרָאֵל

שְׁמַע אֶל הַחֻקִּים.

את הטקסט לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט. הוא זמין תמיד בכתובת הבאה: https://benyehuda.org/read/8618

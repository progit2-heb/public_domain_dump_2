מחברות עמנואל: מחברת רביעית

אָמַר עִמָּנוּאֵל הַמְחַבֵּר:

אַחֲרֵי אֲשֶׁר יָצָאתִי מִבֵּית הַשּׂר בּוֹשׁ וְנִכְלָם

עַל אֲשֶׁר סִבַּבְתִּי מוֹת הַגְּבִירָה אֲשֶׁר לֹא אֶתְנַחֵם אַחֲרֶיהָ לְעוֹלָם

יָשַׁבְתִּי בְאֶרֶץ-נוֹד כְּעֶשֶׂר שָׁנִים

וְעָבְרוּ עָלַי גַּלֵּי יְגוֹנִים

עַד אֲשֶׁר הֵחֵלּוּ לִצְמֹחַ יְמֵי הַזְּקֻנִים

וּבְשִׁבְתִּי מַרְעִיד וּמַשְׁמִים

אָחוּז בְּחַבְלֵי בוּז וְאֵימִים

מִפְּנֵי תוֹלְדוֹת הַיָּמִים

וְהִנֵּה עַל הֶהָרִים רַגְלֵי מְבַשּׂר מַשְׁמִיעַ שָׁלוֹם

בָּא עַד הֲלוֹם

וַיֹּאמֶר אֵלַי הַשּׂר שְׁלָחַנִי אֵלֶיךָ כִּי לִרְאוֹתְךָ יַחְמֹד

וְלָכֵן מְהֵרָה חוּשָׁה אַל תַּעֲמֹד

אֲנִי דִבְרֵי הַמְּבַשּׂר הֶאֱזַנְתִּי

וַיָּפָג לִבִּי כִּי לֹא הֶאֱמַנְתִּי

וָאֹמַר הֲשָׁלוֹם אֶל הַשָּׂר הַחֹמֵל וְהַרַחֲמָן

נֵזֶר הַזְּמָן

וַיֹּאמַר שָׁלוֹם תְּהִלָּה לָאֵל הַנֶּאֱמָן

הִנֵּה הוּא שׁוֹלֵחַ לְךָ זֹאת הַשּׁירָה הַנִּכְבָּדָה

בַּעֲבוּר תִּהְיֶה לְךָ לְעֵדָה

וַיּוֹצִיאָהּ מֵחֵיקוֹ וְהָאָרֶץ הֵאִירָה מִכְּבוֹדָהּ

וְהִנֵּה זֹאת שִׁירַת הַשּׂר יָרוּם הוֹדוֹ וְתִפְאַרְתּוֹ

הַשּׁלוּחָה אֵלַי מֵאִתּוֹ:

אֵי זֶה הַדֶּרֶךְ יִשְׁכָּן-אוֹר

יָשִׂישׂ יִזְרַח לָרוּץ אֹרַח

הוּא דוֹד נִסַּע נִגְלָה מִנִּי

מִיָּדִי בָּרוֹחַ יִבְרָח

פִּנַּת יִקְרַת הוּא לִמְנוֹרַת

הַשּׂכֶל כַּפְתֹּר וָפָרַח

יוֹם מִבְחַן מַטּוֹת בִּין לִפְנֵי

הָעֵדוּת מַטֵּהוּ פָרָח

נַפְשִׁי אִוִּיתִיהוֹ לַיְלָה

אַף רוּחִי בִי מָרָה יִצְרָח

הָיִיתִי יוֹם אָכַל אֹתִי

חֹרֶב וּבַלַּיְלָה קָרַח

שָׂנֵאתְי חַגִּי עִם חָדְשִׁי

הָיוּ עַל נַפְשִׁי לָטֹרַח

כִּי-הִכְאִיב לִבִּי יוֹם נֹדוֹ

יוֹם שׁוּבוֹ לִשְׁחִינִי יִמְרָח

אִם-כֹּפֶר יוּשַׁת עַל שׁוּבוֹ

נָתַתִּי כָפְרוֹ הוֹן קֹרַח

לוּ יִרְצֶה מִנְחָה מִיָּדִי

הֵא מִנְחָתִי אוֹתָהּ יָרָח

רֹאָיו אֵלָיו יַשְׁגִּיחוּ כִּי

אוֹר שִׂכְלוֹ עַל לִבָּם יִזְרָח

הֵמָּה רָאוּ כֵּן תָּמָהוּ

מִי הֵעִיר צֶדֶק מִמִּזְרָח.

וַיְהִי כִּרְאוֹתִי הַשּׁירָה הַהֲדוּרָה אֱלֹהֵי אָבִי בֵרַכְתִּי

וָאָשִׂים לַדֶּרֶךְ פְּעָמַי וְהָרִים וּגְבָעוֹת דָּרַכְתִּי

וּנְדֻדִים שָׂבַעְתִּי

עַד אֲשֶׁר אֶל בֵּית הַשּׂר הִגַעְתִּי

וְכִרְאוֹת הַשּׂר אוֹתִי מֵרָחוֹק

אָז נִמְלָא פִּיו שְׂחוֹק

וַיֹאמַר שָׁלוֹם שָׁלוֹם לָרָחוֹק

בֹּא יָבֹא בְרִנָּה נוֹשֵׂא אֲלֻמּוֹת הַזִּקְנָה

אֲשֶׁר בִּנְדוֹדוֹ לִבִּי עִנָּה

וָאַעַן וָאֹמַר אֲדוֹנִי הַשּׂר אֲשֶׁר גָּבַרְתָּ בְּשִׁכְלְךָ

גַּם הַיּוֹם מִשְׁנֵה שָׁלוֹם אַגִּיד לְךָ

וְאִם זָקֵן אֲנִי אֵיךְ תֵּשֵׁב עַד בּוֹאִי

וְאִם אֲדוֹנִים אָנִי אַיֵּה מוֹרָאִי.


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


עֶלֶם זְכֹר יוֹם מִפְּנֵי זֹקֶן

תָּנוּד כְּנוּד צִפּוֹר דְּרוֹר מִקֵּן

עַל כֵּן בְּחִירִי מִפְּנֵי שֵׂיבָה

תָּקוּם וְהָדַרְתָּ פְּנֵי זָקֵן.


וַיִתְמַהּ הַשּׂר עַל מַעֲנַי וְשִׂיחָי

וַיֹּאַמר אָמוּתָה הַפָּעַם אַחֲרֵי רְאוֹתִי אֶת פָּנֶיךָ כִּי עוֹדְךָ חָי

חֵי הַבְּרִית כִּי עַל שְׁלוֹמְךָ הִתְפַּלָּלְתִּי

וְרָאֹה פָנֶיךָ לֹא פִלָּלְתִּי

וּמֵאָז רָתַמְתָּ פִּרְדֵי הַפְּרִידָה בְּעִמְקֵי הַדְּאָגוֹת צָלַלְתִּי

וְקַרְנִי בֶעָפָר עֹלַלְתִּי

וְעַתָּה בִּרְאוֹתִי פָנֶיךָ אֱלֹהִים הִלַּלְתִּי כִּי נָפְלוּ לִי בַּנְּעִימִים חֲבָלִים

בָּרוּךְ הַמַּחֲזִיר אֲבֵדָה לַבְּעָלִים.

וַיִּשּׂא מְשָׁלוֹ וַיֹּאמַר:


רְאֹה יִפְעָתְךָ לֹא פִלְלוּ עוֹד

יְבֹרַךְ מַחֲזִירְךָ אֶל בְּעָלִים

יְסוֹד לִבְּךָ מְגֻלֶּה עַל-לְשׁוֹנְךָ

כְּהִגָּלוֹת יְסוֹד שֹׁרֶשׁ בְּעָלִים

וְעֵינֵי הַנְּפָשׁוֹת לָךְ נְשֻׂאוֹת

כְּעֵינֵי הַגְּבִירוֹת אֶל-בְּעָלִים

וְכָל לָשׁוֹן לְךָ תִכְסֹף כְּמוֹאָב

אֱלֵי מִלְכֹּם וְעַמּוֹן אֶל-בְּעָלִים.


וְעַתָּה אֱמָר-נָא נֵזֶר הַזְּמָן

לָמָּה נַחְבֵּאתָ לִבְרֹחַ מִמּוֹדָעֲךָ הַנֶּאֱמָן

וַתִּגְנֹב אֶת-לְבָבִי בְּפֵרוּדֶךָ

וְהִכְאַבְתַּנִי בְנוּדֶךָ

הַאֻמְנָם לֹא אוּכַל כַּבְּדֶךָ

וָאַעַן וָאֹמַר חֵי אַהֲבָתְךָ הַשּׂר הַיְקָרָה

אֲשֶׁר הִיא אֵלַי לַעֲטֶרֶת צְבִי וְלִצְפִירַת תִּפְאָרָה

כִּי רוּחַ אַחֵר עָלַי לֹא עָבָר

אִם לֹא מִדְּאָגָה מִדָּבָר

פֶּן תִּפְקֹד עָלַי עֲוֹן הַגְּבֶרֶת הַטְּהוֹרָה

אֲשֶׁר עֲווֹנוֹתַי כָּרְתוּ מִפְלַצְתָּהּ וֶהֱסִירוּהָ מִגְבִירָה

וְעָלֶיהָ לִבִּי כְּלֵב אִשּׁה מְצֵרָה

וַיַעַן הַשּׂר וַיֹּאמַר חַי אָנִי כִּי לְמוֹת הַגְּבִירָה הִתְאַבַּלְתִּי

וּמֵאָז לֶחֶם הַפֶּחָה לֹא אָכַלְתִּי

כִּי הָיְתָה יְדִידוּת נַפְשִׁי

וְצָנִיף הַטָּהוֹר עַל רֹאשִׁי

וּבָאוּ כַמָּה מוֹצִיאֵי דִבָּה

לֵאמֹר כִּי הָיִיתָ לְמִיתַת הַצְּבִיָּה סִבָּה

וַיֹּאמְרוּ הָבָה נִתְחַכְּמָה לוֹ פֶּן יִרְבֶּה

וְעִם כָּל זֶה בָעֲרָה בִּי אֵשׁ חִשְׁקְךָ וְאֵין מְכַבֶּה

וּמוֹצִיאֵי הַדִּבָּה הֶחֱשֵׁיתִי

וְאֶת לִבִּי לְמוּלָם הִקְשֵׁיתִי

עַל אַחַת כַּמָּה וְכַמָּה

אֲשֶׁר לוּ רָצַחְתָּ כָל קְרוֹבַי צָפוֹנָה וָנֶגְבָּה וָקֵדְמָה וְיָמָּה

לִבִּי נָתוּן לְךָ וְשַׂמְתִּי לְךָ מָקוֹם אֲשֶׁר תָּנוּס שָׁמָּה.


וַיִּשּׂא מְשָׁלוֹ וַיֹּאמַר:


לִבִּי לְחִשְׁקָךְ הַצְּבִי יִכְסֹף

וַיְהִי בְלִבִּי חִשְׁקְךָ כָנוּס

אֶתְמַהּ לְלִבִּי יֵט לְחִשְׁקֶךָ

כִּי תֶאֱנֹס אֹתוֹ וְהוּא אָנוּס

לִבִּי הֲכִי תִרְצַח וְתוֹךְ לִבִּי

שַׂמְתִּי לְךָ מָקוֹם אֲשֶׁר תָּנוּס.


וָאַעַן וָאֹמַר מַה גָּדְלוּ מַעֲשֶׂיךָ הַשּׂר אַשְׁרֶיךָ

וּמַה נִּפְלְאוּ מְאֹד מוּסָרֶיךָ

אַשְׁרֵי תִּבְחַר וּתְקָרֵב יִשְׁכֹּן חֲצֵרֶיךָ

כִּי הִכְאַבְתִּיךָ וְרִחַמְתָּנִי

וְהִרְחַקְתִּי נְדֹד וְשִׁחַרְתָּנִי

וַתִּקְטַן זֹאת בְּעֵינֶיךָ וַתַּהַס מוֹצִיאִים עָלַי דִּבָּה

וַתֵּסָעֲרֵם בִּסְעָרָה לְהָכִיל מִרְבָּה

כְּמֹץ יְסֹעַר מִגֹּרֶן וּכְעָשָׁן מֵאֲרֻבָּה.


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


לִבִּי מִצּוּף דִּבְשָׁךְ רָצוּף

עַד מֵחִשְׁקְָך הוּא קָלִי בִי

הַגֵּד עֹפֶר אִם לֹא תֵדַע

כִּי בִנְדֹדִי הוּכָּה לִבִּי

לוּ יוֹלַדְתָּךְ הִיא יוֹלַדְתִּי

כִּי לִבֶּךָ הוּא כַלִּבִּי

כִּי תַהַס כָּל מוֹצִיא דִבָּה

הַגֵּד אִם אתְּ הוּא כָלִבִּי.


וַיֹּאמֶר הַשּׂר בְּחַסְדֶּךָ הַגֶּד-נָא לִי אִם עַל הָרוֹב

מָצָאתָ בִנְדוּדְךָ גּוֹאֵל אוֹ קָרוֹב

וָאַעַן וָאֹמַר חַיֶּיךָ הַשּׂר כִּי יוֹם הַנְּדִיבוֹת פָּנָה לַעֲרוֹב

אֶרְאֶנּוּ וְלֹא עַתָּה אֲשׁוּרֶנּוּ וְלֹא קָרוֹב

וְהַקּוֹפְצִים יָדָם הֵם כְּכוֹכְבֵי הַשּׁמַיִם לָרוֹב

וְרָאִיתִי מִן הַנִּקְרָאִים נְדִיבִים הֵם כְּמוֹ שִׁשּׁים רִבּוֹא

וְלֹא מָצָאתִי אֶל בֵּיתָם מָבוֹא

כִּי סֻגַּר כָּל-בַּיִת מִבּוֹא

וּבְעֵת אֶל בֵּית אֶחָד מֵהֶם אָבוֹא

דָּמִיתִי אֶל יַעֲקֹב הָעֵת אָמְרַה לוֹ לֵאָה אֵלַי תָּבוֹא.


וָאֶשּׂא מְשָׁלִי וָאֹמַר:

יוֹם גֻּנַּבְתִּי דּוֹד מֵחֵיקְךָ

צִירַי חָשׁוּ עָלַי לִצְבֹּא

בָּחַרְתִּי הִסְתּוֹפֵף עִמְּךָ

מִדּוּר עִם מֶלֶךְ בִּמְסִבּוֹ

עִמְּךָ לֹא עִם צִבְאוֹת שַׁחַק

עִמְךָ לֹא עִם שִׁשּׁים רִבּוֹא

לֹא מִלִּבִּי שִׂמְחַת גִּילִי

לוּ לֵב עַבְדָּךְ חוּץ מִקִּרְבּוֹ

כִּי מִיּוֹם מִמְּךָ נָדַדְתִּי

סֻגַּר לִי כָל בַּיִת מִבּוֹא

וּבְעֵת נוּדִי דּוֹד וּמְרוּדִי

אִם בֵּית שָׂר אוֹ טִפְסָר אָבוֹא

דָּמִיתִי אֶל יַעְקֹב הָעֵת

אָמְרָה לֵאָה אֵלַי תָבוֹא.


אָמְנָם נְדִיבוּתְךָ הַשּׂר גָּבְרָה עַל כָּל נְדִיבוּת

וַעֲרֵבוּתְךָ עַל כָּל עֲרֵבוּת

כִּי שְׂחוֹק פִּיךָ יֶאֱסֹף הַדַּלִּים וְהַשּׁקְקִים

וְהַשּׁחָקִים בּוֹכִים לְעֵת טַלָּם מְרִיקִים.


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


לְקוֹל תִּתְּךָ הֲמוֹן גִּשְׁמֵי נְדָבוֹת

שְׂחוֹק פִּיךָ יְאַסֵּף הַמְחַכִּים

וְיָנוּסוּ מְחַכֵּי הַשּׁחָקִים

וְנִדְבָתָם בְּשׁוּרָם כִי מְבַכִּים.


וַיֹּאמֶר הַשּׂר צַר לִי כִּי הִפְסַדְתִּי מִשּׁנוֹתֶיךָ עֲשָׂרָה

וְלֹא תִהְיֶה לִי מֵהֶם לְעוֹלָם תְּמוּרָה

וָאַעַן וָאֹמַר חַיֶּיךָ אֲדֹנִי אֲנִי הַמַּפְסִיד

כִּי קְוֻצּוֹתַי תַּלְתַּלִּים שְׁחוֹרוֹת שָׂד אוֹתָם הַזְּמָן בַּשִּׂיד

וְהִנְנִי אֲחֹרַנִּית פֹּסֵעַ

וּמִיָּטְבָתָה אֶל אֶדְרֶעִי נוֹסֵעַ.


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


הַחֹלִי הוּא זֹקֶן מִקְרִי

אַךְ הַזֹּקֶן חֹלִי טִבְעִי

מִי שֶׁיַּזְקִין הוּא נֹסֵעַ

מִיָּטְבָתָה אֶל-אֶדְרֶעִי.


וַיִּשְׂחַק הַשּׂר וַיֹּאמַר אַשְׁבִּיעֲךָ בְאַהֲבָתֵנוּ הַנֶּאֱמָנָה

לְהוֹדִיעֵנִי אִם קוֹנַנְתָּ עַל הַזִּקְנָה

וָאַעַן וָאֹמַר חַיֶּיךָ קוֹנַנְתִּי בְּעַל כָּרְחִי

כִּי כָהֲתָה עֵינִי וְנָס לֵחִי

וְצָרַעַת הַזֹּקֶן זָרְחָה עַל מִצְחִי

וְנִלְאָה הַזְּמָן לָשֵׂאת טָרְחִי

וְלֹא יִתְּנֵנִי הָשֵׁב רוּחִי.


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


הָהּ כִּי שְׂעַר שֵׂיבָה בְּרֹאשׁ צִמֵּחַ

צִיר מִיְּמֵי זֹקֶן כְּבָר הִגִּיעַ

צַר לִי בְּחוּרוּתַי וְאֵיךְ אַרְגִּיעַ

וִידֵי תְמוּתָה כּוֹנַנוּ מַטְבֵּחַ.

גַּם הַזְּמָן יָקוּם כְּמוֹ רוֹצֵחַ

מִי זֶה יְשׁוּעָה לִי וְטוֹב יַשְׁמִיעַ

מָה אוֹמְרָה אֵין אִישׁ וְאֵין מַפְגִּיעַ

הָהּ כִּי זְמַנִּי יִהְיֶה נֹצֵחַ.

צַר לִי זְמַנִּי וַאֲנִי נִקְרֵאתִי

רָץ אַחֲרי רָץ יוֹם לְיוֹם יַבְהִלוּ

אֹתִי וּמַה לִי עוֹד בְּחַיַּי חֵפֶץ

אוֹי לִי עֲלוּמִים כִּי אֲנִי נִדְמֵיתִי

אֶרְאֶה לְבָנִים לַעֲלוֹת יַעְפִּלוּ

בָּרָד בְּאַמְתַּחְתָּם וְאֶבֶן נֶפֶץ.


וַיַּעַן הַשּׂר וַיֹּאמַר אֱמָר-נָא נֶטַע נַעֲמָן

הִנֵּה הָיִיתָ רֹאשׁ חֹשְׁקֵי הַזְּמָן

וּבָניתָ לַצְּבִיּוֹת בַּיִת נֶאֱמָן

וְשַׂמְתָּם עַל לִבְּךָ חוֹתָם

אֱמָר-נָא אֵיךְ אַתָּה עַתָּה בְאַהבָתָם

הֲקָצוֹר קָצְרָה יָדְךָ מִפְּדוּת

וְחִלַּלְתָּ נֵזֶר הַחֵשֶׁק וְהָעֵדוּת

וָאַעַן וָאֹמַר הֵסִיר הַזֹּקֶן אֵת אוֹפַן מַרְכְּבֹתַי וַיְנַהֲגֵם בִּכְבֵדוּת

וּמִן הַחֵשֶׁק הָלַכְתִּי קְדֹרַנִּית

וְהַזֹּקֶן עַל גַּבִּי חָרַשׁ הֶאֱרִיךְ מַעֲנִית

וְהֵרִיק אַחֲרַי לַהַב חֶרֶבֶ וּבְרַק חֲנִית


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


יַד הַזֹּקֶן עָלַי הוֹיָה

מַה לִּי דּוֹדִים עוֹד לִצְבִיָּה

רִשְׁפֵי זֹקֶן אֶל תּוֹךְ לִבִּי

הֵם רִשְׁפֵי אֵשׁ שַׁלְהֶבֶתְיָה

צֵל הַשּׁחֲרוּת עֶשֶׂר מַעְלוֹת

הָלַךְ מִי יִּתֵּן וִישַׁעיָה

יַשִׁיב הַצֵּל אָחוֹר כַּאְשֶׁר

הֵשִׁיב אָז בִּימֵי חִזְקִיָּה

עֵינֵי חִשְׁקִי מִשּׂיב קָמוּ

נָדוּ אֶל אֶרֶץ מַאְפֵּליָה

אִם יִרְאוּ עוֹד עֵינַי חִשְׁקִי

אָז יִרְאוּ עֵינֵי צִדְקִיָּה

צָרַעַת זֹקֶן זָרְחָה עַל

מִצְחִי אֵיךְ אֶחְשֹׁק עוֹד רַעְיָה

אִם אַעַל עַל מִזְבַּח חֵשֶׁק

לַקְטִר יַקְטִיר עוֹד עֻזִּיָּה

גַּם לִי עֵרֹם עִם חֹסֶר כֹּל

בָּדָד נֶאְלַמְתִּי דוּמִיָּה

נָפַלְתִּי מֵחוּג שָׁמַיִם

צָלַלְתִּי בִּשְׁאוֹל תַּחְתִּיָּה

אֵין אִישׁ יִמְחֶה מֵעַל פָּנַי

דִּמְעַת עֵינִי הַבּוֹכִיָּה

נֶפֶשׁ צָרָה הֵיךְ מַבְכִּירָה

אֵיכָה תִשְׂחַק בִּימֵי עָנְיָהּ

נֹחַם יִסָּתֵר מֵעֵינַי

כִּי אֵין לִי עֶזרָא וּנְחֶמיָה

וּלְיָרְאֵנִי וּלְבַהְלֵנִי

לִי סַנְבַלַּט עִם נוֹעַדְיָה

מַה לִּי וּלְחֵשֶׁק יַעֲלוֹת-חֵן

וּלְחַבֵּק עוֹד חֵק נָכְרִיָּה

וּפְנֵי יַעְלוֹת אֶחְשֹׁב כַּיּוֹם

כִּפְנֵי פֶרֶס אוֹ עָזְנִיָּה

לוּ תוּקַעְנָה כָּל הַנָּשִׁים

וּתְהִי נִבְלָתָם לַחַיָּה

לֹא אַטֶּה הַשּׂק אֶל הַצּוּר

לֹא אֶהְיֶה רִצְפָּה בַת אַיָּה

אֵיךְ אֶחְשֹׁק עוֹד בִּבְנוֹת הַחֵן

וּתְהִי עָפְרַת חֵן בִּי לוֹיָה

כִּי טָבַעְתִּי בִּיוֵן זֹקֶן

שַׁחְרוּתִי יֻכַּת בִּשְׁאִיָּה

בָּנוֹת נָעְמוּ אֵלַי אֶתְמוֹל

רָאִיתִי בָם שַׁעְרוּרִיָּה

עַד לוּ תָמָר אֶל תּוֹךְ חֶדְרִי

וּלְעֵינַי תַּעַשׂ הַבְּרִיָה

אוֹ בַת-שֶׁבַע מֵעַל הַגַּג

אֶרְאֶנָּה רֹחֶצֶת עֶרְיָה

לֹא אֶתְחַל עַל חֵשֶׁק תָּמָר

לֹא אֶחְשֹׁב רַע עַל אוּרִיָּה.


וַיֹּאמֶר הַשּׂר הֲיָכֹלִתָּ עַל הַצְּבִיּוֹת בִּימֵי בְּחוּרוֹתֶיךָ

וְאִם נִלְכְּדוּ בִמְצוּדוֹתֶיךָ

וָאַעַן וָאֹמַר חַיֶּיךָ הַשּׂר לֹא נִמְלְטָה מִיָּדִי עָפְרָה רְאִיתִיהָ

אִם עָלְתָה הַשּׁמַיְמָה מִשּׁם הוֹרַדְתִּיהָ

וְאִם נֶחְבָּאֲתָה בְקַרְקַע הַיָּם מִשּׁם הֶעֱלִיתִיהָ


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


הָהּ לִזְמַנִּי כָּתַב שִׂטְנָה

עָלַי לֹא הִשְׁאִיר לִי עֶדְנָה

בִּימֵי חָרְפִּי הַשּׁיר עַל פִּי

רָתַם רִכְבּוֹ עַל פִּי חָנָה

כָּל הַשּׁירוֹת מִבּוֹר שִׁירוֹ–

תַי תִּקַּחְנָה וַתְּמַלְאֶנָה

מִרְמַס רַגְלַי הֵם תִּרְעֶינָה

מִרְפַּשׂ רַגְלַי הֵם תִּשְׁתֶּינָה

לוּ הִמְלִיכוּ מֶלֶךְ שִׁירֵי

חֵשֶׁק שִׁירוֹתַי תִּמְלֹכְנָה

הָיִיתִי שַׁעְשׁוּעֵי לִבּוֹת

וּצְבִיּוֹת-חֵן בִּי תָגֵלְנָה

בִּי תִּשְׁכַּחְנָה אֵידָן אֹתִי

עַל רֹאשׁ שִׂמְחָתָן תַּעְלֶינָה

וּבְחַצְרֵיהֶם וּבְטִירֹתָם

הָיִיתִי קָרוּא רִאשֹׁנָה

דֹּדַי עָרְבוּ לִצְבִיּוֹת חֵן

וַיָּבֹאוּ אֶל קִרְבֶּנָה

לֹא שָׂגְבָה עָפְרָה מִמֶּנִּי

לֹא תִבָּצֵר עִיר צִיּוֹנָה

אִם תִּסָּתֵר בַּמִּסְתָּרִים

הֶאָנֹכִי לֹא אֶרְאֶנָּה

לוּ תֵחָבֵא אֶל תּוֹךְ הַיָּם

מִשּׁם יָדִי תִקָּחֶנָּה

אִם תַּגְבִּיהַּ כַּנֶּשֶׁר אוֹ

בֵּין כּוֹכָבִים תָּשִׂים קִנָּהּ

אוֹ כָעֵת בַּמָּרוֹם תַּמְרִיא

מִשָּׁם יָדִי תוֹרִידֶנָּה

אַשְׁפִּילֶנָּה עַד הָאָרֶץ

עַד הֶעָפָר אַגִּיעֶנָּה

אָנֹכִי בַּדֶּרֶךְ וּזְמַ-

נִּי נָחַנִי אֶל תּוֹךְ גַּנָּהּ

לִרְאוֹת הַדּוּדָאִים לִרְעוֹת

בַּגַּנִּים לִלְקֹט שׁוֹשַׁנָּה

שָׁם רָאִיתִי עָפְרָה בָרָה

וּמְקֻטֶּרֶת מֹר וּלְבוֹנָה

יוֹם צִיֵּר חוֹנֵן הַצּוּרוֹת

אֹתָהּ מִכָּל עֵשֶׂק פָּנָה

לֹא הוֹלִיד עַל תֵּבֵל כִּדְמוּ-

תָהּ וּכְצַלְמָהּ עוֹד לֹא-שָׁנָה

יוֹם הֻלֶּדֶת הַנִּכְבֶּדֶת

כָּל הֶהָרִים פָּצְחוּ רִנָּה

רֹאשָׁהּ עָלֶיהָ כַּכַּרְמֶל

וּשְׂפָתֶיהָ מוֹר תִּטֹּפְנָה

וּמְאוֹרֵי אוֹר בַּשּׁמַיִם

קֹדְרִים מוּל זֹהַר יִפְעָתָהּ

שָׁרְרָהּ אַגַּן סַהַר לֹא יֶחְ-

סַר וַעֲרֵמַת חִטִּים בִּטְנָהּ

וּלְחָיֶיהָ כָּל הּאוֹרִים

בָּם תִּזְרַחְנָה בָּם תַּעְרֹבְנָה

וּבְנִרְדָּם לֵב נִרְדָּם יֵעוֹר

וּבְנִיר-דָּם כָּאוֹר תָאֹרְנָה

עֵינֶיהָ לַהַט הַחֶרֶב

תִּתְהַפֵּכְנָה אֶל כָּל פִּנָּה

עֶשְׁתּנֹתַי וּמְזִמּוֹתַי

גַּם עֵינַי נִכחָן תַּבֵּטְנָה

מִשְׁתָּאוֹת מַחְשׁוֹת לָדַעַת

אִם תִּגְרֹשְׁנָה אוֹ תִקְרֶאנָה

לֹא הִבְחַנְתִּי אַךְ הֶאְמַנְתִּי

כִּי הֵן אֹמְרוֹת לֵכְנָה שֹׁבְנָה

חַמּוֹתִי רָאִיתִי אוֹר כִּלְ-

יוֹתַי כָּאֵשׁ כֵּן תִּבְעַרְנָה

לֹא יָדַעְתִּי אִם אֵשׁ חִשְׁקָהּ

הוּא אֶת נַפְשִׁי יַלְהִיבֶנָּה

אוֹ מִמָּרוֹם שֻׁלַּח בִּי אֵשׁ

עַל עַצְמוֹתַי וַיִּרְדֶּנָּה

אָמַרְתִּי מַה-יָפִית עָפְרָה

יִפְעַת הוֹדֵךְ מִי זֶה מָנָה

אִם עֵינַיִךְ לִי תִמְחַצְנָה

יָדַיִךְ אֹתִי תִרְפֶּינָה

חוּסִי עָלַי וּתְנִי-נָא לִי

מִדּוּדָאֵי חִשְׁקֵךְ יוֹנָה

כִּשְׁמֹעַ עָפְרָה שָׂשׂ לִבָּהּ

גַּם עֵינֶיהָ בִּי תִצְפֶּינָה

אָמְרָה בִּשְׁתַּיִם תִּתְחַתֵּן

בִּי הַיּוֹם נֵזֶר הַבִּינָה

אֶחְלֹק גּוּפִי לִשְׁתֵי מָנוֹת

וּבְחַר-לָךְ אַחַת מֵהֵנָּה

הָאֵזוֹר הוּא יִגְבּוֹל אֹתָם

בֵּין עֶלְיוֹנָה לַתַּחְתּוֹנָה

הַעֲמֵק שְׁאָלָה אוֹ הַגְבֵּהַּ

כִּי מָנָתְךָ לָךְ אֶתֵּנָה

עֵת שָׁמַעְתִּי לֹא יָדַעְתִּי

אִם אַשְׂמִילָה אִם אֵימִנָה

עֵינַי יָזֹקּוּ מִטְרוֹת עֹז

וַתָּבֹאנָה וַתִּדְלֶנָה

אָז אָמַרְתִּי נִגְזָרְתִּי כִּי

נַפְשִׁי כַּיּוֹם לָקְחָה בָשְׁנָה

לֹא יָדַעְתִּי בִּלְתִּי הַיּוֹם

נַפְשִׁי עַד כֹּה לֹא הֵבִינָה

כִּי עָפְרָה יַד חִשְׁקָהּ תִּקְפֹּץ

וּבְאִישׁ חִשְׁקָהּ תֵּרַע עֵינָהּ

וּנְקַלֹּתִי עוֹד מִזֹּאת כִּי

כָאֵלֶּה אֹתִי תִקְרֶאנָה

אַךְ לִכְבוֹדֵךְ עָפְרָה אֶקַּח

לִי הַמָּנָה הָעֶלְיוֹנָה

כִּשְׁמֹעַ זֹאת עָפְרַת הַחֵן

אָז כִּלְיוֹתֶיהָ תַעְלֹזְנָה

וַתִּשּׁבַע חֵי הַחֵשֶׁק

חֵי חִשְׁקֶךָ לִבִּי עִנָּה

יַעַן הָיְתָה-זֹאת עִם-לִבְּךָ

כִּי לִנְדִיבוּת לִבְּךָ פָּנָה

חַי אָנִי הָאֵזוֹר תַּחַת

כַּפּוֹת רַגְלַי שָׁם אָכִינָה

וּמָשַׁלְתָּ בַכֹּל וּשְׁתֵּי

הַמַּמְלָכוֹת לָךְ תִּהְיֶינָה

אָז הָלַכְתִּי עִם עָפְרַת-חֵן

עַד בָּאתִי אֶל אַפִּרְיוֹנָהּ

אָכַלְתִּי יַעְרָהּ אָרִיתִי

מוֹרָהּ גַּם שָׁתִיתִי יֵינָהּ

וּזְמַנִּי עַתָּה הֶלְאָנִי

וּצְבִיּוֹת חֵן לִי תִבְזֶינָה

לֹא תַחְשֹׁקְנָה בִּי וּבְקוֹלִי

לֹא תִשְׁמַעְנָה לֹא תֹאבֶינָה

אִם אָמַרְתִּי לַחְשֹׁק עָפְרָה

חֹשֵׁק אַחֵר יִשְכָּבֶנָה

לֹא תַחְפֹּצְנָה מִיָּדִי זֶ-

בַח עוֹלָתִי לֹא תִרְצֶינָה

אֶצְעַק חָמָס לִצְבִיּוֹת כִי

כִגְמוּל יָדִי לֹא תִגְמֹלְנָה

לֹא נָשִׁיתִי לֹא נָשׁוּ בִי

כִּי רָעָה עָלַי תַּחְרֹשְׁנָה

אֶתְמַהּ עַל כִּי תָקֹצְנָה בִּי

כִּי טָמֵא וּכְסִיל תִּשְׂכֹּרְנָה

לֹא נֶחְלוּ עַל שִׁבְרִי כַּיּוֹם

אִלּוּ מַתִּי לֹא תִבְכֶינָה

לוּ נִמְכַּרְתִּי בִקְשִׂיטָה אֶל–

עֵקֶר אֹתִי לֹא תִפְדֶינָה

לוּ אֶהְיֶה פֶּטֶר לַחְמוֹרָם

לֹא תִפְדֻּנִי לִי תַעְרֹפְנָה

כִּי הִלְבִּין שַׂעְרִי וּזְקָנִי

מִזָּהָב אֶל כֶּסֶף שֻׁנָּה

כִּי יָצָא קֶצֶף הַשֵּׂיבָה

גַּם הֵחֵל נֶגֶף הַזִּקְנָה.


כִּשְׁמֹעַ הַשּׂר תָּמַהּ עַל יְפִי הַשּׁירָה

וַיֹּאמַר חֵי אֲנִי לֹא יִשָּׁקֵל כֶּסֶף מְחִירָהּ

וְלוּ שָׁמַע אוֹתָהּ אָסָף

אֶל מַעֲלָתְךָ נִכְסָף

וּלְדַבֵּר שִׁיר לֹא יָסָף

בְּרוּכָה תִהְיֶה הַזִּקְנָה

אֲשֶׁר סִבְּבָה לְחַבֵּר זֹאת הַשּׁירָה הָעֲדִינָה


וַיִּשּׂא מְשָׁלוֹ וַיֹּאמַר:


מַה לִּידוּתוּן וּלְדָוִד שִׁיר

לִבְּךָ עַל כָּל שָׁר שִׁיר יָסָף

לוּ הִתְעָרֵב אָסָף עִמְּךָ

אָז יִהְיֶה הַמַּמְעִיט אָסָף


וָאַעַן וָאֹמַר אֶל הַשּׂר הַדֹּבֵר בִּי

מַדּוּעַ בֵּרַכְתָּ הַזֹּקֶן וְהוּא אֹיְבִי

וְאִם בַּעֲבוּר הַשּׁיר אֲשֶׁר עָלָה בְלִבִּי

אִם אֲדַבְּרָה לֹא יֵחָשֵׂךְ כְּאֵבִי

כִּי קוֹל הַשּׁיר לִי קוֹל עַנּוֹת

כִּי סֻגְּרוּ דְלָתַיִם בַּשּׁיר וּבָטְלוּ הַטֹּחֲנוֹת

וְאֵין חוֹר פִּי כַּאֲשֶׁר הָיָה בִּימֵי חָרְפִּי

וְעִם הַזֹּקֶן יוֹמִי הוּא נִשְׁפִּי

וְרִישׁ פִּי רִשְׁפִּי

וְעַל כִּי לֹא הֵקִיץ פִּי קִצְפִּי


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


צַר לִי זֹקֶן מִי יִתֶּן-לִי

כַּאְשֶׁר בִּימֵי חָרְפִּי חֹר-פִּי

אֶקְרָא אֶל פִּי לֹא יַעֲנֵנִי

עַל כִּי לֹא הֵקִיץ פִּי קִצְפִּי

אִמְרִי שֵׂיבָה מַה לִּי וָלָךְ

כִּי תַחְרִיבִי הַר-פִּי הַרְפִּי

מָה אֶצְטַדֵּק מָה אוֹחִיל אֵיךְ

אֶמְצָא אִם יִשְתָּק-פִּי תָּקְפִּי

הָיָה טוּבִי רַב אַךְ לִבִּי

מִיּוֹם שָׁב לֶאְנֹק-פִּי נֹקְפִּי

הָיָה יוֹמִי יוֹם אֵל עִמִּי

אַךְ מֵעֵת הֶעְנִישׁ-פִּי נִשְׁפִּי

הָיוּ רִשְׁפֵי תֵבֵל הֶבֶל

וָרִיק אָכֵן רִישׁ-פִּי רִשְׁפִּי

קַלּוּ שִׁבְרֵי תֵבֵל עָלַי

לֹא יִכְבָּדוּ אַךְ-פִּי אַכְפִּי

הָהּ שָׁכַחְתִּי אָכֹל לַחְמִי

עַד כִּי יִשְאַל אָן-פִּי אָנְפִּי

יִשְׂנָא לִבִּי אֹסֶף שָׂדַי

מִיּוֹם שָׁב לִמְאָס-פִּי אָסְפִּי

לֹא הִבְחַנְתִּי לֹא הֶאְמַנְתִּי

יוֹם הָיָה אֶל כּוֹס-פִּי כָּסְפִּי

כִּי יַד זֹקֶן תַּכְאִיבֵנִי

אוֹ אֶפְנֶה אֶל עוֹר-פִּי עָרְפִּי

הָיָה אֶתְמוֹל לִבִּי עָלֵז

וּלְמרָחוֹק תָּר-פִּי טַרְפִּי

שָׁלֵו הָיִיתִי בִּשְׁלוֹמוֹ

אָכֵן מִיּוֹם זָע-פִּי זַעְפִּי

צַר לִי נִכְאָב בִּבְחוּרוֹתַי

מַדּוּעַ לֹא כָס-פִּי כַּסְפִּי

אֵיךְ הִשְׁאַרְתִּי קִנְיַן כֶּסֶף

אֵיךְ לֹא בָא אָז עַל-פִּי אַלְפִּי

יֹאמַר זֹקֶן מַה תִּרְדֹּף לִי

עַל כִּי אָץ לִמְרָד-פִּי רָדְפִי

זֹקֶן אֶרְדֹּף וּבְצַד אֶהְדֹּף

עַל כִּי לָקַח הוֹד-פִּי הָדְפִי

נַפְשִׁי עוּרִי לִקְרַאת עָרִי

זֹקֶן הַמַּחְרִיד-פִּי רִדְפִי

זֹקֶן אֹיְבֵךְ עַל כֵּן חַרְבֵּךְ

עָלָיו כִּי הִכְשִׁיל-פִּי שִׁלְפִי

לֹא תֵאָסֵף אֶל תַּעְרֵךְ אַךְ

עֹרֶף הַמַּבְעִיר-פִּי עִרְפִי

רוּחַ קָדִים קוּמִי וּפְחִי

זֹקֶן לֹא יַשְׁקִיט-פִּי קִטְפִי

אֵשׁ אֹכֶלֶת וּמְלַהֶטֶת

זֹקֶן כִּי הֶחְסִיר-פִּי שִׂרְפִי

שִׁבֹּלֶת מַיִם וּמְצוּלָה

זֹקֶן כִּי הִפְשִׁיט-פִּי שִׁטְפִי

חַיָּה רָעָה וּמְשַׁכֶּלֶת

זֹקֶן כִּי יַחִיט-פִּי חִטְפִי

חֶשְׁכַת יוֹם צָרָה וּמְבוּכָה

זֹקֶן כִּי הִמְעִיט-פִּי עִטְפִי

אִם יֹאמְרוּ רֵעָי תַּחַת מֶה

לַזֹּקֶן יוֹשִׁיט-פִּי שִׁטְפִּי

זֶה לַזֹּקֶן תַּחַת כִּי לֹא–

יִתֵּן כִּימֵי חָרְפִּי חֹרְ-פִּי.


וַיֹּאמֶר הַשָּׂר:

צָדַקְתָּ בְּכָל אֲשֶׁר אָמַרְתָּ מִן הַזִּקְנָה

אַחֲרֵי אֲשֶׁר בָּהּ יִסְתַּבֵּל הֶחָגָב וְתָפֵר הָאֲבִיּוֹנָה

וְאֵין אַחֲרֵי בְלוֹת הָאִישׁ עֶדְנָה

וְסֻגְּרוּ דְלָתַיִם בַּשּׁוּק בִּשְׁפַל קוֹל הַטַּחֲנָה

וְהַזְּקֵנִים בְּצָרוֹת הַזְּמָן יִלָּבֵטוּ

וּבָטְלוּ הַטֹּחֲנוֹת כִּי מִעֵטוּ

וָאַעַן וָאֹמַר לֹא זֶה הַדֶּרֶך וְלֹא זֹה הָעִיר

כִּי הַזֹּקֶן אֵשׁ בְּלִבִּי יַבְעִיר

בַּעֲבוּר הַמָּוֶת הַבָּא אַחֲרָיו

וְהֵצַר לָאִישׁ בְּכָל שְׁעָרָיו

וְאֵין שָׂרִיד בִּמְגוּרָיו

וַאֲנִי הַגֶּבֶר

הַדָּבָר אֲשֶׁר מֵעוֹדִי הֶחֱרִידַנִי הוּא הַקֶּבֶר

אֲשֶׁר הוּא שֹׁד עַל שֹׁד וְשֶׁבֶר עַל שֶׁבֶר

וּבִרְאוֹתִי נִבְלוֹת הָרוֹזְנִים

הַקּוֹפְאִים עַל שִׁמְרֵיהֶם וְהַשּׁאֲנַנִּים

הָיוּ לְפָנִים

דְּשֵׁנִים וְרַעֲנַנִּים

וְיָבְשׁוּ עַצְמוֹתֵיהֶם

וְעָבשׁוּ פְרֻדוֹת תַּחַת מֶגְרְפֹתֵיהֶם

וְהִתְנַצְּלוּ עֶדְיָם וּמְעִילֵיהֶם

וּנְתוּנִים בְּיַד גֹּזְלֵי עוֹרָם מֵעֲלֵיהֶם

וּשְׁאֵרָם מֵעַל עַצְמוּתֵיהֶם

וַתְּהִי נִבְלָתָם כַּסּוּחָה

לֹא תֶעֱרַב לִי כָּל שִׂמְחָה

ואֵיךְ נַעֲלֹז וּבֵית מוֹעֵד לְכָל חָי לֶעָשִׁיר וְלַהֵלֶךְ

וְהַתְּמוּתָה אֶל הַכֹּל תֵּלֵךְ

וְגַם לְמָחָר אֲנִי קָרוּא לָהּ עִם הַמֶלֶךְ


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


מִי בַזְּמָן הִבְהִילְךָ יִשְׁאָלוּ

וָאֹמְרָה דּוֹדִים אֲנִי הַגֶּבֶר

לֹא הֶחֱרִידַנִי לְבַד הַקֶּבֶר

עֵת אֶזְכְּרֵהוּ כָּל שְׂשׂוֹנַי גָּלוּ.

כִּי אֶזְכְּרָה רַבִּים אֲשֶׁר צָלָלוּ

תּוֹךְ מֵי תְהֹמוֹתָיו וְאִישׁ אֶל עֵבֶר

פָּנָיו יְהַלֵּכוּן וְשֹׁד אֶל שֶׁבֶר

יִרְוּו בְּכֹסוֹת לַעֲנָה הָרְעָלוּ.

אֶרְאֶה נְבֵלוֹת רוֹזְנִים עָבָשׁוּ

הַקֹּדְרִים וּלְפִי שְׁאוֹל נִפְזָרוּ

יֹאמְרוּ עֲצָמֵינוּ אֲשֶׁר יָבָשׁוּ

גַּאְוָה תְמוֹל לָבְשׁוּ וְהַיּוֹם זָרוּ

הִתְעַתְּדוּ חַיִּים וְהִתְקַדָּשׁוּ

לִהְיוֹת אֲשֶׁר נִהְיוּ וְהִתְאַזָּרוּ.


וַיֹּאמֶר הַשּׂר מֵאֲשֶׁר רַעְיוֹנֶיךָ מִן הַמָּוֶת פֹחֲדִים וְרוֹהִים

עָמַדְנוּ תְמֵהִים

בַּאֲשֶׁר הַצַּדִּיקִים וְהַחֲסִידִים אַחֲרֵי יְיָ נִנְהִים

וְאֶל מֵי יְשׁוּעָתוֹ צְמֵאִים וּכְמֵהִים

יֹאמְרוּ מָתַי אָבוֹא וְאֵרָאֶה פְּנֵי אֱלֹהִים

וָאַעַן וָאֹמַר צָדְקְתָ מִן הַחֲסִידִים

אֲשֶׁר לִדְבַר יְיָ חֲרֵדִים

בְּיַעַן לִקְצֹר פְּרִי מַעַלְלֵיהֶם עֲתִידִים

אָמְנָם אֲנִי אֵיךְ לֹא אֶחֱרַד

וָאֶרְאֶה יוֹמִי מְאֹד רַד

וְהִנְנִי מִמַּאֲוַיַּי נִפְרָד

מְעֻתָּד אֶל הַנְּסִיעָה הַגְּדוֹלָה

וְאֵין צֵידָה לְנַפְשִׁי הָאֲמֻלָה

וּמָה אֲדַבֵּר וּמָה אֶצְטַדָּק וְהִנֵּה אִלּוּ אֲנָשִׁים יֵלְכוּ דֶרֶךְ יוֹם אוֹ יוֹמָיִם

יִמַּס לְבָבָם וִיהִי לְמָיִם

יֵלְכוּ וְיִצְטַיָּדוּ

וּלְמַלֹּאת אַמְתְּחֹתֵיהֶם אֹכֶל כַּאֲשֶׁר יוּכְלוּן שְׂאֵת יִתְעַתָּדוּ

וְלִי דֶרֶךְ לִמאֹד רָחוֹק

וִיצוּרַי עָבְרוּ תוֹרוֹת חָלְפוּ חֹק

וְאֵין לֶחֶם וְאֵין צֵידָה

לָשֵׂאת לְנַפְשִׁי הַשּׁדוּדָה


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


אִלּוּ אֲנָשִׁים יֵלְכוּ אֶל-עֵבֶר

לַיָּם יְכִינוּן הוֹן וְיִצְטַיָּדוּ

יִשְׂאוּ לְפִי אָכְלָם וְיִתְעַתָּדוּ

וּלְחוֹף אֳנִיּוֹת יַחֲלִיפוּ אֵבֶר.

צַר לִי וּמָחָר אֵלְכָה אֶל קֶבֶר

תָּנוּד יְחִידָתִי לְמָקוֹם נָדוּ

נַפְשׁוֹת אֱוִילִים פֹּשׁעִים בָּגָדוּ

אֵין הוֹן וְאֵין צֵידָה וְאֵין לִי שֶׁבֶר.

אֵי הוֹן וְאֵי צֵידָה אֲשֶר הֵכַנְתִּי

אֶל-מַעְבְּרוֹת מָוֶת אֲשֶׁר אֶעְבֹרָה

אֵי עֵדוֹת אֵלִי אֲשֶׁר הֶאְמַנְתִּי

אַיֵּה בְרִית הָאֵל אֲשֶׁר אֶצֹּרָה

אַיֵּה חֲסָדִים בַּעֲדָם נִשְׁעַנְתִּי

מִי זֶה זְרָעָם וַאֲנִי אֶקְצֹרָה.


וְאֵיךְ לֹא אֵבוֹשׁ וְלֹא אֶכָּלֵם

וְלֹא יִפְּלוּ פָנַי אֲנִי הַמִּתְעַלֵּם

וְהִנֵּה אִם רָץ לִקְרַאת רָץ וּמַגִּיד לִקְרַאת מַגִּיד

יְבוֹאֵנִי מִפְּאַת שַׂר אוֹ נָגִיד

וְאוֹתִי בְמִשְׁפָּט יָבִיא

יִנָּתְקוּ מוֹרָשֵׁי לְבָבִי

וִינִיעֵנִי עַל-בִּרְכַּי וְכַפּוֹת רַגְלָי

תְּסַמֵּר שַׂעֲרַת בְּשָׂרִי וְאֵימוֹת מָוֶת יִפְּלוּ עָלָי

וְאֵיךְ לֹא אֵחֱרַד וְאָנֹכִי אִישׁ תְּכָכִים

וְנִקְרֵאתִי לִפְנֵי מֶלֶךְ מַלְכֵי הַמְּלָכִים

וְיִזְכְּרֵנִי וְנִשְׁפְּטָה יָחַד

הֲלֹא יֹאחֲזֵנִי פָחַד


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


אוֹיָה לְנפְשִׁי מַה מְּאֹד יָרֵאתִי

אִלּוּ אֲנִי קָרוּא לְמִשְׁפַּט-מֶלֶךְ

יִזַּל כְּטַל דִּמְעִי וְרוּחִי תֵלֵךְ

נַפְשִׁי אֲזַי לַמָּוְתָה הֶעְרֵיתִי.

צַר לִי עֲמִיתַי וַאֲנִי נִקְרֵאתִי

לִפְנֵי אֲדוֹן עוֹלָם רְאוּ אֵיךְ הֵלֶךְ

מֵאֵין חֲסָדִים קִדְּמָם הוּא יֵלֵךְ

טוֹב לִי אֲנִי עוֹד שָׁם וְלֹא נִבְרֵאתִי.

אוֹיָה יְחִידָתִי וְאֵיךְ לֹא-תַחְשְׁבִי

מָחָר אֲשֶׁר תִּסְעִי רְאִי אָן תַּחֲנִי

יוֹם שֶׁיְּדֻבַּר בָּךְ רְאִי מַה-תַּעֲנִי.

מִכֵּס רְדִי נַפְשִׁי וְעַל עָפָר שְׁבִי

עֶרֶשׂ רְבַדְתּוֹ לָךְ תְּמוֹל שָׁם תִּשְׁכְּנִי

מַצְרֵף וכוּר לָאֵל וּבוֹ תִבָּחֲנִי.


וַיֹּאמֶר הַשָּר אַחֲרֵי אֲשֶׁר כָּל מַעֲשֵׂה אַנְשֵׁי הָעוֹלָם

הָאֱלֹהִים יָבוֹא בְּמִשְׁפָּט עַל כָּל נֶעְלָם

וְעַל טוֹב וְעַל רָע גְּמוּל יְשֻׁלָּם

וְיֵדַע זֶה הָאָדָם הַנִּכְלָם

תָּמֵהָּ אֲנִי אֵיךְ יֶחֱטָא אִישׁ לְעוֹלָם

וָאַעַן וָאֹמַר חַיֶּיךָ הַשּׂר אֵין זֶה רַק הַנָּחָשׁ יַשּׂיאֵם

וְלֶאֱכֹל מֵעֵץ הַדַּעַת טוֹב וָרָע יְבִיאֵם

וְכַאֲשֶׁר סַעֲרַת יְיָ תִּזְרֵם וְרוּחַ תִּשּׂאֵם

יִתְפַּלְאוּ עַל רָעָה תְבוֹאֵם

וְיַעַמְדוּ בְעַד תַּגְמוּל מַעֲשֵׂיהֶם תְּמֵהִים

וְיֹאמְרוּ מַה זֹאת עָשָׂה לָנוּ אֱלֹהִים


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


יֵצֶר לְבָבֵנוּ בְרִשְׁעוֹ אַחֲרֵי

עֵינָיו יְהַלֵּךְ הוּא אֲשֶׁר כִּלָּנוּ

נִפֹּל בְּמַכְמֹרָיו וְאַחַר נֹאמְרָה

מַה זֹאת אֲשֶׁר עָשָׂה אֱלֹהִים לָנוּ.


וַתִּקְטַן זֹאת בְעֵינֵי הַיֵּצֶר אֹיְבֵנוּ

וְהֶעְלִים עֵינֵינוּ מֵרְאוֹת דִּבְרֵי נְבִיאֵינוּ

וְאֵיךְ יַעֲמֹד לִבֵּנוּ וְתֶחֱזַקְנָה יָדֵינוּ

וְאִלּוּ מִפְּאַת מֶלֶךְ אִגֶּרֶת תְבוֹאֵנוּ

נֶחֱרַד לְמוּלָהּ וְנָעִיז חֲכָמֵינוּ וּנְבוֹנֵינוּ לִפְתָּר לָנוּ הָאִגֶּרֶת

לָדַעַת הַדְּבָרִים אֲשֶׁר הִיא אֹגֶרֶת

וְלֹא נִשְׁקֹט עַד תִּהְיֶה לָנוּ פְתוּרָה

וְנֵדַע מִסְפַּר הָאִגֶּרֶת וְאֶת שִׁבְרָהּ

וְאַחֲרֵי זֹאת יָשִימוּ כָל בְּנֵי אָדָם

כָּל לְבָבָם וְכָל נַפְשָׁם וְכָל מְאֹדָם

לְמַלֹּאת מִצְוַת מַלְכָּם וְאִם הוּא בָשָׂר וָדָם

וְאֵיךְ לֹא נִכָּלֵם אֲנַחְנוּ הַמִּתְעַלְּמִים

כְּלָבִים אִלְּמִים

אֲשֶׁר מֶלֶךְ מַלְכֵי הַמְּלָכִים קְרָאָנוּ בְיַד עֲבָדָיו הַנְּבִיאִים

וּבְיָדָם מִצְוֹת וְחֻקִּים נִפְלָאִים

וּמִשְׁפּטִים צַדִּיקִים וְנוֹרָאִים

יָעַד טוֹב לַצַּדִּיקִים

וְאָרוּר לַאֲשֶׁר לֹא יָקִים

וְאֵין קוֹרֵא בִשְׁמָם לָדַעַת נִיבָם

וּמִתְעוֹרֵר לְהַחֲזִיק בָּם


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


אֱלֹהֵינוּ קְרָאוּנוּ לְעָבְדוֹ

נְבִיאֵי יָהּ וְהַבֵּט אֶל בְּרִיתוֹ

בְּיָדָם מִכְתְּבֵי חֻקִּים וְתוֹרוֹת

אֲשֶׁר צִוָּה בְסִינַי לַעֲדָתוֹ

וְאָז חָרְדוּ לְמֻלָם הַלְּבָבוֹת

וְאֵין נֵעוֹר כְּהַיּוֹם מִשּׁנָתוֹ

וְלֹא נֵבוֹשׁ וְצִירִים יֹאחֲזוּנוּ

בְּהִשּׁמַע דְּבַר מֶלֶךְ וְדָתוֹ.


וּמִן הַתֵּמַהּ מִכָּל אֲשֶׁר קָדַם

הוּא שֶׁהִטְבִּיעַ הָאֵל יִתְעַלֶּה טְבָעִים וְכֹחוֹת בָּאָדָם

קְצָתָם יְדַעְנוּם וּקְצָתָם נֶעלַם מִמֶּנּוּ סוֹדָם

וְצִוָּה הַטְּבָעִים וְהַכֹּחוֹת הַנִּזְכָּרִים

לִהְיוֹת אֶל מִשְׁמַעַת הָאָדָם סָרִים

וְצִוָּה אֶת הָאָדָם לַעֲבֹד אוֹתוֹ

וְלַחֲקֹר כְּפִי הַיְכֹלֶת מְצִיאוּתוֹ

וְהָיוּ יְצֻרוֹתֵנוּ וְכֹחוֹתֵנוּ

שׁוֹמְרִים מִצוֹת אֱלֹהֵינוּ

יִמְצָאֵם כָּל אִישׁ בְּצֵאתוּ וּבְבוֹאוֹ

נְכוֹנִים אִישׁ אִישׁ עַל עֲבוֹדָתוֹ וְעַל מַשּׂאוֹ

וַאֲנַחְנוּ נִהְיֶה מִצְוֹת אֱלֹהֵינוּ מַמְרִים

וּמֵרְאוֹת מִצְוֹתָיו וְחֻקּוֹתָיו עִוְרִים

וְהִנֵּה אִם יַמְרוּ כֹחוֹתֵינוּ וִיצֻרֵינוּ בְמַה שּׁצֻּוּוּ לַעֲבֹד אוֹתָנוּ

כַּאֲשֶׁר מָרַדְנוּ בְמַה שּׁצֻוֵּינוּ לַעֲבֹד אָנוּ

וְיִמְרְדוּ בָנוּ כַּאֲשֶר מָרָדְנוּ

הֵן גָּוַעְנוּ אָבַדְנוּ כֻלָּנוּ אָבָדְנוּ


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


צֻוּוּ יְצִירוֹתַי עֲבֹד אֹתִי וְצֻ-

וֵּיתִי עֲבֹד אֵל חַי וְלֹא עָבָדְתִּי

אִם כֹּה יְצֻרַי יִמְעֲלוּ אָבָדְתִּי

אֶנְאַץ אֲנִי מוּסָר וְהֵם לא נָאֲצוּ.

אֵיךְ אֶפְרְצָה גָדֵר והֵם לֹא פרְצוּ

לֹא בָגְדוּ בִי וַאֲנִי בָגָדְתִּי

אוֹיָה לְנַפְשִׁי וַאֲנִי רָבַדְתִּי

עַרְשָׂהּ שְׁאוֹל וּמְזַעְזְעֶיהָ יִקְצוּ.

נַפְשִׁי קְשַׁת רוּחַ הֲלֹא תִכָּלְמִי

כִּי כָל יְצֻרַי שָׁמְרוּ מִשְׁמֶרֶת

מִצְוַת אֱלֹהַיִךְ וְאַתְּ מֵאַנְתְּ.

הַשּׁאֲפָה רוּחַ הֲלֹא תִתְנַחֲמִי

מַה-תַּעֲשִׂי רִשְׁעָה כְאֵשׁ בֹּעֶרֶת

יוֹם בֹּא חֲבָלִים לָךְ וּמַה-נֵּחָנְתְּ.


וְאֵיךְ לֹא יִפְּלוּ פָנֵינוּ

וְהִנֵּה אִם אֱנוֹשׁ כְּעֶרְכֵּנוּ

יִשְמֹר תְּנוּעָתֵנוּ

וִימַשּׁשׁ אָהֳלֵנוּ

וְיִתְחַקֶּה עַל שָׁרְשֵׁי רַגְלֵינוּ

נַעְצֹר בְּרוּחֵנוּ

וְנַיְשִׁיר פְּעֻלּוֹתֵינוּ

בְּכָל כֹּחֵנוּ

וְאִם אָוֶן בְּיָדֵנוּ נַרְחִיקֵהוּ

וְאִם יַמְתִּיק בְּפִינוּ דָּבָר רָע בְתוֹךְ חִכֵּנוּ נִמְנָעֵהוּ

וְלֹא יֵבוֹשׁוֹ הַבּוֹגְדִים מֵאֱלהֵיהֶם

הַיֹּצֵר יַחַד לִבָּם הַמֵּבִין אֶל כָּל מַעֲשֵׂיהֶם

וְאֵיךְ לֹא נִפְקַח עֵינֵינוּ לַחֲזוֹת

כִּי הָאֱלֹהִים עָשָׂה לָנוּ אֶת הַנֶּפֶשׁ הַזֹּאת

וְנָתַן בָנוּ חָכְמָה וּתְבוּנָה וָדָעַת

וְעַיִן רוֹאָה וְאֹזֶן שֹׁמָעַת

וְאֵיךְ כָּל לֵב לֹא יֶחֱרַד וְכָל עַיִן לֹא יִדְמָע

הַיֹּצֵר עַיִן הֲלֹא יַבִּיט וְאִם הֲנֹטַע אֹזֶן הֲלֹא יִשְׁמָע

אִם נִסָּתֵר בַּמִּסְתָּרִים וְלֹא יִרְאֵנוּ

וְאֵיךְ לֹא נֵבוֹשׁ מִבּוֹרְאֵנוּ

וְהִנֵּה נַעְלִים רֹעַ מַעֲלָלֵינוּ

מֵאֲנָשִׁים אֲשֶׁר כְּגִילֵנוּ

וְלֹא נֵבוֹשׁ מִן הַמַּשְׁקִיף סְתָרֵינוּ וְנִגְלֵינוּ

הַבּוֹחֵן לבּוֹתֵינוּ וְכִלְיוֹתֵינוּ


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


תַּרְתִּי בְלִבִּי אִם יְלוּד אִשּׁה קְצַר

יָמִים שְׂבַע רֹגֶז יְמַשּׁשׁ אָהֳלִי

יִשְׁמֹר תְּנוּעוֹתַי וְיַשְׁקִיף פָּעֳלִי

אִוָסְרָה אֶתֵּן לְרוּחִי מַעְצָר.

אַיְשִׁיר פְּעֻלּוֹתַי וְאֶהְיֶה נֶעְצָר

אִם בִּי יְהִי אָוֶן וְנֶגַע אוֹ חֳלִי

אֶכְתּוֹשׁ גְּאוֹן לִבִּי בְּרִיפוֹת בַּעֱלִי

אָאִיר פְּנֵי נַפְשִׁי כְּמוֹ אוֹר וָצָר.

מָה אוֹמְרָה נִכְאָב וְיֹצֵר עַיִן

בֹּחֵן לְכִלְיוֹתַי וְיָדוֹ רִקְּמָה

גָלְמִי וְכָל רֵעַי בְּסִפְרוֹ נִכְתְּבוּ.

הוֹצִיא יְסוֹד עוֹלָם לְיֵשׁ מֵאַיִן

מַשְׁקִיף סְתָרֵינוֹ וְאֵיךְ נִתְחַכְּמָה

לַסְתִּר עֲצָתֵנוּ וְדִינָיו נִצְּבוּ.


וַיַּעַן הַשּׂר וַיֹּאמַר חידוֹתֶיךָ וְנִפְלְאוֹתֶיךָ הִקְשַׁבְתִּי

וְאֵין הַזִּקְנָה דָבָר רָע כַּאֲשֶׁר חָשַׁבְתִּי

וַאֲשֶׁר הִגִּיעַ אֵלֶיהָ אַשְׁרָיו

כִּי הִיא תְבִיאֶנּוּ לִרְאוֹת בּמַה-שּׁיִּהיֶה אַחֲרָיו

וָאֹמַר אֶל הַשּׂר חַיֶּיךָ כֵּן דִּבַּרְתָּ

כַּאֲשֶׁר הַזִּקְנָה פֵאַרְתָּ

כִּי הַמַּגִּיעַ אֵלֶיהָ תִּגְדַּל מַעֲלָתוֹ

וְיִהְיֶה חָפְשִׁי מִמַּשּׂא הַזְּמָן וַעֲבוֹדָתוֹ

וְיִכְבֹּשׁ אֶת רֶסֶן תַּאֲוָתוֹ

וְאַל יֹאמַר הַזָקֵן לֹא אֶהְיֶה חֹבֵשׁ

הֵן אֲנִי עֵץ יָבֵשׁ

כִּי בִּהְיוֹתוֹ רֶסֶן תַּאֲוָתוֹ כֹבֵשׁ

הוּא עֲדִי הַשּׂכֶל לֹבֵשׁ


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


יוֹם נֵעֹרְתִּי מִן הַיַּלְדוּת

אָז לָבַשְׁתִּי בִּגְדֵי חֹפֶשׁ

כִּי רָאִיתִי אֶת יַד שִׂכְלִי

כֹּבֶשֶׁת רֶסֶן הַנָּפֶשׁ.


וַיֹּאמֶר הַשּׂר הִשְׁמַעְתַּנִי מִדְּבַר הַזֹּקֶן נוֹרָאוֹת

ומִן הַקֶּבֶר וְהַמִשְׁפָּט נִפְלָאוֹת

עַד אֲשֶׁר אֶל מַעֲלַת הַחֲרֵדִים אֶל דְּבַר הָאֵל יִתְעַלֶּה הִגַּעְתַּנִי

אָמְנָם הָעֹנֶשׁ אֲשֶׁר תִּהְיֶה לַנֶּפֶשׁ הַמְקַצֶּרֶת לֹא הוֹדַעְתַּנִי

וְהִנְנִי לָדַעַת דְּבַר הַמִּשְׁפָּט נִכְסָף

אִם סוֹדוֹ לָאִישׁ נֶחְשָׂף

וָאַעַן וָאֹמַר יֵדַע אֲדֹנִי הַנַּעֲלָה

כִּי סוֹד הַמִּשְפָּט לְאִישׁ לֹא נִגְלָה

פֶּן מִמּוֹרָאוֹ כָּל אִישׁ מֵרְאוֹת טוֹב יִתְיָאֵשׁ

אָמְנָם יֹאמְרוּ הַחֲכָמִים בַּעֲדוֹ כִּי הוּא אֵשׁ אוֹכְלָה אֵשׁ

וְנִמְנְעוּ מִלְּתָאֲרֵהוּ

פֶּן עֵת נְסַפְּרֵהוּ

לְדָבָר מָךְ מֵעֶרְכּוֹ נַעֲרִיכֵהוּ

וְכַאֲשֶׁר נַחְשֹׁב לְהַגְדִּיל עִנְיָנוֹ נַפְחִיתֵהוּ

וְלָכֵן יֵשׁ לִגְעֹר בְּתַאֲוֹות הַזְּמָן

אֲשֶׁר יַפְרִיעוּ הָעָם מִמַּעֲשֵׂה עֲבוֹדַת הָאֵל הַנֶּאֱמָן

וְנִצְטַיְּדָה יִרְאָה וְנַרְגִּיעַ

וְנִרְכַּב מַרְכְּבוֹת הַמַּעֲשִׂים טוֹבִים וְאֶל מְחוֹז חֶפְצֵנוּ נַגִּיעַ


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


דָּבָר שְׁמַעְתִּיו יוֹם לְיוֹם יַבִּיעַ

יֹאמְרוּ אֲשֶׁר תֹּפֶת בְתוֹכוֹ לַהֲבוֹת

אֵשׁ אֹכְלָה רִשְׁעָה וְנַפְשׁוֹת שׁוֹבֲבוֹת

אֵין שָׂר וּמֵלִיץ בַּעֲדָם יפְגִּיעַ.

מַה-לִּי וְלָךְ יִצְרִי אֲשֶׁר תַּפְרִיעַ

מִמַּעֲשַׂי אֹתִי בְּהַבְלֵי עָרְבוֹת

אֶצְטַיְּדָה יִרְאָה וְאֶרְכַּב מַרְכְּבוֹת

הַמַּעֲשִׂים טוֹבִים וְאָז אַרְגִּיעַ.

מִי בַזְּמָן אֹתִי יְהֵא מַבְטִיחַ

קָטֹן וְגָדוֹל עִם-כְּסִיל וּבַעַר

בָּא יוֹם חֲלִיפָתָם וְסָפוּ תָמּוּ

אֵיכָה תְהֹמוֹת הַזְּמָן אָגִיחַ

אֶל פִּי וְלֹא אַשְׂכִּיל כְּבַהְמוֹת יַעַר

לִיסוֹד נְאוֹת שֵׂכֶל אֲשֶׁר נָשָׁמּוּ.


וַיּאֹמֶר הַשּׂר הִנֵּה חִבַּרְתָּ שִׁירִים

בְּעִנְיַן הַחֵשֶׁק בִּימֵי הַנְּעוּרִים

וְשִׁירֵי חֲסִידִים וְקִינִים

בְּעֵת הֵחֵלוּ לִצְמֹחַ יְמֵי הַזְּקֻנִים

וְעַתָּה אֱמָר נָא הַאִם חִבַּרְתָּ שִׁירִים בְּדִבְרֵי חָכְמָה

וּבְמִצְוֹת הַתּוֹרָה התְּמִימָה

וָאַעַן וָאֹמַר חִבַּרְתִּי

וְעַל שְׁלֹשׁ עֶשְרֵה עִקָּרֵי הָאֱמוּנָה שִׁיר אָמַרְתִּי

וְהוּא בְּעִנְיָנָיו מְאֹד נִפְלָא

אֲשֶׁר בְּכֶתֶם אוֹפִיר לֹא יְסֻלָּא

וַיִּתְמַהּ הַשּׂר וַיֹּאמַר וְאֵיךְ יָכֹלְתָּ לְחַבֵּר עִנְיָנִים נִפְרָדִים

וּלְתִתָּם בְּשִׁיר אֶחָד נִצְמָדִים

וָאַעַן וָאֹמַר יָכֹלְתִּי וַיִּהְיוּ בְיָדִי לַאֲחָדִים

וּבְמַעֲרֶכֶת הַשּׁיר רוֹכְבִים צְמָדִים

וְהִנֵּה לְךָ הַשּׁיר קוּם הַשּׂר וּשְׁמַע עִנְיָנָיו

וַעֲנֹד עַל גַּרְגְּרוֹתֶיךָ פְּנִינָיו


וָאֶשּׂא מְשָׁלִי וָאֹמַר:


אֶפְתַּח בְכִנּוֹר עַל עֲלָמוֹת שִׁיר

אָשִׂים יְסוֹד הַדָּת רְפִידָתוֹ

בָּנוּי לְתַלְפִּיּוֹת בְּטוּרֵי בִין

עַל הַרֲרֵי קֹדֶשׁ יְסוּדָתוֹ

גַּלְגַּל תְבוּנוֹת רוֹם בְרֹב דֵּעָה

הוֹלֵךְ וְסוֹבֵב עַל נְקֻדָּתוֹ

בּוֹ כּוֹכְבֵי חָכְמָה יְאִירוּן עַל

תַּשְׁבֵּץ תְּבוּנָתוֹ וְחִידָתוֹ

אֶחְבֹשׁ לְרֹאשׁוֹ מִגְבְּעֹת שֵׂכֶל

סוֹד הָאֱמוּן חֵשֶׁב אֲפֻדָּתוֹ

אָשִׂים סְבִיבוֹ תוֹכְחוֹת מוּסָר

אַף זֵר זָהָב סָבִיב לְמִסְגַּרְתּוֹ

בּוֹ הָאֱמוּנָה תִהְיֶה שֹׁרָה

הַחֵן וְהַצֶּדֶק גְבֻלָתוֹ

עַל אָפנָיו דָּבֻר וְתַפּוּחֵי

זָהָב סְגוּרִים הֵם בְּמַשְׂכִּיתוֹ

יִחְיוּ לְרֵיחוֹ שֹׁכְנֵי עָפָר

עֵת יִזְּלוּ בָשְׂמֵי עֲרוּגָתוֹ

טוּרָיו שְׁנַיִם הֵם עֲלֵי שִׁבְעִים

עַל רִצֲפַת כָּבוֹד מְנֻחָתוֹ

בּוֹ מוֹסְדוֹת הַדָּת שְלֹשׁ עֶשְׂרֵה

הִנָּם יְצֻקִים עִם יְצֻקָתוֹ

אַחְלַי וְתַחְנוּנַי יְהִי עִמָּ-

נוּ אֵל וְרוּחוֹ בַּאֲרִיגָּתוֹ

קִרְבוּ פְּתָאִים לַחֲמוּ לַחְמִי

הִתְבּוֹנְנוּ שִׁירִי וְתוֹכַחְתּוֹ

הַשִּׁיר יְהִי לָכֶם כְּלֵיל הִתְקַ-

דֶּשׁ חַג וְשִׂמְחַת לֵב כְּשִׂמְחָתוֹ

כִּי מֹצְאוֹ הוּא יִמְצְאָה חַיִּים

אַךְ חֹטְאוֹ חֹמֵס יְחִידָתוֹ

עִמּוֹ הֲלֹא עֵצָה וְתוּשִׁיָּה

בּוֹ תּוֹכְחוֹת חַיִּים לְהוֹרֹתוֹ

תַּרְתִּי בְלִבִּי לַחֲקֹר סוֹד אֵל

אֶתְבּוֹנְנָה רַעַם גְּבוּרָתוֹ

עַד כִּי אֲשֶׁר בָּאתִי לְמִקְדָּשׁוֹ

אַשְׂכִּיל וְאָבִינָה לְאַחְרִיתוֹ

וַאְחַשּׁבָה וָאֵדְעָה כִּי הוּא

עִלַּת מְצִיאוּת כֹּל וְסִבָּתוֹ

נִמְצָא אֲבָל לֹא מִזְּמָן מָגְבָּל

רֹכֵב שְׁמֵי קֶדֶם בְּגַאְוָתוֹ

מִי בַשּׁחָקִים יַעֲרֹךְ אֵלָיו

יִדְמֶה לְשֵׁם קָדְשׁוֹ וְתִפְאַרְתּוֹ

נִקְרָא שְׁמוֹ שַׁדַּי וּמִסְתַּפֵּק

כִּי לֹא יְהִי נִצְרָךְ לְזוּלָתוֹ

יִבְטַל מְצִיאוּת כֹּל אֲשֶׁר נִמְצָא

אִם נַחֲשֹׁב בִּטּוּל מְצִיאוּתוֹ

אִם יַעֲלֶה עַל לֵב אֱנוֹשׁ בִּטּוּל

זֶה הַמְּצִיאוּת מִתְּכוּנָתוֹ

לֹא נַאֲמִין עִם זֶה וְלֹא יָשְׂכַּל

חֶסְרוֹן בְּצַד מָה מִשּׁלֵמוּתוֹ

אֶחָד וְיִחוּדוֹ מְאֹד נֶעְלָם

מִי יַעֲמֹד עַל סוֹד אֲמִתָּתוֹ

אֶחָד וְלֹא כַמִּין אֲשֶׁר יִכְלֹל

תַּחְתָּיו פְּרָטִים בַאֲגֻדָּתוֹ

אֶחָד וְלֹא כַגּוּף אֲשֶׁר נֶחְלָק

הוּא לַחֲלָקִים בַּחֲלֻקָתוֹ

לֹא גוּף וְלֹא כֹחַ בְּתוֹךְ הַגּוּף

אָכֵן מְיֻחָד הוּא בְאַחְדוּתוֹ

כָּל תָּאֳרִים תִּמְצָא יְתֹאַר בָּם

הֵם תָּאֳרִים מוֹרִים פְּעֻלָּתוֹ

קַדְמוֹן וְאֵין נִמְצָא כְקַדְמוּתוֹ

רִאשׁוֹן וְאֵין רֵאשִׁית לְרֵאשִׁיתוֹ

לֹא נִמְצְאָה מָצוּי בְּשֵׁם קַדְמוֹן

אִם נַעֲרִיכֵהוּ לְקַדְמוּתוֹ

אָדוֹן לְכָל נִמְצָא אֲשֶׁר הִמְצִיא

שֹׁפֵט יְצוּרָיו בַּאֲדוֹנוּתוֹ

כָּל הַמְּצוּיִם מִלְּבַדּוֹ הֵם

מִתְפָּעֲלִים סָרִים לְמִשְׁמַעְתּוֹ

חָשִׁים וּמָגְבָּלִים וּמָכְרָחִים

הֵם לַעֲבֹד אִישׁ אִישׁ עֲבוֹדָתוֹ

רִשְיוֹן וְחֵפֶץ לֹא יְהִי בָהֶם

אָכֵן רְצוֹנוֹ עִם גְּזֵרָתוֹ

כָּל מִי אֲשֶׁר יִפנֶה לְזוּלָתוֹ

יַמְרֶה רְצוֹן קוֹנוֹ וּמִצְוָתוֹ

צֹפֶה וְיֹדֵעַ סְתָרֵינוּ

סֹעֵד יְצוּרָיו בֶּאֱלֹהוּתוֹ

אֶרֶץ וְשָׁמַיִם וְצִבְאוֹתָם

לֹא יַעֲזֹב אֹתָם לְגַבְהוּתוֹ

גֹּמֵל לְאִישׁ צַדִּיק כְּצִדְקָתוֹ

יָשִׁיב לְאִישׁ רָשָׁע כְּרִשְׁעָתוֹ

הֵכִין וּפָעַל זָהֳרֵי חָכְמָה

שֶׁפַע נְבוּאַת עַם סְגֻלָּתוֹ

הִשְׁקִיף בְּאוֹר שֵׂכֶל וְזָרַח עַל

מֹשֶׁה נְבִיאוֹ נֶאֱמַן בֵּיתוֹ

נָבִיא כְמֹשֶׁה לֹא קְדָמָהוּ

גַּם אַחֲרָיו לֹא קָם כְּמַעֲלָתוֹ

אִם יִהְיֶה נָבִיא תְהִי עַל יַד

מַלְאָךְ וְאֶמְצָעִי נְבוּאָתוֹ

אָכֵן נְבוּאַת הַמְחוֹקֵק פֶּה

אֶל פֶּה וּמַבִּיט אֶת תְּמוּנָתוֹ

נָבִיא נְבוּאָה לֹא תְבוֹאֵהוּ

רַק בַּחֲלוֹם לַיִל לְחָלְשָׁתוֹ

אוֹ בַהֲקִיצוֹ אִם יְהִי נִבָּא

יִתְבַּטְּלוּ חוּשָׁיו כְּשָׁמְעָתוֹ

אָכֵן נְבוּאַת נֶאֱמַן בַּיִת

יוֹמָם וְהוּא עֹמֵד בְּעֶמְדָּתוֹ

כָּל אִישׁ יְהִי נִבָּא אֲזַי יִמַּס

לִבּוֹ וְיִתְרוֹפֵף לְחֶרְדָּתוֹ

אָכֵן נְבוּאַת נֶאֱמַן בַּיִת

עָלָיו וְהוּא נִצָּב בְּחֶזְקָתוֹ

כָּל הַנְּבִיאִים נִבְּאִים הֵמָּה

בִּרְצוֹן אֱלֹהִים חַי וְדִבְרָתוֹ

אָכֵן נְבוּאַת הַמְחֹקֵק הִיא

תָּמִיד בְּכָל עִתִּים וּבִרְצוֹתוֹ

תּוֹרַת אֱמֶת הָיְתָה בְּמוֹ פִיהוּ

וּבְרִית אֱלֹהִים הָיְתָה אִתּוֹ

תּוֹרָה אֲשֶׁר הִיא תַעמֹד נֶצַח

וּבְרִית לְעַד עוֹלָם עֲמִידָתוֹ

וּבבֹא לְקֵץ יָמִין מְשִׁיחֵנוּ

לֹא יַחֲלִיף דָת אֵל וְאִמְרָתוֹ

שָׂפָה בְרוּרָה יַהֲפֹךְ אֶל עָם

לֹא יֵדְעוּ אֵל חַי וְסוֹד דָּתוֹ

יִשְׁתַּחֲווּ אַף יִקְּדוּ עַמִּים

אָז לַהֲדֹם רַגְלֵי שְׁכִינָתוֹ

יָשִׁיב נְוֵה צֶדֶק וְהַר קֹדֶשׁ

צִיּוֹן כְּלִיל יֹפִי לְקַדְמוּתוֹ

יָצִיץ וּפָרַח יַעֲקֹב עַבְדּוֹ

יִחְיוּ וְיָקֻמוּ נְבֵלָתוֹ

אַל תִּירְאוּ הֹלְכֵי קְדֹרַנִּית

הַשּׁמְרִים חֻקָּיו וּמִשְׁמַרְתּוֹ

שׁוּבוּ לְבִצָּרוֹן אֲסִירֵי הַ-

תִּקְוָה מְחַכֵּי קֵץ גְּאְלָּתוֹ

וּלְכוּ חֲזוּ בֵּין עֹבְדֵי אֵל-חַי

וּבֵין אֲשֶׁר לֹא יַעֲבֹד אֹתוֹ

יוֹם יִהְיוּ זֵדִים וְכָל עֹשֵׂה

רִשְׁעָה הֲדוּפִים אָז בְּעֶבְרָתוֹ

יִהְיוּ כְקַשׁ וּכְמֹץ לְיוֹם הַבָּא

וּתְלַהֲטֵם אִשּׁוֹ וְלַבָּתוֹ

יִזְרַח עֲלֵיכֶם אָז יְרֵאֵי אֵל

שֶׁמֶשׁ צְדָקָה אוֹר יְשׁוּעָתוֹ

יִגְדַּל וְיִשְׁתַּבַּח אֲדוֹן עוֹלָם

בָּרוּךְ עֲדֵי עַד שֵׁם קְדֻשּׁתוֹ

הַצּוּר נְצָחָנוּ נְעִימוּתוֹ

הַנֶּעֱלָם סוֹדוֹ וְעַצְמוּתוֹ

יוֹם נִכְסְפוּ לִבּוֹת לְהַגְדִּילוֹ

וּלְרוֹמְמוֹ אִישׁ אִישׁ כְּבִרְכָתוֹ

תָּשׁוּב תְּהִלַּת הַלְּשׁוֹנוֹת אָז

לֵאוּת וְקִצּוּר מִגְּדֻלָּתוֹ

וּבְעֵת אֲשֶׁר יִשְׁתַּדְּלוּ לִבּוֹת

לַחְקֹר יְסוֹד הוֹדוֹ וּמָהוּתוֹ

תָּשׁוּב יְדִיעָתָם אֲזַי סִכְלוּת

תִקְצַר יְדֵיהֶם מִיְּדִיעָתוֹ

לָכֵן בְּשִׁירִי אֶשְׁמְרָה אֶל פִּי

מַחְסוֹם וְדֻמִיָּה תְהִלָּתוֹ.


וַיֹּאמֶר הַשּׂר רָאִיתִי קוֹל רַעַמְךָ בְּגַלְגַּל הַחֲרָזִים

שֹׁבֵר אֲרָזִים

וְכָל אִישׁ בִּלְתְּךָ מִידִידוּת יִתְיָאֵשׁ

כִּי קוֹל אַהֲבָתְךָ חֹצֵב לַהֲבוֹת אֵשׁ

וְהִנְנִי נִשְׁבָּע בַּאֲשֶׁר עָשָׂה לָנוּ אֶת הַנֶּפֶשׁ הַזֹּאת

כִּי אֵין עֲרוֹךְ אֵלֶיךָ בִמְלִיצָה וּבַחֲרוּזוֹת

בָּרוּךְ אֲשֶׁר הֱבִיאֲךָ עַד הֲלוֹם

כִּי בִלְעָדֶיךָ נַפְשִׁי תִּזְנַח מִשּׁלוֹם

וְעַתָּה הִשּׁבְעָה לִי רֹאשׁ שִׂמְחָתִי

אִם תִּשְׁקֹר בְּאַהֲבָתִי

וְאִם תְּעַנֶּה עוֹד בִּנְדוֹדְךָ לְבָבִי

וָאַעַן וָאֹמַר יֵדַע אֲדֹנִי נֵר הַמַּעֲרָבִי

כִּי אַהֲבָתְךָ מִקֶּדֶם חֲמַדְתִיהָ

וְעַל גַּרְגְּרוֹתַי עֲנָק עֲנַדְתִּיהָ

וְאַחֲרֵי אֲשֶׁר מְצָאתִיהָ

פֹּה אֵשֵׁב כִּי אִוִּיתִיהָ

וָאֵשֵׁב עִם הַשּׂר וָאֶאֱרֶה מוֹרוֹ וַעֲדָנָיו אֶאֱכֹל

בְּשִׂמְחָה וּבְטוּב לֵבָב מֵרֹב כֹּל.

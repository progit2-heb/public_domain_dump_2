מחברות עמנואל – הַמַּחְבֶּרֶת הָעֶשְׂרִים וּשְׁתַיִם

אָמַר הַמְחַבֵּר: הָיִיתִי עִם הַשָּׂר הַמְאֻשָּׁר חֶמְדַת הַלְּבָבוֹת • וּמִדֵּי הֱיוֹתִי עִמּוֹ חָשַׁבְתִּי לִהְיוֹת בְּגַלְגַּל עֲרָבוֹת • כִּי הוּא בְחָכְמֹתָיו הַמַּאֲלִיפוֹת וְהַמְרֻבָּבוֹת • וּבְמִדּוֹתָיו הָעֲרֵבוֹת וְהַנְּדִיבוֹת • בִּזֵּר עַמִּים יַחְפְּצוּ קְרָבוֹת • וַתְּהִי אַהבֲתָנֵוּ יוֹם יוֹם נוֹסֶפֶת • וְעַל כָּל אַהֲבַת חֹשְׁקִים עוֹדֶפֶת • וַיְהִי הַיּוֹם בְּצֵאתֵנוּ הַשַּׁעֲרָה עֲלֵי קָרֶת • וְהִנֵּה שָׁם אֲנָשִׁים נִכְבָּדִים לְכָבוֹד וּלְתִפְאָרֶת • אָמְנָם מִקָּצָה בַּחוֹבָרֶת • נִקְבְּצוּ שָׁם לַחַיָּה אַנְשֵׁי כְסִילוּת • נָטְלוּ לְחֶלְקָם תִּשְׁעָה קַבִּין מִן הַהוֹלֵלוּת • וְאֶחָד נָטַל כָּל הָעוֹלָם • עַד אֲשֶׁר לְחֶסְרוֹן שִׂכְלָם • לֹא יֵדְעוּ בֵין יְמִינָם לִשְׂמֹאלָם • נָתַן הַזְּמָן לָהֶם הַפְּתַיּוּת לְמוֹרָשָׁה • וּשְׂדֵה הַסִּכְלוּת וּמִגְרָשָׁהּ • וְלָהֶם צוּרַת אֲנָשִׁים וְשֵׂכֶל אִשָּׁה • וַיְהִי בִּרְאוֹתָם אוֹתָנוּ חָשְׁבוּ כִּי דָבָר גָּדוֹל מָצָאוּ • וּמִכָּל הַמְּקוֹמוֹת אֲשֶׁר נָפוֹצוּ שָׁם נִצְבָּאוּ • וּמִן הַחֹרִים אֲשֶׁר שָׁם נֶחְבָּאוּ • לִלקֹט פְּנִינֵי מַאַמְרוֹתֵינוּ יָצָאוּ • וַיֹּאמְרוּ אֵלֵינוּ אֲדֹנֵינוּ בֹּאֲכֶם יִהְיֶה לְשָׁלוֹם • בָּרוּך יִהְיֶה הַמִּקְרֶה אֲשֶׁר הֵבִיא אֶתְכֶם הֲלוֹם • וְהַיָּמִים הָרִאשׁוֹנִים יִפְּלוּ כִּי מַה שֶּׁעָבַר חֲלוֹם • הִנֵּה אֲנַחְנוּ בְּעִמְקֵי הַסְּפֵקוֹת • וְכִכְלָבִים אִלְּמִים בְּסֶלַע הַמַּחֲלוֹקוֹת • וְיֶשׁ-לָנוּ מִן הַקֻּשְׁיוֹת יוֹתֵר מִדַּי • וְהַמִּתְקַדְּשִׁים לֹא הִתְקַדְּשׁוּ לְמַדַּי • וְנֶאֱמַר לָנוּ כִּי לָכֶם לכָל קֻשְׁיָא תֵּרוּץ • וְטוֹב פִּרְיְכֶם מִפָּז וּמֵחָרוּץ • וְהָעֹמֵד מְרֻבֶּה עַל הַפָּרוּץ • וּבִזְמַנְכֶם הָאֱמוּנָה נֶחְפָּה בַכָּסֶף וְאֶבְרוֹתֶיהָ בִּירַקְרַק חָרוּץ • וְעַתָּה קְחוּ נָא אֶת בִּרְכָתֵנוּ • וְהָסִירוּ מִלִּבֵּנוּ אֶת מְבוּכוֹתֵינוּ • כִּי לָנוּ שְׁאֵלוֹת עֲצוּמוֹת • וְלָכֶם נִגְלוּ כָּל תַּעֲלוּמוֹת.

כִּשְׁמֹעַ הַשָּׂר עָלַז לִבּוֹ • וַיֹּאמֶר אֵלָי מַה נִּכְבָּד זֶה הַיּוֹם וּמָה רַב טוּבוֹ • נָגִילָה וְנִשְמְחָה בוֹ • הָבָה נִתְחַכְּמָה לָהֶם וְנִשְׁמַח בְּמַפַּלְתָּם • בּעֲנוֹתֵנוּ לַכְּסִילִים כְּאוַּלְתָּם • וַיִּפֶן הַשָּׂר אֶל הָאֲנָשִׁים • וַיֹּאמַר אַשְׁרֵיכֶם זֶרַע קְדוֹשִׁים • אַשְׁרֵיכֶם וְאַשְרֵי יוֹלַדְתְּכֶם • זֶה דוֹדֵי וְזֶה רֵעִי יִגְאַל גְּאֻלַּתְכֶם • וְיָשִׂים שָׁלוֹם פְּקֻדַּתְכֶם • פְּגַעְתֶּם שָׂשׂ וְעוֹשֵׂה צֶדֶק • הַמְחַזֵּק בְּשִׂכְלוֹ כָּל בֶּדֶק • הוּא יִהְיֶה לָכַם לִישׁוּעָה • וְיָשִׂים הֶעָקֹב לְמִישׁוֹר וְהָרְכָסִים לְבִקְעָה • אִם תִּשְׁאֲלוּ טַל יַעֲרֹף כַּמָּטָר לִקְחוֹ • זָרַח נֵר הַתְּבוּנָה בְמִצְחוֹ • וְאִם שִׂכְלְכֶם נָחָשׁ הִנּוֹ בְרִיחוֹ • אֲדֹנָי צְבָאוֹת שְׁלָחָהוּ וְרוּחוּ • וְעִמּוֹ צְרִי גִלְעָד וּקְטָף מִירֵחוֹ • יֵלְכוּ הַמְפָרְשִׁים לְאוֹרוֹ וְהַגְּאוֹנִים לְנֹגַהּ זַרְחוֹ • וְעַתָּה הָאֲצִילִים וְהַשּׁוֹעִים • אִם לִי אַתֶּם וּבְקוֹלִי אַתֶּם שׁוֹמְעִים • לֹא תִשְׁאֲלוּ אוֹת וְלֹא תְנַסּוּ • מִמֶּנִּי תִרְאוּ וְכֵן תַּעֲשׂוּ • וַיִּפְשֹׁט הַשָּׂר אַדֶּרֶת כְּבוֹדוֹ • וַיְמַהֲרוּ הָאֲנָשִׁים וַיִּקְחוּ אִישׁ אֶת בִּגְדוֹ • וַיָּשִׂימוּ תַחְתָּיו אֶל גֶּרֶם הַמַּעֲלוֹת • וַיִּתְקְעוּ בַשּׁוֹפָר וַיֹּאמְרוּ מָלַךְ מְתָרֵץ הַשְּׁאֵלוֹת • וַיְהִי בִּרְאוֹתִי כִּי שָׂמוּ אוֹתיִ בִּמְרוֹם הַתְּהִלּוֹת • וָאֹמַר אֶל הַקְּהִלּוֹת • הִתְקַבְּצוּ וְשִׁמְעוּ אַנְשֵׁי הַשְּׁאֵלוֹת הַמְהֻלָּלוֹת • כִּי לִי קוֹל בְּעִנְיַן הַתֵּרוּצִים יְחוֹלֵל אַיָּלוֹת • אֶחֱשֹׂף יַעֲרוֹת הַסּוֹדוֹת בִּמְשָׁלָי • וּמִי בַּעַל דְּבָרֵים יִגַּשׁ אֵלָי.

וַיִּגַּשׁ הָרִאשׁוֹן וַיֹּאמַר הִנְנִי נָבוֹךְ בְּכָל אָרְחוֹתַי • כִּי מָצָאתִי כָתוּב הֵפֶךְ מִמַּה שֶּׁקִּבַּלְתִּי מֵאֲבוֹתַי • וְהֶאֱמַנְתִּיו כָּל יְמוֹתַי • כִּי אֲבוֹתַי אָמְרוּ בְאָזְנַי • כִּי הַתּוֹרָה נִתְּנָה מִסִּינַי • וּבָאֵשׁ יָרָד עָלָיו אֲדֹנָי • וּמוֹכִיחִים זֶה מִמַּאֲמַר ‘וְהַר סִינַי עָשַׁן’ • וַאֲנִי מָצָאתִי כָתוּב בְּסֵפֶר מְדֻקְדָּק יָשָׁן • ‘וְהַדָּת נִתְּנָה בְּשׁוּשַׁן’ • וָאַעַן וָאֹמַר חַיֶּיךָ בְּסִינַי נִתְּנָה הַתּוֹרָה • וְהַכָּתוּב הָאוֹמֵר ‘והַדָּת נִתְּנָה בְּשׁוֹשַׁן הַבִּירָה’ • רוֹצֶה לוֹמַר כִּי הַתּוֹרָה נְתָנָהּ הָאֵל לְעַם קְרוֹבוֹ • בְּחֹדֶשׁ סִיוָן שֶׁהַשּׁוֹשַׁנִּים נִמְצָאִים בּוֹ • וְלָכֵן נִתְּנָה בְּאוֹתוֹ הַחֹדֶשׁ • לְפָאֵר בָּהֶם הֵיכַל הַקֹּדֶשׁ • וַיֹּאמֶר הָאִישׁ אַשְׁרַי שֶׁכָּכָה הִשְמַעְתַּנִי • אֶמְצָא חֵן בְּעֵינֶיךָ אֲדֹנִי כִּי נִחַמְתָּנִי.

וַיַּרְא הַשֵּׁנִי כִּי טוֹב פָּתַרְתִּי • וּלְהָשִׁיב אֶת רֵעֵהוּ מִיַּם הַמְּבוּכָה אֶל יַבָּשֶׁת הַמְּנוּחָה חָתַרְתִּי • וַיֹּאמֶר אֵלָי אֲדֹנִי רָאִיתִי תְשׁוּבוֹתֶיךָ נְכֹחוֹת • וּבְפָנֶיךָ שֹׂבַע שְׂמָחוֹת • וְעַתָּה הוֹצִיאֵנִי מִזֶּה הַצַּעַר • אֲשֶׁר רַעְיוֹנַי עָלָיו אָחֲזוּ שָׂעַר • מַדּוּעַ אָמַר הַפָּסוּק ‘חֲנוֹךְ לַנַּעַר’ • וְלֹא אָמַר אַרְפַּכְשָׁד לַנָּעַר • אוֹ מְתוּשֶׁלַח • וָאַעַן וָאֹמַר מְתוּשֶׁלַח • וְאַרְפַּכְשָׁד בָּאוּ אֱלֵי קֶבֶר בְּכֶלַח • וְהָיְתָה לָהֶם עִם הַזְּמָן בְּרִית מֶלַח • וּמִי שֶׁהָיָה קוֹרֵא אוֹתָם בְּשֵׁם נָעַר • הָיָה כְסִיל וָבָעַר • אָמְנָם חֲנוֹךְ בְּשֵׁם נַעַר יִקְרָאֶנּוּ • בַּעֲבוּר ‘וַיִּתְהַלֵּךְ חֲנוֹךְ אֶת הָאֱלֹהִים וְאֵינֶנּוּ’ • וַיֹּאמֶר הָאִישׁ הִנְנִי נִשְׁבָּע בְּחַיֵּי אֲבוֹתַי • כִּי עַל דָּבָר זֶה נִצְטָעַרְתִּי כָּל יְמוֹתַי • וְלֹא יָכֹלְתִּי לַעֲמֹד בְּסוֹדוֹ • בָּרוּךְ שֶׁחָלַק לְךָ מִכְּבוֹדוֹ.

וַיִּגַּשׁ הַשְּׁלִישִׁי וַיֹּאמַר הִנְנִי תוֹעֶה בְּפָסוּק אֶחָד כְּשִׁכּוֹר • אֵיךְ הַכָּתוּב בְּשִׁבְחֵי הָאִשָּׁה יִזְכֹּר • ‘סָדִין עָשְׂתָה וַתִּמְכֹּר’ • וְיָדוּעַ כִּי בְעֵת תִּמְכֹּר הָאִשָּׁה נִכְסֵי בֵיתָהּ • יוֹרֶה עַל עֶצֶם רָעָתָהּ • וָאַעַן וָאֹמַר הַסָּדִין בַּמָקוֹם הַזֶה מַטְלִית • וְהַחֲסִידָה תִמְכֹּר אוֹתוֹ לְרַבָּנִים לַעֲשׂוֹת מִמֶּנּוּ טַלִּית.

וַיִּגַּשׁ הָרְבִיעִי וַיֹּאמַר הוֹשִׁיעָה אֲדֹנִי אֲבִי הַמּוֹעַצה • אֵיךְ יֹאמַר הַפָּסוּק ‘רַק בְזָדוֹן יִתֵּן מַצָּה’ • וְהִנֵּה אֲנַחְנוּ רוֹאִים אֲשֶׁר בֵּין אַנְשֵׁי הַחָכְמָה לֹא יִתְחַשָּׁב • כַּגֵּר כַּתּוֹשָׁב • נוֹתֵן מַצָּה בְּפָנִים יָפוֹת לְכָל עוֹבֵר וָשָׁב • וָאַעַן וָאֹמַר הַסְכֵּת וּשְׁמַע מִלַּי • זֶה הַפָּסוּק אָמוּר עַל הָאִישׁ הַכִּילַי • אֲשֶׁר יִתֵּן הַמַּצָּה לֶעָנִי בְּפָנִים רָעוֹת • וּבְשִׂיחוֹת גְּרוּעוֹת.

וַיִּגַּשׁ הַחֲמִשִּׁי וַיֹּאמַר אֱמָר‑נָא שַׂר הָעֶדְנָה • אֵיךְ תְּפָרֵשׁ ‘וּפוֹתֶה תָּמִית כִּנָּה’ • כִּי מַה פֱּתַיּוּת יֵשׁ בַּהֲרִיגַת הַכִּנִּים • הַמִּתְאַכְזְרִים עָלֵינוּ כַיְעֵנִים • וָאַעַן וָאֹמַר דַּע אֲדֹנִי הַנָּבָר • כִּי שְלֹשָׁה פֵרוּשִׁים בַּדָּבָר • הָרִאשׁוֹן הוּא פֵּרוּשׁ רַבֵּנוּ פִּינְחָס מִתַּחְפְּנֵס • שֶׁפֵּרַשׁ כִּי פּוֹתֶה הוּא הַמֵּמִית כִּנָּה בִּפְנֵי אַסְתְּנֵס • וְהַשֵּׁנִי פֵּרוּשׁ רַבֵּנוּ יִצְחָק מֵרַבַּת • שֶׁפֵּרַשׁ כִּי פּוֹתֶה הוּא הַמֵּמִית כִּנָּה בְשַׁבָּת • וְהַשְּׁלִישִׁי פֵּרְשׁוּ הַגְּאוֹנִים • שֶׁאָמְרוּ פּוֹתֶה הוּא מִי שֶהֱמִיתוּהוּ הַכִּנִּים.

וַיִּגַּשׁ הַשִּׁשִּׁי וַיֹּאמַר אֲדֹנִי אֵיךְ צִוְּתָה הַתּוֹרָה ‘אַל תִּשְׁכַּח עֲנָבִים’ • וְלֹא הִזְכִּירָה הַתְּאֵנִים הָעֲרֵבִים • וְהָרִמּוֹנִים הַמְתוּקִים • וְהַחַבּוּשִׁים וְהָאֲפַרְסְקִים • וָאַעַן וָאֹמַר הִזְכִּירָה הַתּוֹרָה הָעֲנָבִים מפְּנֵי שֶׁנִּמְצָא בָהֶם דְּבַר חִדּוּשׁ • כִּי עִם הַיַּיִן הַיּוֹצֵא מֵהֶם נַעֲשֶׂה הַבְדָּלָה וְקִדּוּשׁ • וְהֵם הַפֵּרוֹת הַמְהֻלָּלִים • בַּעֲבוּר הֱיוֹתָם מְשַׁלְשְׁלִים.

וַיִּגַּשׁ הַשְּׁבִיעִי וַיֹּאמַר אֵיךְ אָמַר שְׁלֹמֹה ‘אָמַרְתִּי אֶעֱלֶה בְתָמָר’ • מַדּוּעַ יַעֲלֶה הוּא בָאִילָן וּלְעַבְדּוֹ הַדָּבָר לֹא יֹאמַר • וָאַעַן וָאֹמַר הֶחָכָם שְׁלֹמֹה רָצָה לַעֲשׂוֹת הַמִּצְוָה בְּעַצְמוֹ וְלַעֲלוֹת עַל עֵץ הַתַּמָר הִתְאַוָּה • לַחֲתֹךְ לוּלָב לְמִצְוָה • כִּי הוּא לַעֲשׂוֹת הַמִּצְוָה כְּאַיָּל יַעֲרוֹג • וְכֵן אוֹתִיּוֹת אֶעֱלֶה הֲדָס לוּלָב עֲרָבָה אֶתְרוֹג.

וַיִּגַּשׁ הַשְּׁמִינִי וַיֹּאמַר לָמָּה הִזְכִּיר הַכָּתוּב ‘הַלְלוּהוּ בְּצִלְצְלֵי תְרוּעָה’ • וְלֹא אָמַר הַלְלוּהוּ בְצִלְצְלֵי שְׁלֹשָׁה שְׁבָרִים וּתְקִיעָה • וָאַעַן וָאֹמַר כּבָר פֵּרְשׁוּ הַמּוֹרִים • כִּי בְאָמְרוֹ ‘בְּתֵקַע שׁוֹפָר’ רָמַז תְּקִיעָה וּשְׁלֹשָׁה שׁבָרִים.

וַיִּגַּשׁ הַתְּשִׁיעִי וַיֹּאמַר מַה מְּאֹד עָלַי קָשֶׁה • אֵיךְ לֹא אָמַר הַכָּתוּב ‘עַל כֵּן לֹא יֹאכְלוּ בְנוֹת יִשְׂרָאֵל אֶת גִּיד הַנָּשֶׁה’ • וַאֲנִי לָמַדְתִּי בְּמַסֶּכֶת כְּתֻבּוֹת • כִּי הוּא אָסוּר לַזְּכָרִים וְלַנְּקֵבוֹת • וָאַעַן וָאֹמַר חַיֶּיךָ אִלּוּ הִזְכִּיר כֵּן הַכָּתוּב הָיִינוּ לִשְׁנִינָה וּלְמָשָׁל • כִּי הָיוּ אוֹסְרוֹת אוֹתוֹ עֲלֵיהֶן חַי וּמְבֻשָּׁל.

וַיִּגַּשׁ הָעֲשִׂירִי וַיֹּאמַר אֱמָר‑נָא פֶּלֶא יוֹעֵץ • מָה רָצָה הַכָּתוּב בְאָמְרוֹ ‘הִיא נָתְנָה לִּי מִן הָעֵץ’ • וְהָיָה לוֹ לוֹמַר מִפְּרִי הָעֵץ לְפִי הַנִּרְאֶה • וְעַתָּה אֱמָר‑נָא בְחַסְדֶּךָ מַה פֵּרוּשׁ בּוֹ אַתָּה רוֹאֶה • וָאַעַן וָאֹמַר חַיֶּיךָ יְדִידוּת נַפְשִׁי • פֵּרוּשׁ הַפָּסוּק הוּא הִיא נָתְנָה מִן הָעֵץ עַל רֹאשִׁי • וְדִכְּאָה לָאָרֶץ חַיָּתִי • עַד שֶׁאָכַלְתִּי עַל כָּרְחִי שֶׁלֹּא בְטוֹבָתִי.

וַיִּגַּשׁ הָאַחַד עָשָׂר וַיֹּאמַר אָמַר הַכָּתוּב ‘שְׁנֵי גוֹיִם בְּבִטְנֵךְ’ לְרִבְקָה • וְאָחָד מֵהֶם הָיָה יַעקֹב אֲשֶׁר צוּרָתוֹ תַחַת כִּסֵּא הַכָּבוֹד חֲקוּקָה • וּמִן הַתֵּמַהּ שֶׁהַפָּסוּק בְּשֵׁם גּוֹי יִקְרָאנּוּ • וּבָנָיו יִהְיוּ זֶרַע יַעַבְדֶנּוּ • וָאַעַן וָאֹמַר הַכָּתוּב יִקְרָאֶּנּוּ גוֹי וְלֹא יַחְמוֹל • מִפְּנֵי שֶעֲדַיִן לֹא נִמּוֹל • וְאַחַר שֶׁנִּמּוֹל בָּחַר בּוֹ • וּקְרָאוֹ זֶרַע אַבְרָהָם אֹהֲבוֹ.

וַיִּגַּשׁ הַשְּׁנֵים עָשָׂר וַיֹּאמַר מַה מְּאֹד הָיָה דָוִד בְּעֵינַי אֱוִילִי • שֶׁאָמַר ‘חֲבָלִים נָפְלוּ לִי’ • כִּי אִם נָפְלוּ מִיָּדוֹ הַחֲבָלִים וְאִבְּדָם • מַדּוּעַ הָיָה מִתְרָעַם מֵהֶם לְאָדָם • וָאַעַן וָאֹמַר חַיֶּיךָ ‘נָפְלוּ לִי’ הֵם דִּבְרֵי הַגַּנָּבִים • שֶׁתּוֹלִים אוֹתָם בַּחֲבָלִים עַל שַׁעַר בַּת רַבִּים.

וַּיִּגַּשׁ הַשְּׁלֹשָׁה עָשָׂר וַיֹּאמַר הִנְנִי נִפְרָד מִמַּאֲוַיַּי • אִם לֹא תְפָרֵשׁ לִי ‘אֲהַלְלָה אֲדֹנָי בְּחַיָּי’ • כִּי מִי לֹא יֵדַע בְּכָל אַנְשֵׁי הַחָכְמָה • כִּי לֹא הַמֵּתִים יְהַלְּלוּ יָהּ וְלֹא כָל יוֹרְדֵי דוּמָה • וָאַעַן וָאֹמַר חַיֶּיךָ רָחוֹק אַתָּה מִזֹּאת הַיְדִיעָה • כִּי מִלַּת ‘בְּחַיַּי’ הִיא שְׁבוּעָה • ‘וְקַצְתִּי בְחַיַּי מִפְנֵי בְּנוֹת חֵת’ הִיא שְׁבוּעָה גַם כֵּן • וּלְפִי זֶה הָעִנְיָן הַפָּסוּק יִתָּכֵן.

וַיִּגַּשׁ הָאַרְבָּעָה עָשָׂר וַיֹּאמַר אֲדֹנִי שָׂא מְשָׁלֶיךָ • וְהוֹדִיעֵנִי פֵּרוּשׁ ‘בַּל כְּרוּב אֵלֶיךָ’ • כִּי אֵיךְ אָסַר לָנוּ הַכָּתוּב לֶאֱכֹל הַכְּרוּב • וְהוּא הַדָּבָר הֶחָבִיב אֵלֵינוּ וְהַקָּרוֹב • וָאַעַן וָאֹמַר אֲסָרַתּוּ הַתּוֹרָה כִּי הוּא מַזִּיק לָרֹב • וּלְהַחֲשִׁיךְ הָעֵינַיִם יֶאֱרֹב • וְהַגּוּף הָרָגִיל בּוֹ יֶחֱרַב חָרֹב • תִּשָּׁחֵת הָאָרֶץ מִפְּנֵי הֶעָרֹב • וְעַל כֵּן אָמַר הֶחָכָם הַיִּשְׁמְעֵאלִי • ‘פָּדָה בְשָׁלוֹם נַפְשִׁי מִכְּרוּב לִי’.

וַיִּגַּשׁ הַחֲמִשָּׁה עָשָׂר וַיֹּאמַר חַיֶּיךָ אֲדֹנִי הָיִיתִי יוֹדֵעַ בְּרָצוֹן • פֵּרוּשׁ ‘וְרָחֵל בָּאָה עִם הַצֹּאן’ • כִּי לִהְיוֹת אֵשֶׁת נָבִיא רוֹעָה • הִיא חוֹלָה רָעָה • וָאַעַן וָאֹמַר חַיֶּיךָ לֹא כַאֲשֶׁר יַחְשְׁבוּ הַחֲסֵרִים • רַק רָחֵל הוּא מִן הָ’רְחֵלִים מָאתַיִם וְאֵילִים עֶשְׂרִים'.

וַיִּגַּשׁ הַשִּׁשָּׁה עָשָׂר וַיֹּאמַר רָאִיתִי כָתוּב בַּפָּרָשָׁה • ‘רְאֵה חַיִּים עִם אִשָּׁה’ • וְכִי מָה חַיִּים יִהְיוּ לָנוּ • עִם עָם אֲשֶׁר כִּלָּנוּ • וַאֲשֶׁר דִּמָּה לָנוּ • הֵן הֵנָּה סִבְּבוּ הַמָּוֶת לְכָל הַיְקוּם • וּשְׁדֵי הַמָּוֶת לָאֲנָשִׁים יְנִיקוּם • וָאַעַן וָאֹמַר חַיִּים הַנִּזְכָּר בַּפָּרָשָה הִיא שֵׁם אִישׁ הָיָה מִמָּרֵשָׁה • בַּעַל נֶפֶשׁ קְדוֹשָׁה • וְכַאֲשֶׁר רָאוּהוּ שְׁכֵנָיו מְדַבֵּר עִם הַקְּדֵשָׁה • אָמְרוּ רְאֵה חַיִּים עִם אִשָּׁה.

וַיִּגַּשׁ הַשִּׁבְעָה עָשָׂר וַיֹּאמַר חַיֶּיךָ הַשָּׂר חַיַּי לֹא יִנְעֲמוּ • וּמֵעַי כַּכִּנּוֹר יֶהֱמוּ • עַד שֶׁתְּפָרֵשׁ לִי מַדּוּעַ הִזְכִּיר הַכָּתוּב פַּעֲמַיִם ‘נַחֲמוּ נַחֲמוּ’ • וְאִלּוּ הָיָה נַחֲמוּ שְׁתֵּי פְעָמִים בַּשָּׁנָה הֶחֱרָשְׁתִּי • אָמְנָם אַחֲרֵי הֱיוֹתוֹ פַּעַם אֶחָת בִּלְבָד אָמַרְתִּי נִגְרָשְתִּי • וָאַעַן וָאֹמַר חַיֶּיךָ לִשְׁאֵלָתְךָ נִדְרָשְׁתִּי • וּבְפָרְשִׁי סֵפֶר יְשַׁעְיָה עַל שְׁנֵי פָנִים אוֹתוֹ פֵרַשְׁתִּי • הָאֶחָד הוּא נַחֲמוּ מֵהַתַּעֲנִית הַיּוֹשְׁבִים בְּעָרֵי הַפְּרָזוֹת • כִּי הִגִּיעַ נַחֲמוּ שֶׁתֹּאכְלוּ הָאַוָּזוֹת • וְהַפֵּרוּשׁ הַשֵּׁנִי הוּא נַחֲמוּ בַגָּלוּת בְּאַוָּזוֹת מִדֵּי שָׁנָה בְשָׁנָה • וְנַחֲמוּ לֶעָתִיד בָּאַוָּזוֹת דְּרַבָּה בַּר בַּר‑חָנָא.

וַיִּגַּשׁ הַשְּׁמֹנָה עָשָׂר וַיֹּאמַר אֲדֹנִי שָׁמַעְתִּי שֶׁאֹמְרִים הָרַבָּנִים • ‘בְּכָל עֵת יִהְיוּ בְגָדֶיךָ לְבָנִים’ • וְאָנָה הִנִּיחוּ הַצְּבָעִים הַמְשַׁעַשְׁעִים • וְהָרְקָמוֹת מַעֲשֵׂה צַעֲצוּעִים • הַמְשַׂמְחִים לֵב אַנְשֵׁי הַזְּמָן • וְאֶת הַתְּכֵלֶת וְאֶת הָאַרְגָּמָן • וְיָדוּעַ כִּי כָל אִישׁ אֲשֶׁר יִהְיֶה לְרֹאשׁ • אַחַר כָּל יָרֹק יִדְרֹשׁ • וְאִם הָיוּ בְנֵי דָעַת • אֵיךְ לֹא הִזְכִּירוּ אֶת שְׁנִי הַתּוֹלָעַת • עוֹד שָׁמַעְתִּי שֶׁאֹמְרִים ‘ְושֶׁמֶן עַל רֹאשְׁךָ אַל יֶחְסָר’ • וְזֶה יוֹרֶה כִּי שִׂכְלָם סָר • כִּי בְּיוֹם מָשְׁחָם אֶת רֹאשָׁם • יְכַסּוּ חָמָס עַל לְבוּשָׁם • וְתַעַל צַחֲנָתָם וּבָאְשָׁם • וְלוּ רָאוּ בִזְמַנָּם אִשָּׁה בִּבְתוּלֶיהָ • אוֹ כַּלָּה תַעְדֶּה כֵלֶיהָ • וּכְתֹנֶת הַפַּסִּים אֲשֶׁר עָלֶיה • בְּצֶבַע רִקְמָתַיִם • תִּסְחֹב שׁוּלֶיהָ עַל פְּנֵי הָאָרֶץ כְּאַמָּתַיִם • וַעֲלֵיהֶם מְפֻתָּחִים • כָּל הַצִּיצִים וְהַפְּרָחִים • אֲשֶׁר יָצַר הָאֵל בְּגַנּוֹ • וְאֵת כָּל עֹרֵב לְמִינוֹ • וְרִמּוֹנֵי הַתְּכֵלֶת וְהָאַרְגָּמָן • וְקַרְסֵי הַזָּהָב מַעֲשֶׂה יְדֵי אָמָּן • וְעַטְרוֹת הַזָּהָב אֲשֶׁר עַל רֹאשׁ פָּרוֹת הַבָּשָׁן • אָז יֵעוֹר לִבָּם וְלֹא יִישָׁן • וְלֹא יִזְכְּרוּ הַשֶּׁמֶן וּמַעֲלֵה עָשָׁן • וְיִהְיוּ הַבְּגָדִים הַלְּבָנִים בְּעֵינֵיהֶם כְּפִיחַ הַכִּבְשָׁן • וָאַעַן וָאֹמַר לָמָּה תַרְבֶה דְּבָרֶיךָ • וְתַחְשֹׁב לְהַצְדִּיק אֲמָריךָ • אִם קָרִיתַ לֹא שָׁנִיתָ • כַּאֲשֶׁר לְדִבְרֵי הֶבֶלֶ פָּנִיתָ • חַיֶּיךָ מִי שֶׁלִמֶּדְךָ כִּי לְבָנִים הוּא מֵעִנְיַן לַבְנוּת • לִמֵּד אוֹתְךָ סִכְלוֹת וּגְנוּת • וְלָכֵן פֵּרוּשִׁי מִפִּיךָ לֹא יָמוּשׁ • הַלָּמֶד בְּמִלַּת לְבָנִים הִיא לְשִׁמּוּשׁ • וּפֵרְשׁוּ אַנְשֵׁי הַתְּבוּנוֹת • כִּי הַמִּלָּה מִגִּזְרַת ‘וַיּוֹלֶד בָּנִים וּבָנוֹת’ • וְאָמַר הֶחָכָם אִם תַחְפֹּץ לְהַרְבּוֹת הוֹנֶךָ • שְׁמֹר שְׁמִירַת הַכִּילַי קִנְיָנֶיךָ • וַעֲשֵׂה מִבְּלוֹיֵי בְגָדֶיךָ לְבָנֶיךָ • וְלֹא תַעֲשֶׂה כְּמִשְׁפַּט הַטִּפְּשִׁים • אֲשֶׁר יַעֲשׂוּ לִבְנֵיהֶם בְּגָדִים חֳדָשִׁים • וְאִם תַּחְפֹּץ לָדַעַת חָכְמָה וּמוּסָר • לְמִשְׁמַעַת אַנְשֵׁי הַחָכְמָה הֱיֵה סָר • וְנֵר דָּלוּק מֻנָּח עַל רֹאשְׁךָ אַל יֶחְסָר • לִלְמֹד הַחָכְמוֹת וְהַתְּעוּדוֹת • בַּשָּׁעוֹת הַנִּכְבָּדוֹת • וּבְעִנְיָן זֶה הַסִּכְלוּת יִפְנֶה עֹרֶף • כִּי אֵין רִנָּה שֶׁל תּוֹרָה אֶלָּא בְּלֵילוֹת הַחֹרֶף.

וַיִּגַּשׁ הַתִּשְׁעָה עָשָׂר וַיֹּאמַר אֲדֹנִי בְּכָל עֵת אֲשֶׁר אֲנִי רוֹאֶה • שְׁנֵי פְסוּקִים אֶעֱמֹד מִשְׁתָּאֶה • הָאֶחָד מֵהֶם ‘וּבְעֵינֵי מֹשֶׁה רָע’ • וְהַשֵּׁנִי הוּא ‘וְהֵנִיף יָדוֹ אֶל הַמָּקוֹם וְאָסַף הַמְּצֹרָע’ • וְהַדָּבָר אֲשֶׁר לִבִּי יִקְרָע • וְעָלַי מְאֹד קָשֶׁה • הוּא קְרֹא הַפָּסוּק רַע לְמֹשֶׁה • וְהוּא הָיָה מִבְחַר הַבְּרוּאִים • וְרִאשׁוֹן וְאָב לַנְּבִיאִים • וְכֵן אָמְרוֹ וְהֵנִיף יָדוֹ אֶל הַמָּקוֹם וְאָסַף הַמְּצֹרָע • מָתַי אָסָף הַמְשׁוֹרֵר נִצְטָרָע • וְהִנֵּה אִם לְאַשְׁמַת הָעָם הָרָע • הַדָּבָר הַזֶּה בוֹ אֵרַע • אֵיךְ חֹק הַמּוּסָר פָּרַע • לְהָנִיף יָדוֹ כְּנֶגֶד הַשֵּׁם • הֲלֹא יָדַע כִּי יֶחֱטָא וְאָשֵׁם • וְלָכֵן אֲדֹנִי בְּחַסְדֶּךָ פָּרְשֵׁם • וָאַעַן וָאֹמַר וּבְעֵינֵי מֹשֶׁה רָע • רוֹצֶה לוֹמַר שֶׁחָשַׁשׁ בְּעֵינָיו • וְנִתְרַפֵּא כַּאֲשֶׁר קָרַן אוֹר פָּנָיו • וְאָסַף הַמְּצֹרָע הָיָה מִבְּנֵי הַפְּרִיזִי • וְיֶשׁ‑מְפָרְשִׁים שֶׁהָיָה בְנוֹ שֶׁל גַחֲזִי.

וַיִּגַּשׁ הָעֶשְׂרִים וַיֹּאמַר אֲדֹנִי מַדּוּעַ צִוּוּ לָנוּ הָרַבָּנִים ‘אֶחֱזוּ לָנוּ שֻׁעָלִים שֻׁעָלִים קְטַנִּים’ • וָאַעַן וָאֹמַר חַיֶּיךָ צִוּוּ אוֹתָנוֹ אַחֲרֵיהֶם לַחֲקֹר • וְאוֹתָם בַּחֶרֶב לִדְקֹר • לְצֹרֶךְ עוֹרוֹתֵיהֶם בִּזְמַן הַקֹּר.

וַיִּגַּשׁ הָאֶחָד וְעֶשְׂרִים וַיֹּאמַר נִפְלֵאתִי פַעַם וּפַעֲמָיִם • עַל מַה שֶׁאָמַר הַכָּתוּב ‘הוֹי כָּל צָמֵא לְכוּ לַמָּיִם’ • וְאֵיךְ לֹא אָמַר לְכוּ לַיָּיִן אֲשֶׁר יְחַיֵּנוּ מִיּוֹמָיִם • וָאַעַן וָאֹמַר חַיֶּיךָ הֶחָבֵר • בְּתִשְעָה בְאָב הַכָּתוּב מְדַבֵּר • בַּעֲבוּר הַצָּרָה שֶׁהָיְתָה בוֹ וְרֹב הַמְּבוּכָה • וְעַל כֵּן זֶה הַפָּסוּק כָּתוּב בִּמְגִלַּת אֵיכָה.

וַיִּגַּשׁ הַשְּׁנַיִם וְעֶשְׂרִים וַיֹּאמַר שְׁמָעֵנִי נֵזֶר הַתּוֹרָה • אֵיךְ צִוְּתָה הַתּוֹרָה ‘לֹא יִכְבֶּה בַּלַּיְלָה נֵרָהּ’ • וְהֲלֹא אָסוּר לָאָדָם לְשַׁמֵּשׁ מִטָּתוֹ בִּמְקוֹם אוֹרָה • וְאָסוּר לָאָדָם בַּיּוֹם לְהַעְפִּיל • רַק אִם יִהְיֶה בְּטַלִּיתוֹ מַאֲפִיל • וְאָמְרוּ עַל צַד הָרֶמֶז בַּעֲלֵי הַקַּבָּלָה • וְהָיִיתָ מְמַשֵּׁשׁ בַּצָּהֳרַיִם כַּאֲשֶׁר יְמַשֵּׁש הָעִוֵּר בָּאֲפֵלָה • וָאַעַן וָאֹמַר כְּבָר פֵּרְשׁוּ הַמּוֹרִים • כִּי לֹא יִכְבֶּה בַּלַּיְלָה נֵרָהּ אָמוּר בְּיוֹם הַכִּפּוּרִים • שֶׁאֵין צֹרֶךְ אָז לַעֲלָטָה • לִהְיוֹתוֹ אָסוּר בְּתַשְׁמִישׁ הַמִּטָּה.

וַיִּגַּשׁ הַשְּׁלֹשָׁה וְעֶשְׂרִים וַיֹּאמַר אֲדֹנִי רָאִיתִי כָתוּב בַּמִּזְמוֹר • ‘וַיְדַבֵּר חֲמוֹר אִתָּם לֵאמֹר’ • וְעַתָּה אֱמָר‑נָא הֶחָבֵר אֵיךְ הַחֲמוֹר עִם הָאֲנָשִׁים מְדַבֵּר • וָאַעַן וָאֹמַר אַל תִּתְמַהּ אִם חֲיָלִים יְגַבֵּר • כִּי הַחֲמוֹר הַזֶּה הָיָה בֶּן אֲתוֹן בִּלְעָם • אֲשֶׁר עָשְׂתָה הָאוֹתוֹת לְעֵינֵי הָעָם.

וַיִּגַּשׁ הָאַרְבָּעָה וְעֶשְׂרִים וַיּאֹמַר הִנְנִי וַאֲשִׁישַׁת יַיִן אוֹתִי תַעֲנֹשׁ • אַךְ תְּפָרֵשׁ לִי ‘אָכֵן רוּחַ הוּא בֶאֱנוֹשׁ’ • וָאַעַן וָאֹמַר כְּבָר פֵּרְשׁוּהוּ הָרַבָּנִים • כִּי רוּחַ הוּא בֶאֱנוֹשׁ כְּשֶׁיֹּאכַל הָאֲפוּנִים.

וַיִּגַּשׁ הַחֲמִשָּׁה וְעֶשְׂרִים וַיֹּאמַר אֱמָר‑נָא הַשָּׂר וְהָאַמַרְכֵּל • מַדּוּעַ צִוְּתָה הַתּוֹרָה ‘אֵשׁ לְפָנָיו תֹּאכֵל’ • וְיָדוּעַ כִּי הָאֵשׁ יַחֲשִׁיךְ הָעֵינַיִם • וְהָרָגִיל בּוֹ יִשְכַּב בֵּין שְׁפַתָּיִם • וָאַעַן וָאֹמַר שְׁפֵל לְסֵיפָא דִקרָא • שֶׁהוּא וּסְבִיבָיו נִשְׂעֲרָה' • וְזֶה אָמוּר בְּעֵת הַקֶּרַח הַנּוֹרָא.

וַיִּגַּשׁ הַשִּׁשָּׁה וְעֶשְׂרִים וַיֹּאמַר אֱמָר‑נָא הַשָּׂר בְּחַסְדֶּךָ • מַדּוּעַ אָמַר דָּוִד ‘גַּם מִזֵּיתִים חֲשֹׂךְ עַבְדֶּךָ’ • וַהֲלֹא הַתּוֹרָה צִוְּתָה לָנוּ וְאֵלֶיךָ • ‘זֵיתִים יִהְיוּ לְךָ בְּכָל גְּבוּלֶךָ’ • וָאַעַן וָאֹמַר צִוְּתָה הַתּוּרָה ‘זֵיתִים יִהְיוּ לְךָ בְּכָל גְּבוּלֶךָ’ לְצֹרֶךְ הַמְּנוֹרָה • וּמֵאֱכֹל זֵיתִים חֲשֹׂךְ עַבְדֶּךָ מִפְּנֵי שֶׁמַּזִּיקִים אֶל הַזְּכִירָה.

וַיִּגַּשׁ הַשִּׁבְעָה וְעֶשְׂרִים וַיֹּאמַר אֵיךְ אָמַר יִתְעַלֶּה ‘וְעַמִּי תְלוּאִים’ • וְהָיָה לוֹ לִתְלוֹת הַגּוֹיִם • וְלָתֵת לְיִשְרָאֵל חֶלקָם בַּחַיִּים • וָאַעַן וָאֹמַר זֶה מְבֹאָר עַל יְדֵי שְׁמוּאֵל • שֶׁאָמַר ‘וְהַגִּבְעֹנִים לֹא מִבְּנֵי יִשְׂרָאֵל’ • שֶׁתָּלוּ בְנֵי רִצְפָּה בַת אַיָּה הָאֲנוּנִים • עַל שָׁאוּל וְעַל בֵּית הַדָּמִים אֲשֶׁר הֵמִית אֶת הַגִּבְעוֹנִים.

וַיִּגַּשׁ הַשְּׁמֹנָה וְעֶשְׂרִים וַיֹּאמַר מִמַּה שֶּׁצִּוְּתָה הַתּוֹרָה ‘שְׁתֵה גַם אַתָּה וְהֵעָרֵל’ יֵשׁ לִתְמֹהַּ • כִּי אֵין לִשְׁתּוֹת יַיִן נֶסֶךְ יְצַוֶּה גָבֹהַּ מֵעַל גָּבֹהַּ • וּכְבָר גִּנָּה הֶחָכָם • אֲשֶׁר חֵלֶב זְבָחֵימוֹ יֹאכֵלוּ יִשְתּוּ יֵין נִסְכָּם • וָאַעַן וָאֹמַר זֶה שֶׁהִתִּיר הַכָּתוּב לִשְתּוֹת עִם הֶעָרֵל אָמוּר בְּיַיִן מְבֻשָּׁל • וּבְעִנְיָן אַחֵר יַכֵּהוּ שָׁם עַל הַשַּׁל.

וַיִּגַּשׁ הַתִּשְׁעָה וְעֶשְׂרִים וַיֹּאמַר תָּמֵהַּ אֲנִי תְמִיהָה רַבָּה • מִמַּה שֶּׁצִּוְּתָה הַתּוֹרָה לְהָבִיא כִּנִּים אֶל הַתֵּבָה • וְהִנֵּה אִם נִכְרְתוּ בְּמֵי הַמַּבּוּל מֵעַל פְּנֵי הָאֲדָמָה • עוֹד כָּל ימֵי הָאָרֶץ תְּהִי הַכִּנָּם בָּאָדָם וּבַבְּהֵמָה • וְיָבֹאוּ מִדֵי יוֹם יוֹם לִצְבֹּא צָבָא • דּוֹר הוֹלֵךְ וְדוֹר בָּא • וָאַעַן וָאֹמַר ‘כִּנִּים תַּעֲשֶׂה אֶת הַתֵּבָה’ פֵּרְשׁוּ הַגְּאוֹנִים • שֶׁהוּא מִבְּשָׂרָם כִּי בִהְיוֹתָם בְּתוֹךְ הַתֵּבָה דְחוּקִים וּמְעֻנִּים • יִשְׁרְצוּ בִגְדֵיהֶם כִּנִּים.

וַיִּגַּשׁ הַשְּׁלֹשִׁים וַיֹּאמַר חַיֶיךָ הַשָּׂר יֶשׁ‑לִי שְׁאֵלָה • אֵיךְ אָמַר הַכָּתוּב ‘לָתֵת אֶת אֲחֹתֵנוּ לְאִישׁ אֲשֶׁר לוֹ עָרְלָה’ • הֲלֹא יָדְעוּ שִׁבְטֵי יִשְרָאֵל מַה שֶּׁצִּוְּתָה הַהֲלָכָה • לֹא יָבֹא פְצוּעַ דַּכָּה וּכְרוּת שָׁפְכָה • וְאֵיךְ יִרְצוּ לָתֵת אֶת אֲחוֹתָם • לְאִישׁ שֶׁיֵשׁ לוֹ טַבַּעַת שֶׁאֵין עָלֶיהָ חוֹתָם • וָאַעַן וָאֹמַר חַיֶּיךָ לֹא הֵבַנְתָּ הַדָּבָר עַל בֻּרְיוֹ • וְלָקַחְתָּ מִן הַפָּסוּק עָלֵהוּ וְלֹא פִרְיוֹ • וְלָכֵן אָמְרוּ חֲכָמִים טוֹב יוֹם בְּחַצְרוֹת הַחָכְמָה מֵאָלֶף • הֲלֹא יָדַעְתָּ כִּי מִלַּת לוֹ כְּתוּבָה בְּאָלֶף • וְלָכֵן לֹא רָצוּ לָתֵת לְמִי שֶׁאֵין לוֹ עָרְלָה • כִּי הָעָרְלָה הִיא קֶרֶן הַבְּתוּלָה.

וַיִּגַּשׁ הָאֶחָד וּשְׁלֹשִׁים וַיֹּאמַר מַה זֶּה שֶׁכָּתוּב בַּסְּפָרִים ‘הַמַּמְעִיט אָסַף עֲשָׂרָה חֳמָרִים’ • וְאָנָה נִמְצְאוּ כָּל כָּךְ חֲמוֹרִים בַּתּוֹרָה • שֶׁיָּבֹאוּ מֵהֶם לְכָל אֶחָד עֲשָׂרָה • וָאַעַן וָאֹמַר זֶה הָעִנְיָן הָיָה בְּמִלְחֶמֶת מִדְיָן • שֶׁהָיָה הַמַּלְקֹחָ וְהַשְּׁבִי פִּלְאֵי פְלָאוֹת • וְהָיוּ הַחֲמוֹרִים שְׁלֹשִׁים אֶלֶף וַחֲמֵשׁ מֵאוֹת • וַיִּשְׂמַח הָאִישׁ וַיִּשָּׁבַע בָּאַיָּלוֹת וּבַצְּבָאוֹת • כִּי אֵין אֵלּוּ אֶלָּא דִבְרֵי נְבוּאוֹת.

וַיִּגַּשׁ הַשְּׁנַיִם וּשְׁלֹשִׁים וַיֹּאמָר אֲדֹנִי זַה כַּמָּה עָמַדְתִי נִפְעָם • אֵיךְ אָמַר יְשַׁעְיָה ‘אָכֵן חֲזִיר הָעָם’ • וְכֵן ‘כָּל בָּשָׂר חֲזִיר’ • וְאֵיךְ קָדוֹשׁ וְנָזִיר • יִקְרָא עַם הַקֹּדֶשׁ חֲזִירִים • וּבָזֶה הַשֵּׁם לֹא נִקְרֵאנוּ בְכָל הָאַרְבָּעָה וְעֶשְׂרִים סְפָרִים • וָאַעַן וָאֹמַר חָלִילָה חָלִילָה שֶׁהַנָּבִיא בְּשֵׁם חֲזִירִים יִקְרָאֵנוּ • רַק עַל אֲכִילַת הַחֲזִיר יוֹכִיחֵנוּ • וּכְבָר פֵּרֵשׁ זֶה הַדָּבָר • בְּאָמְרוֹ ‘אֹכְלֵי בְּשַׂר הַחֲזִיר וְהַשֶּׁקֶץ וְהָעַכְבָּר’ • כִּי הַיְּהוּדִים הָהֵם הָאֲמֵלָלִים • הָיוּ קוֹנִים הַחֲזִירִים הַמְּלוּחִים וְהַיְבֵשִׁים מֵהָעֲרֵלִים • וְזֶה נִגְלֶה לַאֲשֶׁר מֵחֲרַכֵּי הַפָּסוּק מֵצִיץ • הָאוֹמֵר ‘יָבשׁ חֲזִיר נָבֵל צִיץ’.

וַיִּגַּשׁ הַשְּׁלֹשָׁה וּשְׁלֹשִׁים וַיֹּאמַר אֶשְׁאָלְךָ הַשָּׂר וּלִתְשׁוּבָתְךָ אֲחַכֶּה • לְמִי שֶׁצִּוְתָה הַתּוֹרָה שֶׁנַּכֶּה • בְּאָמְרוֹ ‘הַזְּרֹעַ וְהַלְחָיַיִם וְהַכֵּה‑בָהּ’ • וָאַעַן וָאֹמַר כָּךְ מְקֻבָּל אֲנִי מִבֵּית אֲבִי אַבָּא • שֶׁשָּׁאַל לְדוֹר רִאשׁוֹן • וְאָמְרוּ כִּי זֶה הַפָּסוּק נֶאֱמַר עַל שִׁמְשׁוֹן • שֶׁהָיתָה לוֹ זְרוֹעַ רָמָה לְהִלָּחֶם בַּפְּלִשְׁתִּים • וְדֹעֲכוּ לְפָנָיו כַּפִּשְׁתִּים • וּבִלחִי חֲמוֹר חֲמוֹרוֹתָיִם סָרוּחַ וּמַבְאִישׁ • הִכָּה אֶלֶף אִישׁ • וַיִּפֹּל הָאִישׁ עַל פָּנָיו • וַיִּשָּׁבַע בְּחַיֵּי אֲבוֹתָיו כִּי אוֹרוּ עֵינָיו • וְלֹא שָׁמַע בִּלְתִּי הַיּוֹם דָּבָר דָּבוּר עַל אָפְנָיו • וַיְהִי כִרְאוֹתִי כִּי נִפְלָא עַל הַפֵּרוּשׁ • אָמַרְתִּי לוֹ שְׁמַע עוֹד פֵּרוּשׁ אַחֵר עַל הַדְרוּשׁ • דַּע כִּי לְפִי מַה שֶּׁכָּתוּב בַּמָּסֹרֶת ‘וְהַקֵּבָה’ • מִלָּה אַחַת בְּקוֹף יֶשׁ‑לוֹ פֵּרוּשׁ אַחֵר וְתוֹעַלְתּוֹ מְרֻבָּה • כִּי בְּזֶה הַפָּסוּק נִרְמַז אֶל הָאִשָּׁה • לִהְיוֹת צְנוּעָה וּקְדוֹשָׁה • כִּי אִם תִּתֵּן לַחֹשֵׁק הַזְּרוֹעַ וְהַיָּד • וּלבַסּוֹף תִּתֵּן לוֹ אֶת הַלְּחָיַיִם וּמֵהֶם יִצְטַיָּד • וְאַחֲרֵי כֵן תִּתֵּן לוֹ הַקֵּבָה • וְלָכֵן תִּזָּהֵר מִזֶּה הַנְקֵבָה.

וַיִּגַּשׁ הָאַרְבָּעָה וּשְׁלֹשִׁים וַיֹּאמַר נִפְלֵאתִי וְעָמַדְתִי מַרְעִיד וּמַשְׁמִים • זֶה חֹדֶשׁ יָמִים • כִּי הַפָּסוּק הֵעִיד עַל נֹחַ שֶׁהָיָה ‘אִישׁ צַדִּיק תָּמִים’ • וַאֲנִי שָׁמָעְתִּי אוֹמְרִים בְּקָרְאָם מְגִלַּת פּוּרִים • הֵפְֶך אֵלֶּה הַדְּבָרִים • כִּי נִתְחַבֵּר עִם הָמָן וּבָנָיו הַבּוֹגְדִים • לְהַשְׁמִיד לַהֲרֹג וּלְאַבֵּד אֶת כָּל הַיְּהוּדִים • וְהוּא אָמְרֹו ‘ְונֹוחַ מֵאֹיְבֵיהֶם • וְהָרוֹג בְּשֹׂנְאֵיהֶם’ • וָאַעַן וָאֹמַר זֶה הַדָּבָר פֵּרֵשׁ בַּר קַפָּרָא כִּי נֹחַ הֵעִיד בּוֹ הַכָּתוּב שֶׁהָיָה צַדִּיק בְּדוֹרוֹתָיו • אָמְנָם בִּימֵי הָמָן לְרֹב זִקנָתוֹ שָׁכָח תּוֹרַת אֲבוֹתָיו • גַּם מִן הַיּוֹם אֲשֶׁר נָטַע כֶּרֶם וְחִלְּלו • וַיֵּשְׁתְּ מִן הַיַּיִן וַיִּשְׁכָּר וַיִּתְגַּל בְּתוֹךְ אָהֳלוֹ • חָדַל מִתְקוֹ וְטַעְמוֹ • וְהָפַךְ לִבּוֹ לִשְׂנֹא עַמּוֹ.

וַיִּגַּשׁ הַחֲמִשָּׁה וּשְׁלֹשִׁים וַיֹּאמַר הוֹשִׁיעָה אֲדֹנִי כִּי מֵעַי כַּכִּנּוֹר יֶהֱמוּ • מַה נֹּאַכל בְּשַׁבָּת‑נַחֲמוּ • וּמַה יַעֲשׂוּ הַיהוּדִים הַפְּרָזִים הַיּוֹשְבִים בּעָרֵי הַפְּרָזוֹת • אַחֲרֵי שֶׁהַכָּתוּב אָסַר אֶת הָאַוָּזוֹת • וְהוּא שֶׁאָמָר יְחֶזְקֵאל • ‘ולֹא תִהְיֶינָה עוֹד לָאַוָּז’ בָּרוּך אַתָּה אֲדֹנָי גָּאָל יִשְׂרָאֵל • וָאַעַן וָאֹמָר הִשָּׁמֵר וְהַשְׁקֶט אַל תִּירָא וְאַל תֶּחֱרַד ולֹא תִבָּעֵת • כִּי לֹא אָסַר הַכָּתוּב אֲכִילַת הָאַוָּזוֹת בְּכָל עֵת • רַק אָסַר אֲכִילָתָם בְּיוֹם שֵׁנִי מֵהַשָּׁבוּעַ • וְצַו אֲכִילָתָם בְּנַחֲמוּ עוֹמֵד לְעוֹלָם קָבוּעַ.

וַיִּגַּשׁ הַשִּׁשָּׁה וּשְׁלֹשִׁים וַיֹּאמַר אֲדֹנִי אִם אֶת הַדָּבָר הַזֶּה תַגִּיד לָנוּ תִּהְיֶה לְרֹאשׁ • מִי הוּא זֶה ואֵיזֶה הוּא אֲשֶׁר רָעָה עַל הַמֶּלֶךְ יַחֲרֹשׁ • וְלָקַחַת אִשְׁתּוֹ מִמֶּנּוּ יִדְרֹשׁ • וְלּא יִרְוֶה לַעֲנָה וָרֹאשׁ • שֶׁאָמַר הַכָּתוּב ‘וַתִּלָקַח אֶסְתֵּר אֶל הַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ’ • וְנִרְאֶה בְעֵינַי כִּי זֶה הַלּוֹקֵחַ • הָיָה סָכָל וְלֹא פִּקֵּחַ • וְהִבְאִישׁ שֶׁמֶן רוֹקֵחַ • וָאַעַן וָאֹמַר חַיֶּיךָ כִּי מִי שׁלְּקָחָהּ לָקָה עַל פִּי מִדּוֹתָיו • וְלָקַח מִיַּד הַמֶּלֶךְ כִּפְלַיִם בְּכָל חַטֹּאתָיו • וְהוּא הָמָן אֲשֶׁר חָשַׁב רַע עָלֶיהָ • כְּמוֹ שֶׁהֵעִיד אָמְרוֹ ‘וְהָמָן נוֹפֵל עַל הַמִּטָּה אֲשֶׁר אֶסְתֵּר עָלֶיהָ’ • וְלָכֵן סָבַל מַה שֶּׁסָּבָל • וּבְאַחֲרִיתוֹ הָיָה נָבָל.

וַיִגַּשׁ הַשִּׁבְעָה וּשְלֹשִׁים וַיֹּאמַר אֲדוֹנִי עַל מַה שֶּׁצִּוְתָה תּוֹרָה • ‘חֲלוּצִים תַּעַבְרוּ’ אֶת הַיַּרְדֵּן תָּמַהְתִּי תְמִיהָה גְּמוּרָה • כִּי לוּ עָבְרוּ יִשְׂרָאֵל אָז אֶת הַיַּרְדֵּן בַּמַּיִם • הָיְתָה חָכְמָה נִפְלָאָה לְצַוּוֹת לַחֲלֹץ הַנַּעֲלַיִם • אָמְנָם אַחֲרֵי שֶׁהוֹבִישׁ אֶת מֵי הַיַּרְדֵּן הָאֵל מִמָּעַל • מַה צֹּרֶךְ לָלֶכֶת חֲלוּצֵי נָעַל • וָאַעַן וָאֹמַר בְּטוֹב חָשַׁבְתָּ • אָמְנָם הַפֵּרוּשׁ הַנָּכוֹן בּוֹ לֹא הִקְשַׁבְתָּ • כִּי לֹא צִוְּתָה הַתּוֹרָה הַנֶּאֱמָנָה • שֶׁיַּעַבְרוּ הַּיַרְדֵּן חֲלוּצִים בַּפַּעַם הָרִאשׁוֹנָה • רַק בְּעָבְרָם בּוֹ בִּשְׁאָר יְמוֹת הַשָּׁנָה.

וַיִּגַּשׁ הַשְּמֹנָה וּשְׁלֹשִׁים וַיֹּאמַר אֱמֹר‑נָא הָאֲמַרְכֵּל • מַדּוּעַ הָיָה מִתְאוֹנֵן אָדָם בְּאָמְרוֹ ‘הִיא נָתְנָה לִּי מִן הָעֵז וָאֹכֵל’ • וַהֲלֹא הָעֵז מַפְרֶסֶת פָּרְסָה • וּמַדּוּעַ הָיְתָה הָעֵז בְּעֵינָיו מְאוּסָה • שְׁאַל נָא הַתּוֹרָה וְתַעֲנֶךָ • שֶׁאָמְרָה וְעַתָּה שְׁלַח הָעֵז אֶת מִקְנֶךָ • וְאֵיךְ בְּאָמְרוֹ ‘הִיא נָתְנָה לִּי מִן הָעֵז’ לֹא יִבָּעֵת • וַהֲלֹא מַעֲשִׂים בְּכָל יוֹם שֶׁדַּדֶּיהָ יְרַוּוּנוּ בְּכָל עֵת • מְצֹרָף אֶל זֶה שֶׁמְּבָרְכִים בְּכָל קְהִלּוֹת עַם לֹעֵז • עַל אֲכִילַת הֶחָלָב ‘בּוֹרֵא פְּרִי הָעֵז’ • וָאַעַן וָאֹמַר חַיֶּיךָ בְּעֵת שֶׁהֶאֱכִילַתּוּ בְשַׂר עֵז הָיָה חוֹלֶה • וּכְסָלָיו מָלְאוּ נִקְלֶה • וְהֻצְרַךְ לְבַרְבּוּר אוֹ בְשַׂר סָרִיס • וְהִיא נָתְנָה לוֹ עֵז מַקְרִין וּמַפְרִיס.

וַיִּגַּשׁ הַתִּשְׁעָה וּשְׁלֹשִׁים וַיֹּאמַר אֲדֹנִי אֵיךְ תְּפָרֵשׁ בְּמַדָּעֲךָ • ‘אַל תַּחֲרֹשׁ עַל רֵעֲךָ’ • כִּי אֵיךְ יַחֲרֹשׁ הָאָדָם בַּבְּקָרִים עַל רֵעוֹ • וָאַעַן וָאֹמָר שְׁמַע הַפָּסוּק בְּרֵעֹה • זֶה הָיָה בִימֵי פַרְעֹה • וְהוּא מַה שֶּׁאָמְרוּ הַנְּבִיאִים הַקְּדוֹשִׁים • ‘עַל גַּבִּי חָרְשׁוּ חֹרְשִׁים’ • גַּם פֵּרְשׁוּהוּ הַמּוֹרִים • שֶׁהוּא אַזְהָרָה שֶׁלֹּא יַחֲרֹשׁ אָדָם עַל הַקְּבָרִים.

וַיִּגַּשׁ הָאַרְבָּעִים וַיֹּאמַר עַד כַּמָּה פְעָמִים אֲנִי מַשְׁבִּיעֶךָ • לְהוֹדִיעֵנִי ‘וְשַׂמְתָּ שַׂכִּין בְּלֹעֶךָ’ • וָאַעַן וָאֹמַר זֶה צִוּוּ אֶל הָרוֹפֵא • לַחֲתֹךְ בְּסַכִּין הָאַסְכְּרַא.

וַיִּגַּשׁ הָאֶחָד וְאַרְבָּעִים וַיֹּאמַר יָדָעְתִּי כִּי מִמְּקוֹר הַשֵּׁכֶל תִּשְׁאָב • וְעַתָּה הוֹדִיעֵנִי פֵרוּשׁ ‘בֵּן חָכָם יְשַׂמַּח אָב’ • כִּי שָׁמַעְתִּי אוֹמְרִים כִּי הַשָּׂמֵחַ בְּאָב לֹא יֵדַע עֲשׂוֹת נְכֹחָה • וּמִשֶּׁנִּכְנָס אָב מְמַעֲטִים בְּשִׂמְחָה • וָאַעַן וָאֹמַר אִם אֵין שִׂמְחָה בְּאָב יֶשׁ‑בּוֹ לְשִׂמְחָה תִקְוָה • לְשַׂחֶק הַנְּעָרִים בּוֹ בְּשַׁעֲוָה.

וַיִּגַּשׁ הַשְּׁנַיִם וְאַרְבָּעִים וַיֹּאמַר חִלִּיתִי פָנֶיךָ לְהוֹדִיעֵנִי עַל אֲמִתָּתוֹ • פֵּרוּשׁ ‘אֲשֶׁר שַׂמְתִּי עֲרָבָה בֵיתוֹ’ • כִּי אֲנִי לֹא רָאִיתִי אִישׁ שֶׁיִּהְיֶה בֵיתוֹ מֵעֲרָבָה • מִמִּזְרַח שֶׁמֶשׁ וּמִמַּעֲרָבָהּ • וָאַעַן וָאֹמַר זֶה הַפָּסוּק פֵּרְשׁוֹ רַבָּה • שֶׁהוּא אָמוּר עַל יוֹם הוֹשַׁעְנָא רַבָּה.

וַיִּגַּשׁ הַשְּׁלֹשָׁה וְאַרְבָּעִים וַיֹּאמַר אִם נַפְשִׁי חֵן בְּעֵינֶיךָ מוֹצֵאת • שָׂא אֵלַי מֵאֵת פְּנֵי חָכְמָתְךָ מַשְׂאֵת • וּפָרֵשׁ לִי ‘רֹחֲצוֹת בֶּחָלָב יֹשְׁבוֹת עַל מִלֵּאת’ • מִי הֵם אֲשֶר רָאוּ בִימֵיהֶם • אֲנָשִׁים יִרְחֲצוּ בֶחָלָב עֵינֵיהֶם • וָאַעַן וָאֹמַר חַיֶּיךָ זֶה אָמוּר עַל הָאֲנָשִׁים • אֲשֶׁר בְּעֵינֵיהֶם חוֹשְׁשִׁים • שֶׁיִּרְחֲצוּ אוֹתָם בַּחֲלֵב הַנָּשִׁים.

וַיִּגַּשׁ הָאַרְבָּעָה וְאַרְבָּעִים וַיֹּאמַר אִם אַתָּה חָכָם וְתָבֶן • מַה פֵּרוּשׁ ‘הִשָּׁמֶר אַל תֵּפֶן אֶל אָבֶן’ • וָאַעַן וָאֹמַר זֶה יָדוּעַ אַף לִקְטַנִּים • כִּי בְעֵת שֶׁיַּשְׁלִיכוּ הַנְּעָרִים הָאֲבָנִים • אֵין לִפְנוֹת אָז פֶּן יַשְׁחִיתוּ הַפָנִים.

וַיִגַּשׁ הַחֲמִשָּׁה וְאַרבָּעִים וַיֹּאמַר אֵיךְ אָמַר הַפָּסוּק ‘יְדֵי נָשִׁים רַחֲמָנִיּוֹת בִּשְּׁלוּ יַלְדֵיהֶם’ • אֵיךְ יִקְרָאֵם רַחֲמָנִיּוֹת וְהֵם הֵם עֵדֵיהֶם • כִּי נִאֲפוּ וְדָם בִּידֵיהֶם • וָאַעַן וָאֹמַר חַיֶּיךָ בְּטוֹב פֵּרַשְׁתָּ • וּבַפֵּרוּשׁ רַחֲמָנִיּוֹת צֶדֶק לָבַשְׁתָּ • כִּי פֵּרוּשׁ רַחֲמָנִיּוֹת זוֹנוֹת יְדַבֵּק אֲדֹנָי בָּם אֶת הַדֶּבֶר • נִגְזַר מִן רַחַם רַחֲמָתִַים לְרֹאשׁ גֶּבֶר.

וַיִּגַּשׁ הַשִּׁשָּׁה וְאַרְבָּעִים וַיֹּאמַר מָה רָצָה הַכָּתוּב בְּאָמְרוֹ ‘פִּיהָ פָּתְחָה בְחָכְמָה’ • וָאַעַן וָאֹמַר שֶׁלֹּא יִכָּנְסוּ בְּפִיהָ הַזְּבוּבִים בִּימוֹת הַחַמָּה.

וַיִּגַּשׁ הַשִּׁבְעָה וְאַרְבָּעִים וַיֹּאמַר אֲדֹנִי אַחֲרֵי אֲשֶׁר זָכִיתָ אֶל זֹאת הַמַּעֲלָה הָרָמָה • רָצִיתִי שֶׁתְּפָרֵשׁ לִי עַל דֶּרֶךְ הַחָכְמָה • ‘מַרְבֶּה בָשָׂר מַרְבֶה רִמָּה’ • כִּי אֲנִי רוֹאֶה אִישׁ יִקְנֶה בָשָׂר עַד צַוָּאר יַגִּיעַ • וִימַלֵּא מִמֶּנּוּ הַבַּיִת וְהַיָּצִיעַ • וּבַמֶּלַח יִמְלָחֵהוּ וְלָכֵן לֹא יַתְלִיעַ • וָאַעַן וָאֹמַר כְּבָר פֵּרְשׁוּ אַנְשֵׁי הַחָכְמָה • כִּי מַרְבֶּה בָשָׂר מַרְבֶּה רִמָּה • אָמוּר עַל מִי שֶׁיִּקְנֶה בָשָׂר רַב בִּימוֹת הַחַמָּה.

וַיִּגַּשׁ הַשְּׁמֹנָה וְאַרְבָּעִים וַיֹּאמַר אֵיךְ אָמַר הַפָּסוּק ‘מַרְבֶּה נְכָסִים מַרְבֶּה דְאָגָה’ • וְאָנוּ רוֹאִים כִּי הַנְּכָסִים הָרַבִּים מְסִירִים הָעֶצֶב וְהַתּוּגָה • וְיָשִׂימוּ הַנֶּפֶשׁ בְּשׁוֹשַׁנֵּי הַשְּׂמָחוֹת סוּגָה • וָאַעַן וָאֹמַר מִי שֶׁכָּתַב דְּאָגָה בְאָלֶף בְּסִפְרָךְ • לֹא הָיָה רַבִּי אֶלְעָזָר בֶּן עֲרָךְ • וְלֹא עֲבַד נְגוֹ וּמֵישָׁךְ וְשַׁדְרַךְ • דִי שְׂעַר רֵאשְׁהוֹן לָא הִתְחָרַךְ • וְכָתַב דְּאָגָה בְאָלֶף בִּשְׁגָגָה • וּבַסְּפָרִים הַמְדֻיָּקִים כָּתוּב בְּחֶסְרוֹן הָאָלֶף דָּגָה • ‘מֵן הַדָּגָה אֲשֶׁר בַּיְאֹר תָּמוּת’ • וְאָמַר כִּי הַמַּרְבִּים נְכָסִים בִּשְׁלֵמוּת • יְנַהֲגֵם הַזְּמָן עַלְמוּת • וְיַרְבּוּ לִקְנוֹת דָּגִים • וְיִהְיוּ אוֹכְלִים וְשׁוֹתִים וְחוֹגְגִים • וְהָעֶצֶב וְהַדְּאָגָה יִשָּׁאֲרוּ עִם הַנּוּגִים.

וַיִּגַּשׁ הַתִּשְעָה וְאַרְבָּעִים וַיֹּאמַר רְאִיתִיךָ הֶחָבֵר • מְפָרֵק הָרֵי הַשְּׁאֵלוֹת וְסַלְעֵיהֶם מְשַׁבֵּר • וְעַתָּה רַעְיוֹנַי נִכְסָפִים • לָדַעַת פֵּרוּשׁ ‘מַרְבֶּה נָשִׁים מַרְבֶּה כְשָׁפִים’ • וָאַעַן וָאֹמַר חֲבָל עַל אַנְשֵׁי חֶסֶד בְאֵין מֵבִין נֶאֱסָפִים • חָשַׁבְתָּ הֱיוֹת כְּשָׁפִים • מִן ‘מְכַשֵּׁפָה לֹא תְחַיֶּה’ • וַאשֶׁר פֵרְשׁוֹ לְךָ כֵן דָּמָיו בּוֹ יִהְיֶה • אִם בְּהֵמָה אִם אִישׁ לֹא יִחְיֶה • וְהִנְנִי קוֹרֵא עַל עַצְמִי אוֹיָה לִי כִּי גַרְתִּי מֶשֶךְ • כְּסָפִים צָרִיךְ לִהְיוֹת בְּסָמֶךְ מִן כַּסְפְּךָ לֹא תִתֵּן לוֹ בְּנֶשֶׁךְ • וְעִנְיַן זֶה הַפָּסוּק כִּי מִי שֶׁיַּרְבֶּה נָשִׁים יִרְבּוּ כְסָפָיו • וְעַשְתְּרוֹת צֹאנוֹ וּשְׁגַר אֲלָפָיו • כִּי לְרֹב נְדֻנְיוֹתָיו יִהְיֶה שֶׁמֶשׁ צְדָקָה וּמַרְפֵּא בִכְנָפָיו • וְיַבְרִיחַ הַיָּגוֹן בַאֲלָפָיו • וְהָיוּ מֻטּוֹת כְּנָפָיו • מְלֹא רֹחַב הָאֲרָצוֹת • וְיַלְבִּישׁ אוֹתוֹ הַזְּמָן מַחֲלָצוֹת.

וַיִּגַּשׁ הַחֲמִשִּׁים וַיֹּאמַר רוֹאֶה אָנִי כִּי כֻלָּנוּ חַבְרֵי עֲצַבִּים • נֶהֱמֶה כַדֻּבִּים • וְלָקַחְנו מִן הַפְתַיּוּת תִּשְׁעָה קַבִּים • וְעַתָּה הוֹדִיעֵנִי פֵרוּשׁ ‘הַמַּלְבִּין פְּנֵי חֲבֵרוֹ בָרַבִים’ • וַאֲנִי לֹא אֵדַע אֶת פֵּשֶר הַדָּבָר וְשִׂבְרוֹ • כִּי לֹא רָאִיתִי בְיָמַי וְלֹא רָאָה אָדָם בְּדוֹרוֹ • אִישׁ שֶׁיַּלְבִּין פְּנֵי חֲבֵרוֹ • אָמְנָם רָאִיתִי וְרָאוּ הָאֲנָשִׁים לְהִשְׁתַּעֲשַׁע בּוֹ • זֶרֶת אָרְכּוֹ וְזֶרֶת רָחְבּוֹ • נָבוּב וְיַעֲשׂוּ בְאֶמְצָעִיתוֹ • מְחִיצָה מַפְסֶקֶת לַסְתִּיר עֲצָתוֹ • הַצַּד הָאֶחָד יַנִּיחֵהוּ רֵק • וּבַצַּד הַשֵּׁנִי קֶמַח סֹלֶת בּוֹ זוֹרֵק • וְאִם יִפְּחו בוֹ אֲשֶׁר לֹא יָבִינוּ • אִם יִהְיוּ שַׂעֲרוֹתֵיהֶם כַּשָּׁנִים כַּשֶּׁלֶג יַלְבִּינוּ • וְהָיוּ כְלֹא הָיוּ • וְאִם יַאְדִּימוּ כַתּוֹלָע כַּצֶּמֶר יִהְיוּ • וְכַאֲשֶׁר יִקָּהֲלוּ הַקְּהִלּוֹת • יִקְרְאוּ לְאַחַד מִן הָרֵיקִים וְיָשִׂימוּ עֲלִילוֹת • וְהַצַּד הָרֵק כָּמוֹהוּ יַרְאוּהוּ • וְלָפַחַת בּוֹ פַּעַם אַחַר פַּעַם יַחְנְכוּהוּ • וְיַעֲשׂוּ כֵן כְּדֵי לְהַבְטִיחוֹ • כְּדֵי שֶׁיִּפַּח בּוֹ לְבַקָּשָׁתָם בַּחֲמַת רוּחוֹ • וְהוּא עַל רְאוֹתוֹ הַכְּלִי רַק יִבְטָח • וְיַחְשֹׁב הֱיוֹתוֹ אָז שִׁמְעוֹן בֶּן‑שָׁטָח • כִּי עֵינָיו מֵרְאוֹת טָח • וְלֹא יֵדַע כִּי הַצַּד הָאַחֵר לַחֲרָמִים וְלִמְצוֹדִים מִשְׁטָח • בּוֹשׁ כִּי בָטָח • כִּי פִתְאֹם יַהֲפֹךְ הַלֵּיצָן יָדוֹ • וְיָכִין הַצַּד הַמָּלֵא סֹלֶת לְצִדּוֹ • וְהוּא לֹא יֵדַע כִּי בָא אֵידוֹ • וְרוּחוֹ בְקִרְבּוֹ נָבְקָה • וְהַלֵּיצָן יְצַוֵּהוּ לָפַחַת בּוֹ בְחָזְקָה • וְהָרֵק יִפַּח עַל מְקוֹם הַקֶּמַח • וְהַלֵּיצָן קוֹרֵא הִנְנִי מֵבִיא אֶת עַבְדִּי צֶמַח • וְאָז יִתְפַּזֵּר הַקֶּמַח עַל פָּנָיו • וְעַל רֹאשׁוֹ וְעַל זְקָנוֹ וְעַל גַּבּוֹת עֵינָיו • וְהָרֵק חוֹרֵק שִׁנָּיו • וְהוּא נָבוֹךְ בְּכָל עִנְיָנָיו • וְיֹּאמַר הִנְנִי מַכִּיר אֵת אֲשֶׁר לֹא הִכַּרְתִּי • וָאֵדַע פֵּרוּשׁ ‘עִם לָבָן גַּרְתִּי’ • וְאַנְשֵׁי הַקְּהִלָּה יָרִיעוּ עָלָיָו תְּרוּעָה גְדוֹלָה • כְּקוֹל מְחַצְצִים בֵּין מַשְׁאַבִּים • עַד שֶׁיָּזוּעוּ צַלְעוֹת הַבַּיִת וְהָעֻבִּים • וְזֶה הוּא פֵרוּשׁ ‘הַמַּלְבִּין פְּנֵי חֲבֵרוֹ בָרַבִּים’.

וַיִּגַּשׁ הָאֶחָד וַחֲמִשִּׁים וַיֹּאמַר נַפְשִׁי עָלַי תִּשְׁתּוֹחָח וּמַה גַּם עַתָּה • וְאִם תְּפָרֵשׁ לִי ‘וְיִהְיוּ תֹאֲמִים מִלְּמַטָּה’ • כָּל אַנְחוֹתַי הִשְׁבַּתָּה • כִּי זֶה הַפָּסוּק לֹא יָדַעְתִּי לְפָרְשׁוֹ • מִפְּנֵי שֶׁאָמַר אַחֲרָיו ‘וְיַחְדָּו יִהְיוּ תַמִּים עַל רֹאשׁוֹ’ • וְנִרְאֶה לְפִי פְשׁוּטֵי הַמִּקְרָאוֹת וְעִנְיָנֵיהֶם • שׁצָּרִיךְ לִהְיוֹת שָׁוִים בָּאֹרֶךְ הָאֲנָשִׁים עִם נְשֵׁיהֶם • וְאָנוּ רוֹאִים אֲנָשִׁים גָּבְהָה קוֹמָתָם • וְאֶל בֵּין עֲבֹתִים הָיְתָה צַמַּרְתָּם • וְלָהֶם נָשִׁים שִׁפְלוֹת קוֹמָה • הָיוּ דֹמֶן לָאֲדָמָה • חָשַׁךְ מִשְּׁחֹר תָּאֳרָם • וְצָפַד עַל עַצְמָם עוֹרָם • וְלֹא יַגִּיעוּ עַד טַבּוּרָם • וְכֵן תִּמְצָא אִישׁ זֶרֶת אָרְכּוֹ וְזֶרֶת רָחבּוֹ • וְעָלְתָה חֲזָזִית עַל רְבּוֹ • וְלוֹ אִשָּׁה פֶּרַח הַצְּבִיּוֹת • צַוָּארָהּ כְּמִגְדַּל דָּוִד בָּנוּי לְתַלְפּיּוֹת • ואַחֲרֵי אֲשֶׁר צִוְּתָה הַתּוֹרָה לִהְיוֹת הָאִישׁ וְאִשְׁתּוֹ שָׁוִים בָּאֹרֶךְ • אֵיךְ לֹא יָבֹא בְלֵב הָאָרֹךְ הַנֹּשֵׂא קְצָרָה מֹרֶךְ • וְכֵן הַקָּצָר הַנֹּשֵׂא אֲרֻכָּה • אֵינֶנּוּ עֹשֶׂה כַּדָּת וְכַהֲלָכָה • וְעַתָּה אֲדֹנִי הַשְׁקֵט שְׁאוֹן לִבִּי • וּנְחֵנִי בְדֶרֶךְ אֱמֶת אִם דֶּרֶךְ עֹצֶב בִּי • וָאַעַן וָאֹמַר חַיֶּיךָ לֹא אֲכַחֵד מִמְּךָ וְלֹא אֲכַסֶּה • ‘וְיִהְיוּ תֹאֲמִים מִלְּמַטָּה’ אֵינֶנָּה מִצְוַת עֲשֵׂה • רַק הַפָּסוּק עֵצָה טוֹבָה יַשְׁמִיעֵנוּ • וְדֶרֶךְ תְבוּנוֹת יוֹדִיעֵנוּ • כִּי הַטּוֹב וְהַנָּעִים לְהַשְׁווֹת בַּעֲלֵי הַחֲבוּרָה • שֶׁהָאָרֹךְ יִשָּׂא אֲרֻכָּה וְהַקָּצָר קְצָרָה • וַיֹּאמֶר הָאִישׁ נִחַמְתַּנִי • אַחֲרֵי שֶׁכָּכָה הִשְׁמַעְתַּנִי • שֶׁאֵין זֶה הַפָּסוּק מִצְוַת עֲשֵׂה • כִּי הָיִיתִי יָרֵא לְנַפְשִׁי מִזֶּה הַמַּעֲשֶׂה • וְזֶה כַמָּה עָמַדְתִּי מִשְתָּאֶה • בַּעֲבוּר הֱיוֹתִי קְצַר קוֹמָה כַּאֲשֶׁר אֲדֹנִי רוֹאֶה • וְנָתַן לִי הַזְּמָן אֵשֶׁת יְפַת תֹּאַר וְטוֹבַת מַרְאֶה • אֲרֻכָּה מֵאֶרֶץ מִדָּהּ • וּבִהְיוֹתֵנוּ תֹאֲמִים מִלְּמַטָּה לֹא יַגִּיעַ רֹאשִׁי לְדַדָּהּ •

וַיִּגַּשׁ הַשְּׁנַיִם וַחֲמִשִּׁים וַיֹּאמַר נִפְלֵאתִי פַעַם וּפַעֲמַיִם • עַל אָמְרוֹ ‘יְהִי רקִיעַ בְּתוֹךְ הַמָּיִם’ • כִּי אֵיךְ יִהְיוּ בְתוֹךְ הַמַּיִם הַשָּׁמָיִם • וָאַעַן וָאֹמַר כְּבָר בֵּאֵר זֶה הַדָּבָר הֶחָכָם • שֶׁיִרְאֶה הַמַּבִּיט בַּמַּיִם צוּרַת הַשָּׁמַיִם בְּתוֹכָם.

וַיִּגַּשׁ הַשְּׁלֹשָׁה וַחֲמִשִּׁים וַיֹּאמַר תָּמַהְתִּי עַל מִי שֶׁאָמַר ‘אַל תִלְחַם אֶת לֶחֶם רַע עָיִן’ • כִּי נַעֲמָה אֲחוֹת תּוּבַל קָיִן • אֲנִי הָיִיתִי חָפֵץ לֶאֱכֹל מִשֶּׁלּוֹ וְלִשְתּוֹת יַיִן • עַד אֲשֶׁר הָיִיתִי מַחֲזִיר מָמוֹנוֹ לְאָיִן • וָאַעַן וָאֹמַר אִם לְדִבְרֵי מְפָרְשֵי הַתּוֹרָה הִקְשַׁבְתָּ • אָז יָדַעְתָּ כִּי אֵין פֵּרוּשׁ ‘רַע עַיִן’ כַּאֲשֶׁר חָשַׁבְתָּ • כִּי פֵרוּשׁ ‘רַע עַיִן’ הוּא הַחוֹשֵׁשׁ בּעֵינָיו • וְהָאוֹכֵל מִן הַלֶּחֶם אֲשֶׁר הוּא אוֹכֵל יִרְבּוּ יְגוֹנָיו • כִּי הוּא מִן הֶחֳלָאִים הַנִדְבָּקִים • כַּאֲשֶׁר בֵּאֲרוּ הַחֲכָמִים וְהַדְּבָרִים עַתִּיקִים.

וַיִּגַּשׁ הָאַרְבָּעָה וַחֲמִשִּׁים וַיֹּאמַר תֶּן‑לִי מַתָּנָה אַחַת עַל מְנָת לְהַחֲזִיר • מָתַי הָיָה שֶׁאָמַר זֶה הַכָּתוּב ‘הַמַּצְמִיחַ הָרִים חֲזִיר’ • וָאַעַן וָאֹמַר זֶה יִהְיֶה בְּבִיאַת הַגֹּאֵל • וְלָכֵן נִקְרָא חֲזִיר שֶׁעָתִיד הַקָּדוֹשׁ בָּרוּךְ הוּא לְהַחֲזִירוֹ לְיִשְׂרָאֵל.

וַיִּגַּשׁ הַחֲמִשָּׁה וַחֲמִשִּׁים וַיֹּאמַר תָּמֵהַּ אֲנִי אֵיךְ הַמְחַבֵּר לֹא יִבָּעֵת • לוֹמַר ‘דַּדֶּיהָ יְרַוֻּךָ בְכָל עֵת’ • וָאַעַן וָאֹמַר חַיֶּיךָ אִלּוּ אָמַר כֵּן עַל הָאִשָּׁה הָיָה אִוֶּלֶת • אָמְנָם אָמַר כֵּן עַל הָאַיֶּלֶת • כִּי בִהְיוֹת הָאַיָּלָה שְׁמֵנָה • יִהְיוּ דַדֶּיהָ לְךָ לְמָנָה • וּבְעִניָן זֶה תִּהְיֶה הָאֱמֶת בּוֹחֵן • כִּי רֹאשׁ הַפָּסוּק ‘אַיֶּלֶת אֲהָבִים וְיַעֲלַת חֵן’.

וַיִּגַּשׁ הַשִּׁשָּׁה וַחֲמִשִּׁים וַיֹּאמַר דּוֹמֶה שֶׁהָיוּ אֲבוֹתֵינוּ כַּת לֵצִים • שֶׁאָמְרוּ ‘נַחְנוּ נַעֲבֹר חֲלוּצִים’ • אוֹ הָיוּ עֲנִיִּים וּמְרוּדִים דְּחוּפִים וַהֲדוּפִים • אַחֲרֵי שֶׁהָיוּ הוֹלְכִים יְחֵפִים • וָאַעַן וָאֹמַר שְׁפֵל לְסֵיפֵיהּ דִּקְרָא כִּי הַכָּתוּב אָמַר, נַחְנוּ נַעֲבוֹר חֲלוּצִים לִפְנֵי אֲדֹנָי • וְקַבָּלָה בְיָדֵינוּ מֵהַר סִינָי • כְּפִי מַה שֶׁהוֹרוּ הַמּוֹרִים • כִּי זֶה הַכָּתוּב מְדַבֵּר בְּיוֹם הַכִּפּוּרִים • שֶׁהוּא מִבֵּין שְׁאָר הַיָּמִים לְטוֹבָה נִבְדָּל • וְעַל כֵּן אָסוּר בִּנְעִילַת הַסַּנְדָּל.

וַיִּגַּשׁ הַשִּׁבְעָה וַחֲמִשִּׁים וַיֹּאמַר מִי קִלֵּל מֹשֶׁה בְאָמְרוֹ ‘אִם בְּהֵמָה אִם אִישׁ לֹא יִחְיֶה’ • וּבְאָמְרוֹ ‘כָּל חֵרֶם בְּיִשְׂרָאֵל לְךָ יִהְיֶה’ • וָאַעַן וָאֹמַר קִלֵּל עֲמָלֵק בַעֲבוּר שֶׁנִּלְחַם עִם יִשְרָאֵל בִרְפִידִים • וְקִלֵּל הָמָן עַל אֲשֶׁר שָׁלַח יָדוֹ בַיְּהוּדִים.

וַיִּגַּשׁ הַשְּמֹנָה וַחֲמִשִּׁים וַיֹּאמַר שָׁמַעְתִּי אוֹמְרִים כִּי תַּעֲנוּג הַנְּשָׁמוֹת • יִהְיֶה לַנֶּפֶשׁ אַחֲרֵי מוֹת • הַגּוּף כְּגַן רַעֲנָן וְרָטוֹב • וּמַה זֶּה שֶׁאָמַר הַכָּתוּב ‘שִׁמְעוּ שָׁמוֹעַ אֵלַי וְאִכְלוּ טוֹב • וְתִתְעַנַּג בַּדֶּשֶׁן נַפְשְׁכֶם’ • וְעַתָּה שְׁמָעוּנִי בְחַיֵי רָאשֵׁיכֶם • מַה תַּעֲנוּג לְנֶפֶשׁ אִישׁ יֵשׁ • מִטּוּב הָעוֹלָם הַזֶּה מִתְיָאֵשׁ • בְּתוֹךְ הַדֶּשֶׁן אֲשֶׁר תֹּאכַל הָאֵשׁ • וָאַעַן וָאֹמַר זֶה הַפָּסוּק לֹא נֶאֱמַר עַל הַנְּפָשׁוֹת הַטְּהוֹרוֹת • רַק אֶל הַנְּפָשׁוֹת הַמְּקַצְּרוֹת • שֶׁאָמַר לָהֶם הַכָּתוּב אִכְלוּ בָעוֹלָם הַזֶה אֲשֶׁר אֲהַבְתֶּם • הַמְעַט מִן הַטּוֹב אֲשֶׁר עֲשִׂיתֶם • וְאַחַר הַמָּוֶת לְכוּ בְאוֹר אִשְּׁכֶם • וְתִתְעַנַּג בַּדֶּשֶׁן נַפְשְׁכֶם.

וַיִּגַּשׁ הַתִּשְׁעָה וַחֲמִשִּׁים וַיֹּאמַר אֲדֹנִי לְעֶזְרָתִי חוּשָׁה וְאַל תִּרְחָק • מַה זֶּה שֶּׁאָמַר הַכָּתּוב ‘יֹשֵׁב בַּשָּׁמַיִם יִצְחָק’ • וְאַחֲרֵי שֶׁיָּדַעְנוּ שֶׁהָאָבוֹת הֵם הֵם הַמֶרְכָּבָה • לְפִי מהַ שֶׁשָּׁמַעְתִּי בַאֲרַם צוֹבָה • וְהָאָבוֹת הָיוּ אַבְרָהָם יִצְחָק וְיַעֲקֹב • מַדּוּעַ יִצְחָק לְבַדּוֹ בְּשֵׁם יִנְקֹב • וְיֹאמַר כִּי הוּא לְבַדּוֹ בַשָּׁמַיִם יֹשֵׁב • וָאַעַן וָאֹמַר לִדְבָרַי הַקְשֵׁב • אֱמֶת שֶׁאַבְרָהָם וְיַעֲקֹב גַּם הֵם בַּמָּרוֹם אָמְנָם מְשַׁמְּשִׁים בַּעֲבוֹדָה בַּעֲמִידָה • אֲבָל יִצְחָק לְבַדּוֹ יוֹשֵׁב בִּזְכוּת הָעֲקֵדָה.

וַיִּגַּשׁ הַשִּׁשִּׁים וַיֹּאמַר חַיֶּיךָ אֲדֹנִי קִבַּלְתִּי מֵאֲבוֹתַי • וְשָׁמַעְתִּי כָל יְמוֹתַי • כִּי מִצְוֹת הַתּוֹרָה אֲשֶׁר נִתְּנוּ בְסִינַי לַאֲבוֹתֵינוֹ • לֹא יָמוּשׁוּ מִפִּינוּ וּמִפִּי זַרְעֵנוּ • בִּפְרָטָם וּבִכְלָלָם • מִן הָעוֹלָם וְעַד הָעוֹלָם • וְהַדְּבָרִים אֲשֶׁר נֶאֶסְרוּ יִשָׁאֲרוּ בְאִסּוּרָם • וַאֲשֶׁר הֻתְּרוּ יִהְיוּ בְהֶתֵּרָם • וַאֲנִי מָצָאתִי בַמִּזְמוֹרִים • הֵפֶךְ אֵלֶּה הַדְּבָרִים • שֶׁאֹמֵר ‘אֲדֹנָי מַתִּיר אֲסוּרִים’ • שֶׁמַּשְׁמַע שֶׁהוּא יַתִּיר הַדְּבָרִים שֶׁכְּבָר אָסָר • וָאַעַן וָאֹמַר שְׁמָעֵנִי הַשָּׂר וְהַטַּפְסָר • דַּע כִּי עַל דֶּרֶךְ הָאֱמֶת מִצְוֹת הַתּוֹרָה לֹא יֹתַּרוּ וְלֹא יֶחְסָרוּ • וְלֹא יִהְיֶה הֶתֵּר אֶל הַדְּבָרִים שׁכְּבָר נֶאֱסָרוּ • אָמְנָם אָמְרוֹ ‘אֲדֹנָי מַתִּיר אֲסוּרִים’ • לְפִי מַה שֶּׁפֵּרְשׁוּ הַמּוֹרִים • הוּא רֶמֶז לְאָמְרוֹ ‘הַסּוּס אָסוּר וְהַחֲמוֹר אָסוּר’ • שֶׁעִנְיָנוֹ שֶׁבַּעֲבוּר עֲוֹנוֹתֵינוּ שֶׁרַבּוּ בַּזְּמָן אֲשֶׁר לֹא סָר וְיִרְבּוּ בְמַה שֶׁלֹּא יָסוּר • סוּס הַיְשׁוּעָה אֲשֶׁר יִרְכַּב הָאֵל לְהוֹשִׁיעֵנוּ • וְהַחֲמוֹר אֲשֶׁר יִרְכַּב עָלָיו מְשִׁיחֵנוּ • הֵם בַּעֲוֹנוֹתֵינוּ הַחֲדָשִׁים וְהָעַתִּיקִים • אֲסוּרִים בָּזִיקִים • וְכַאֲשֶׁר תִּגְזֹר חָכְמָתוֹ יִתְעַלֶּה • זְרוֹעַ תִּפְאַרְתּוֹ לְעֵינֵי הָעַמִּים יְגַלֶה • וְיוֹדִיעַ צִדְקָתוֹ בַחַיִּים • לֹא לְמַעֲנֵנוּ רַק לְמַעַן שְׁמוֹ הַמְחֻלָּל בַּגּוֹיִים • וְיִשְׁפּוֹךְ עַל מַמְלְכוֹת הָאָרֶץ חֲמַת רוּחוֹ • וְיַתִּיר הַסּוּס הָאָסוּר וְהַחֲמוֹר בְּכֹחוֹ • וְיֵצֵא לְיֵשַׁע עַמּוֹ לְיֵשַׁע אֶת מְשִׁיחוֹ.

וַיְהִי כִּשְמֹעַ הַקְּהִלּוֹת • קוֹל הַתֵּרוּצִים יְחוֹלֵל אַיָּלוֹת וּדְבָרִים דְּבוּרִים עַל אָפְנֵיהֶם • וַיִּפְּלוּ עַל פְּנֵיהֶם • וַיֹּאמְרוּ אֵלָי נִחַמְתָּנוּ הַשָּׂר נִחַמְתָּנוּ • שֶכָּכָה הִשְׁמַעְתָּנוּ • וַיִּשָּׁבְעוּ בְנֶפֶשׁ אֲבוֹתָם הַיְתוֹמָה • כִּי לֹא שָׁמְעוּ כָהֵם דִּבְרֵי חֵפֶץ וְחָכְמָה • מִיּוֹם הֱיוֹתָם עַל הָאֲדָמָה • וְלֹא יָמִירוּ מַה שֶּׁלָמְדוּ בַיּוֹם הַהוּא בְּכָל אוֹצְרוֹתָם • וְלֹא יֻתַּן סְגוֹר תַּחְתָּם.

כִּשְׁמֹעַ הַשָּׂר הָיָה לוֹאֵט פָּנָיו וְשֹׂחֵק • וּבִרְאוֹתִי מַעֲשֵׂהוּ הָלַכְתִּי וְיָשַׁבְתִּי לוֹ מִנֶּגֶד כִּמְטַחֲוֵי קֶשֶׁת הַרְחֵק • פֶּן יַרְגִּישׁוּ הָאֲנָשִׁים בְּשַׂחֲקוֹ • וְיָשִׁיבוּ לוֹ מִן הַחֶרְפָּה שִׁבְעָתַיִם אֶל חֵיקוֹ • וְכַאֲשֶׁר חָדְלוּ הַקּוֹלוֹת • וְנִפְרְדוּ הַקְּהִלּוֹת • חָזַרְנוּ הַשָּׂר וַאֲנִי שָׂשִׂים וּשְׂמֵחִים • וְצָפַנּוּ מִן הַשִּׂמְחָה הַהִיא שְׁלֹשָׁה יְרָחִים • וַיֹּאמֶר אֵלַי הַשָּׂר בָּרוּךְ אֲדֹנָי צבָאוֹת • אֲשֶׁר נָתַן הַיּוֹם בְּפִיךָ דִבְרֵי נְבוּאוֹת • וּלְעוֹלָם לֹא עָשִׂיתִי בִּלְתִּי הַיּוֹם גְּבוּרוֹת נִפְלָאוֹת • וַיְצַו הַשָּׂר לִכתֹּב הַדְּבָרִים הָאֵלֶּה לְזִכָּרוֹן • לְמַעַן יֵדְעוּ אוֹתָם דּוֹר אַחֲרוֹן • וָאֶעֱשֶׂה כֵן כַּאֲשֶׁר צִוַּנִי וָאָשִׂים אוֹתָם בָּאָרוֹן.

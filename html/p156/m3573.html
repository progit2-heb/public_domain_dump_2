מחברות עמנואל

מחברת שביעית

אָמַר עִמָּנוּאֵל הַמְּחַבֵּר:

יָצָאתִי עִם הַשָּׂר בְּחֹדֶשׁ אֲדָר

לִרְאוֹת הֲפָרְחָה הַגֶּפֶן פִּתַּח הַסְּמָדַר

וְהִנֵּה גְבִירָה לִקְרָאתֵנוּ מְלֵאָה הוֹד וְהָדָר

וְהַטֶּבַע כָּלַל בְּגִשְׁמָהּ כָּל יֹפִי אִישׁ לֹא נֶעְדָּר

וּבִרְאוֹתִי יִפְעַת הוֹדָהּ

נִפְלֵאתִי מִזִּיו כְּבוֹדָהּ

וָאֹמַר אֶל הַשָּׂר מָה הַיּוֹצֵאת הַזֹּאת הָעַלִּיזָה

וּמִי הוּא הַמַּחֲזִיק בְּמָעֻזָּהּ

וַיַעַן הַשָּׂר וַיֹּאמַר דַּע הַדּוֹד הַנֶּאֱמָן

כִּי הָאִשָּׁה אֲשֶׁר רָאִיָת הִיא צִנְצֶנֶת הַמָּן

וַעֲטֶרֶת צְבִי וּצְפִירַת תִּפְאָרָה לַזְּמָן

וּלְיָפְיָהּ וּמַעֲלוֹתֶיהָ עַל כָּל הַגְּבִירוֹת מוֹלֶכֶת

וְלָהּ אַהֲבַת חֶסֶד וְהַצְנֵעַ לֶכֶת

וּבַעֲלָהּ הוּא עֵגֶל יָרָבְעָם

חֶרְפַּת אָדָם וּבְזוּי עָם

יַעֲמֹד בֵּין הָאֲנָשִׁים כְּנִפְעָם

בָּא שִׁמְשׁוֹ וִירֵחוֹ קָדָר

לֹא תֹּאַר לוֹ וְלֹא הָדָר

הִשָּׁהוּ אֵל דַּבֵּר נְכוֹנָה

וְלֹא חָלַק לוֹ בַּבִּינָה

לֹא יָדַעְתִּי מֶה הָיָה לוֹ

לַעֲלוֹת זֹאת הַצְּבִיָּה בְגוֹרָלוֹ

אִם מֵיְיָ הָיְתָה לוֹ יָרַח מִנְחָה

וְאִם מִבְּנֵי אָדָם אֲרוּרִים הֵם כִּי לֹא יָדְעוּ עֲשׂוֹת נְכוֹחָה

כִּי גֵרְשׁוּהַ הַיּוֹם מֵהִסְתַפֵּחַ בְּנַחֲלַת אֲדֹנָי

וְעָלֶיהָ יִבְכְּיוּן עֵינָי

וְכִרְאוֹתִי אֲשֶׁר הַזְּמָן הִתְאַכְזְרוּ נִחוּמָיו

וְעַל הַנְּדִיבִים יַהֲלֹךְ אֵימָיו

וַאֲשֶׁר לֹא טוֹב יַעֲשֶׂה בְּתוֹךְ עַמָּיו

וְלֹא יַבְחִין בַּאֲשֶׁר יַעֲשֶׂה

שָׂנֵאתִי אֶת הַחַיִּים כִּי רַע עָלַי הַמַּעֲשֶׁה

כִּי מַדּוּעַ לֹא יְחַלֵּק הַזְּמָן בְּצֶדֶק אֶת כְּבוֹדוֹ

וִיחַבֵּר הַטּוֹב עִם הַטּוֹב וְהָרַע יִרְאוּ עֵינָיו כִּידוֹ

אֵיךְ סָכַל הַזְּמָן וְכָהֲתָה עֵינוֹ

בְּשׂוּם לַנֶּשֶׁר עִם הָעֹרֵב קִנּוֹ

וּמַדּוּעַ לֹא הִשְׁכִּין הַנֶּשֶׁר עִם

הַנֶּשֶׁר וְאֶת כָּל עֹרֵב לְמִינוֹ

וְלֹא אֶחָד עָשָׂה וּשְׁאָר רוּחַ לוֹ

כִּי הִנְנִי נִשְׁבָּע בְּחֵי הַחֵשֶׁק וּבְנֵר הִלּוֹ

כִּי נָתַתִּי אֶת לִבִּי לָדַעַת וְלָבוּר אֶת כָּל זֶה

וְלֹא מָצָאתִי אֵשָׁת יְפַת תֹּאַר רַק לַנִּבְזֶה.

וַיְנַהֵג הַזְּמָן אֶת בְּנוֹתָיו כִּשְׁבוּיוֹת

בְּתִתּוֹ מִבְחַר הַצְּבִיּוֹת וְיוֹנֵי הַגְּאָיוֹת

לִמְזֵי רָעָב וּלְחֻמֵּי רֶשֶׁף וּפְרִיצֵי חַיּוֹת

וּבְהֵיכְלֵי עֹנֶג וּבִמְעוֹנוֹת אֲרָיוֹת

שָׁם נִקְבְּצוּ דַיּוֹת

וְעַתָּה הֲשָׁכַח חַנּוֹת הַזְּמָן אִם קָפַץ בְּאַף רַחֲמָיו

כִּי מַדּוּעַ לֹא נִכְמְרוּ נִחוּמָיו

בְּתִתּוֹ יְפוֹת הַמַּרְאֶה וְהַבְּרִיאוֹת

אֶל הַסְּכָלִים אֲשֶׁר לֹא יֵאוֹת

וְנָתַן לַחֲכָמִים וְלַחֲסִידִים

צְנֻמּוֹת דַּקּוֹת וּשְׁדֻפוֹת קָדִים

וְהַשָּׂרִים וְהַסְּגָנִים בְּגוֹרָלָם נָפָל

שְקַעֲרוּרוֹת יְרַקְרַקֹּת אוֹ אֲדַמְדַּמֹּת וּמַרְאֵיהֶן שָּפָל

עַל זֹאת תֶּאֱבַל הָאָרֶץ וְאֻמְלַל כָּל יוֹשֵׁב בָּהּ

כִּי זֶה הֶבֶל וְרָעָה רַבָּה

וָאַעַן וָאֹמַר אֶל הַשָּׂר אֲדֹנִי

אַל יֹאמַר כֵּן נֵזֶר זְמַנִּי

כִּי הִנְנִי נִשְׁבָּע בִּגְאוֹנִי

כִּי שִׂכֵּל הַזְּמָן יָדָיו

כַּאֲשֶׁר נָתַן נְזָמָיו וּצְמִידָיו

וַיָּפֹזוּ זְרֹעֵי יָדָיו

בְּתִתּוֹ שְׁבִיסָיו וְשַׂהֲרֹנָיו

לִסְכָלָיו וְנִרְגָּנָיו

וְהַמְּלָאכָה נְמִבְזָה וְנָמֵס לַחֲכָמָיו וּנְבוֹנָיו

וְהָרַקּוֹת וְהָרָעוֹת

לְשָׂרָים וּקְצִינָיו

כִּי רָצָה הָאֵל שֶׁהֶחָכָם לֹא יִבְעָט

וְשֶׁהַסְּכָלִים בְּהִכָּשְׁלָם יֵעָזְרוּ עֵזֶר מְעָט

כִּי הַחֲכָמִים בְּחָכְמָתָם יַקְדִּימוּ לְמַחֲלַת הַחֶסְרוֹן וְהָאֵבֶל

צֳרִי גִיל-עַד וְרִפְאוּת הַסֵּבֶל

יָבוּזוּ לְכָל חֶסְרוֹן

וְיִשְׂחֲקוּ לְיוֹם אַחֲרוֹן

וְתִהְיֶה לָהֶם בְּחָכְמָתָם עַל הַכְּסִילִים יִתְרוֹן

וְהַסְּכָלִים וְהַפְּחוּתִים

בְּחַיֵּיהֶם קְרוּאִים מֵתִים

וְלוּ הִתְחַבְּרָה לָהֶם צָרַת הַזְּמָן עִם סִכְלוּתָם

תִּתְהַפֵּךְ בְּחֹמֶר חוֹתָם

וְלֹא קָפַץ הַזְּמָן בְּאַף רַחֲמָיו וְלֹא שָׁכַח חַנּוֹת

אָמְנָם רָצָה הָאֵל שֶׁלֹּא יִזְכֶּה אָדָם לִשְׁתֵּי שֻׁלְחָנוֹת

וּלְכָךְ סִבְּבָה הַחָכְמָה הָעֶלְיוֹנָה

לָתֵת אֵל גְּדוֹלֵי הַמַּעֲלָה הַחָכְמָה וְהַתְּבוּנָה

וּמָנַע מֵהֶם עֲרֵבוּת קְטַנָּה

וְאִם נָתַן לַסָּכָל יְפִי אִשְׁתּוֹ

פָּרַשׂ עָלָיו רִשְׁתּוֹ

וְחִסְרוֹ מַעֲלַת הַחָכְמָה וְיָשׂם עַל עֵקֶב בָּשְׁתּוֹ

עֵת הִשּׁפֵך נְחֻשְׁתּוֹ

וְנָתַן לֶחָכָם יְפִי הַחָכְמָה וְלֹא יִירָא בְגִשְׁתּוֹ

וַתֵּשֶׁב בְּאֵיתָן קַשְׁתּוֹ

וְאִם יְפִי הָאִשּׁה וְהוֹן עָתָק חִסְּרוֹ

הֵן פָּרַשׂ עָלָיו אוֹרוֹ

כִּי לוֹ הִתְחַבֵּר יְפִי הַחָכְמָה הַקְּדוֹשָׁה

וִיפִי הַמָּמוֹן וְהָאִשּׁה

בַּחֲכָמִים

תִּהְיֶה מִלְחָמָה בֵּין הַתַּאֲוָה וְהַשֵּׂכֶל כָּל הַיָּמִים

וּפֶן תִּהְיֶה נֶפֶשׁ הַסְּכָלִים בְּסִכְלוּתָם נוּגָה

רָצָה הָאֵל לִרְעוֹת אֶת צֹאן הַהֲרֵגָה

וְזִמֵּן לָהֶם מִיפִי הַזְּמָן עֲרֻגָּה

בְּשׁוֹשָׁנֵי הַחֵן סוּגָה

כַּאֲשֶׁר זִמֵּן הַטֶּבַע לָעַכָּבִיש הָאֲרִיגָה

לָצוּד צַיִד לְהָבִיא הַזְּבוּב אֲשֶׁר יִפֹּל שָׁם בִּשְׁגָגָה

וְכַאֲשֶׁר זִמֵּן הַחֲרִיצוּת לַנְּמָלָה

וּבְטֶבַע לֹא בְּחָכְמָה וְהַשְׂכָּלָה

תָּכִין בַּקַּיִץ לַחְמָהּ אָגְרָה בַקָּצִיר מַאֲכָלָהּ

וְכַאֲשֶׁר צִיר רַעֲבוֹן הַדְּבֹרִים חָבָשׁ

בְּהַטְבִּיעוֹ אוֹתָם עַל בִּנְיַן חַדְרֵי מַשְׂכִּית וַעֲשׂות דְּבָשׁ

וְהִנֵּה כַּאֲשֶׁר חָנַן הַטֶּבַע הַבְּרִיּוֹת הַנִּזְכָּרוֹת הַסְּכָלוֹת

וְהִטְבִּיעָם עַל מְלָאכוֹת מְהֻלָּלוֹת

כֵּן חָמַל הָאֵל עַל הַסְּכָלִים

וְנִחֵם אוֹתָם מֵהֲבֵל הֲבָלִים

כִּי בְּלֹא זֹאת הַחֶמְלָה יָמוּתוּ לְשָׁעָתָם

וְתִהְיֶה רָעָה לָעוֹלָם בְּרָעָתָם

לִהְיוֹתָם חֹטְבֵי עֵצִים וְשֹׁאֲבֵי מַיִם לַחֲכָמִים וְלַנְּדִיבִים

וּלְכֹרְמִים וּלְיֹגְבִים

מְצֹרָף אֶל זֶה הֱיוֹת יֵצֶר לֵב הָאָדָם רַע מִנְּעֻרָיו

וְלֹא יְבִיאֶנּוּ לִרְאוֹת בְּמָה שֶׁיִהְיֶה אַחֲרָיו

וְהַחֵשֶׁק דָּבָר טִבְעִי בָאָדָם

יָמִירוּ בוֹ הָאֲנָשִׁים אֶת כְּבוֹדָם

וְהַחֹשְׁקִים בִּמְצוּדַת הַיֲּעֵלוֹת יִלָּכֵדוּ

כִּי הֵם עִם הַזּוֹנוֹת יְפָרֵדוּ

וְלוּ חָלַק הַזְּמָן הַיָּפוֹת אֶל הַחֲכָמִים וְהַנִּכְבָּדִים

הָיוּ הַחשְׁקִים בִּגְדוֹלֵי הַמַּעֲלָה בוֹגְדִים

וְלָכֵן בָּחֲרָה הַחָכְמָה הָעֶלְיוֹנָה

לִהְיוֹת הַיָּפוֹת אֶל הַסְּכָלִים לְמָנָה

יִהְיֶה הַסָּכָל נוֹתֵן שְׁאֵרָהּ וּכְסוּתָהּ

וְהַחשְׁקִים יִתְּנוּ אֶת עֹנָתָהּ

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

שִׂכֵּל הַזְּמָן יָדָיו בְּתִתּוֹ יַעֲלוֹת

הַחֵן לְכָל סָכָל וּמִשְׁתַּגֵּעַ

לַיְלָה וְיוֹמָם לִהְיוֹת יָגֵעַ

לָתֵת שְׁאֵר וּכְסוּת וְחֹזֶק נַחֲלוֹת.

לִהְיוֹת בְּחַדְרֵי חשְׁקִים מִתְהֹלְלוֹת

חשֵׁק בְּיַעְלָה לִהְיוֹת פֹּגֵעַ

יִשַׁק שְׂפָתֶיהָ וְכִמְתַעְתֵּעַ

יִשְׂאַף כְּעֶבֶד צֵל וְלַיְלָה לַעֲלוֹת

לוּ הַצְּבִיּוֹת לַצְּבָאִים נִתְּנוּ

אָז הַצְּבָאִים בַּצְּבָאִים יִבְגְּדוּ

אוֹ מַעֲלַת חֵשֶׁק תְּהִי נֶעְדָּרֶת

נִקַּח צְבִיּוֹת מֵאֳוִילִים שָׁמְנוּ

נִתְעַלְסָה נַחְנוּ וְהֵמָּה יֹאבְדוּ

רָעָה תְּבוֹאֵמוֹ וְהַמִּגְעֶרֶת

וְאֵיךְ יִפָּלֵא בְּעֵינֵיךָ הַשּׂר וְתִתְמַהּ

אִם כְּסִיל יִשָּא כִימָה

וְאִם הַשְׂנוּאוֹת אֲשֶׁר יַעֲלֶה בְּאֵשׁ מַחֲנֵיהֶם נִתְּנוּ לְכוֹכְבֵי הַשָּׁמַיִם וּכְסִילֵיהֶם

וַיַּגִיעוּ בַעֲדָם אֶל שַׁעֲרֵי מָוֶת

כִּי כֵן הָאֵל עֹשֵׂה כִימָה וּכְסִיל וְהֹפֵךְ לַבֹּקֶר צַלְמָוֶת

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

יוֹם נִשֵׂאת עָפְרָה לֶחָסִיל

אָמְרוּ מָה אֶל כִּימָה לִכְסִיל

אָמַרְתִּי כֵּן הָאֵל עֹשֵׂה

פֶּלָא עֹשֵׂה כִימָה וּכְסִיל.

וַיֹּאמֶר הַשָּׂר תָּמֵהַּ אֲנִי אִם תִּמָּלֵט אִשּׁה מִן הַיָּפוֹת

אֲשֶׁר שִׂפְתוֹתֶיהָ שׁוֹשָׁנִים מוֹר נוֹטְפוֹת

שֶׁלֹא תִּהְיֶינָה רוּחַ הַחֹשְׁקִים שׁוֹאֲפוֹת

אַחֲרֵי אֲשֵׁר הַזְּמָן יְצוּדֵם לְמַדְחֵפוֹת

וְנָתַן לַשָּׂרִים הַשְּׁפָחוֹת הַחֲרוּפוֹת

וְהָיוּ הַכְּסִילִים לְחַיֵּי עוֹלָם וְהַשּׂרִים וְהַסְּגָנִים לַחֲרָפוֹת

וָאַעַן וָאֹמַר דַּע אֲדֹנִי כִּי עַל הָרֹב אֱמֶת הַדָּבָר

אָמְנָם בִּקְצָתָם נִכְבָּדוֹת מְדֻבָּר

וְהָיוּ הַצְּבִיּוֹת הַבּוֹגְדוֹת בְּבַעֲלֵיהֶם

חֲלוּקוֹת לְמִינֵיהֶם

קְצָתָם מִתְקַדְּשׁוֹת וּמִטַּהֲרוֹת

בְּעֵינֵי הַסְּגָנִים וְאַנְשֵׁי הַשּׂרָרוֹת

לֹא תַחְפֹּצְנָה מִבְּנֵי אָדָם

רַק שְׂעִירֵי חַטָּאת לְכַפֵּר בַּעֲדָם

תָּקֹצְנָה בְאִישׁ מַרְאֵהוּ כַלְּבָנוֹן בָּחוּר כָּאֲרָזִים

וְתַעְגֹּבְנָה בַאֲנָשִׁים רֵיקִים וּפוֹחֲזִים

וּשְׂעִירֵי עִזִּים

כַּאֲשֶׁר הָאֲתוֹנוֹת תִּבְזֵינָה לְאֶשְׁכֹּל הַכֹּפֶר וּצְרוֹר הַמֹּר

וְלֹא תִּכְסֹפְנָה רַק לְפֶטֶר חֲמוֹר

וְלוּ פָגְעוּ בְעַיִּר בֵּן אֲתוֹנוֹת

אָמְרוּ עָלָיו אֲגָג מַעֲדַנּוֹת

כַּאֲשֶׁר קָרָה לִי עִם אַחַת מִבְּנוֹת הָעֲדִינוֹת

אֲשֶׁר מִימֵי קֶדֶם אֲהַבְתִּיהָ

וְעַל גַּרְגְּרוֹתַי עֲנָק עֲנַדְתִּיהָ

וְהָיְתָה יְדִידוּת נַפְשִׁי וְשַׁעֲשׁוּעַי וְחָשַׁבְתִּי כִּי גַם הִיא תֶאֱהָבֵנִי

וּלְחשֶׁק נֶאֱמָן תַּחְשְׁבֵנִי

וַיְהִי הַיּוֹם וַיִּקֶר מִקְרֶהָ חֶלְקַת הַשָּׂדֶה אֲשֶׁר לִי וּפְגַעְתִּיהָ

וּכִרְאוֹתִי אוֹתָה לְשָׁלוֹם קְרָאתִיהָ

וּבֵרַכְתִּי אוֹתָהּ בְּשֵׁם אֲדֹנָי

וְלֹא עָנְתָה וְלֹא שָׁתָה לִבָּהּ וַתִּבְרַח מִפָּנַי

לֹא יָדַעתִּי אִם חֲשָׁבַתְנִי לְשָׂטָן

אוֹ אִם הָיִיתִי בְעֶרְכָּה קָטָן

אוֹ אִם חֲשָׁבַתְנִי לְמַלְאָךְ וְשָׂרָף

אוֹ לַאֲרִי נֹהֵם וְדֹב שׁוֹקֵק וּזְאֵב יִטְרָף

וְלוּ הָיִיתִי עַיִר בֶּן אֲתוֹנוֹת אָז

יִמָּלֵא פִּיהָ שְׂחוֹק

וַתְּשַׁלַח לִי צִירֶיה עַד לְמֵרָחֹק

וַתֹּאמֶר כִּי אֵין זוּלָתִי לִגְאוֹל

וַתַּשְׁפִּיל עַד שְׁאוֹל

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

צְבִיָּה תַּאֲוָה הָיְתָה לְעֵינַי

פְּגַשְׁתִּיהָ תְמוֹל וַתֵּט לְפָנָי

כְּמוֹ עָשָׂת בְּעֵת כִּי רָאֲתָה אָז

אֲתוֹן בִּלְעָם פְּנֵי מַלְאַךְ אֲדֹנָי

חֲשָׁבַתְנִי כְּמַלְאַךְ הָאֱלֹהִים

וְעַל כֵּן יָרְאָה לִרְכֹּב עֲנָנָי

וְאוּלַי כִּי חֲשָבַתְנִי לְשָׂטָן

וְעַל כֵּן זָחְלָה מֵרֹאשׁ פְּתָנָי

וְלוּ אֶהְיֶה כְּעַיִר בֶּן אֲתֹנוֹת

אֲזַי מִלְאָה כְרֵשָׂה מַעֲדָנָי

וְלִקְצָתָם גַאֲוָה וְגֹדֶל לֵבָב וְעֹרֶף קָשֶׁה

לֹא תַעֲגֹב רַק לִבְנֵי חוֹבָב חֹתֵן מֹשֶׁה

תְּשׁוֹטֵט אַחַר בְּכוֹר פֶּטֶר רֶחֶם

וְלֹא בְּמִסְכֵּנֻת תֹּאכַל בָּה לֶחֶם

תִּבֶז לַחֹשֶׁק מָךְ מֵעֶרְכָּהּ

וְתִמְשׁוֹך לִוְיָתָן בְּחַכָּה

לֹא תַחְפֹּץ מִתֵבֵל כִּי אִם מַלְכָּהּ

כַּאֲשֶׁר קָרָה לִי עִם אַַחַת מִבְּנוֹת הַנְּדִיבִים

אֲשֶׁר הָיוּ דַדֶּיהָ בְּעֵינַי עֲרֵבִים

וְכַאֲשֶׁר חָשַבְתִּי לְהִתְעַנֵּג בְּדֹדֶיהָ

וְלֶאֱכֹל פְּרִי מְגָדֶיהָ

אָמְרָה עִם אֶחָד מְאַנְשֵׁי סוֹדֶיהָ

אֱמֹר לִפְלוֹנִי כֹּה אָמְרָה פְלוֹנִית דּוֹם כִּי

אַתָּה חשֵׁק מָךְ מֵעֶרְכִּי

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

לֵב עָפְרָה בִּשְׁמֵי שָׁמַיִם

אֶל חשֵׁק מִסְכֵּן לֹא פָּנָה

עֵינֶיהָ מִלִּבּוֹת חשְׁקִים

אֹכְלוֹת בָּשָר וַתִּרְעֵינָה

בָּא הֵלֶךְ וַתִּבֶז לוֹ תּוֹך

לִבָּהּ בָּא שַׂר וַתִּיר עֵינָהּ.

וְלִקְצָתָם בְּקִרְבָּהּ רוּחַ עִוְעִים

הֶבֶל הִיא מַעֲשֶׂה תַּעֲתֻּעִים

לֹא תֵּדַע דֶּרֶךְ שַׁעֲשׁוּעִים

יִהְיוּ לָהּ אֲחֻזַּת מֵרֵעִים

וְהֵם בְּאַהֲבָתָה נָדִים וְנָעִים

לֹא בַחשֵׁק תַּחְפֹּץ

וְלֹא יַד חִשְׁקָהּ מִמֶּנוּ תִקְפֹּץ

פַּעַם תִּרְחַק פַּעַם תִּקְרָב

פַּעַם לַמִּזְרָח וּפַעַם לַמַּעֲרָב

כְּבִכְרָה קַלָּה דְרָכֶיהָ מְשָׂרֶכֶת

וּכְלַהַט הַחֶרֶב הַמִתְהַפֶּכֶת

פַּעַם תַּחֲשֹׁק בְּדוֹדָהּ וּתְקָרְבֵהוּ

וּפַעַם עַל דָּבָר נָקֵל תַּרְחִיקֵהוּ

כַּאֲשֶׁר קָרָה לִי עִם אַחַת הָעֲלָמוֹת חֲשַׁקְתִּיהָ

וְאַהֲבַת עוֹלָם אֲהַבְתִּיהָ

פַּעַם הָיְתָה לִי מֵעִיר לַעְזוֹר

וְהִדְבִּיקַתְנִי אֵלֶיהָ כַּאֲשֶׁר יִדְבַּק הָאֵזוֹר

וְכֶסֶף וְזָהָב וּרְכוּשׁ אֵלַי תִּבְזוֹר

וּפַעַם הָיִיתִי מֵחִשְׁקָהּ אָחוֹר נָזוֹר

אֵין דָּן דִּינִי לְמָזוֹר

וּלְהֶבֶל וָרִיק לְמַעֲצֵבָה תַּשְׁכִּיבֵנִי

וְעַל מַעַל מָה מַטָּרָה לְחִצֶּיהָ תַּצִּיבֵנִי

וָאֹמְרָה אֶל לִבִּי מַדּוּעַ זֹאת הָעַלְמָה פַּעַם תְּקָרְבֵנִי וּפַעַם תַּרְחִיקֵנִי

אוֹ לִנְדוֹד תְּמַהֲרֵנִי

אוֹ “לִין דּוֹד” תְּבַשְׂרֵנִי

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

עַל מָה עַלְמָה מַעַל מָה

עַלְמָהּ עֵת יִשְׁקֹט עֵת יִדּוֹד

אוֹ בּוֹ תַחְפֹּץ אוֹ יָד תִּקְפֹּץ

וַתְּצַו לוֹ לִין דּוֹד אוֹ לִנְדּוֹד

וּקְצָתָם יְבִיאֵם הַדַּלוּת

לְהַנִיַח הַחָכְמָה וְלֶאֱחֹז בְּסִכְלוּת

וְעֵינֵיהֶן אֶל הַהוֹן תִּשְׁעֵינָה

וּבְהֵיכָלֵי הַחֹשְׁקִים תְּסֻבֵּינָה

וְיָבֹאוּ אֶל קִרְבֶּנָה

וְלֹא נוֹדַע כִּי בָּאוּ אֶל קִרְבֶּנָה

שָׁם תִּרְבַּצְנָה

בְּנָוֶה טוֹב וּבְמִרְעֶה שָׁמֵן תִּרְעֵינָה

וְלוּלֵי עָנְיָם וּמְרוּדָם

וְכִי יָבֵשׁ הָיָה נִקֻּדִים לֶחֶם צֵידָם

לֹא הֵמִירוּ בְקָלוֹן אֶת כְּבוֹדָם

וְעָמְדוּ הַרְחֵק מְאֹד מֵאָדָם

יְפַתּוּם הַחֹשְׁקִים בְּעַזּוּתָם

וּבְשִׁקְרֵיהֶם וּבְפַחֲזוּתָם

וּמֵהִלָּחֵם עֲלֵיהֶם לֹא יָמִישׁוּ

וּבְאַהֲבָתָם יַחֲרִישׁוּ

כְּנָחָש יִשְּכוּ וּכְצִפְעֹנִי יַפְרִישׁוּ

יַאֲכִילוּם וְיַשְׁקוּם וְכָל

מַעֲרֻמֵּיהֶם הִלְבִּישׁוּ

וּבָזֶה עֲצַת הַצְּבִיּוֹת הַדַּלּוֹת יָבִישׁוּ

וּבְתִתָּם סִפָּם אֶת סִפָּם

וּצְמִידִים עַל יְדֵיהֶם וּנְזָמִים עַל אַפָּם

וּבְתוֹרֵי זְהָבָם וּבִנְקֻדּוֹת כַּסְפָּם

וּבְהַמְצִיאָם בְּבֵיתָם טַרְפָּם

יִהְיוּ הַחֹשְקִים מַמְצִיאִים שֶׁבֶר רַעֲבוֹנָם

וְהֵם תִּרְצֵינָה אֶת עֲוֹנָם

וְלָכֵן בִּימֵי חָרְפֵּנוּ אֲנִי וְרֵעַי הַחֹשְקִים

הֶעֱבַרְנוּ קוֹל בָּרְחוֹבוֹת וּבַשְׁוָקִים

לְבַשֵּׂר הַצְּבִיּוֹת הַדַּלּוֹת וְהַיָּפוֹת

כִּי צֹאנֵנוּ מַאֲלִיפוֹת

וְחִכֵּנוּ מַמְתַּקִּים

וּמְזָוֵינוּ מְלֵאִים מִזַּן אֶל זַן מְפִיקִים

וְזֶה נוּסַח שִׁיר עֲלִיצוּתֵנוּ

אֲשֵׁר שָׁלַחְנוּ בְּכָל אַרְצֵנוּ

צְבִיַּת חֵן הִלְכִי

וְעֹז יַעְלוֹת דִּרְכִי

וְכַמַּיִם שִׁפְכִי

אַל תִּתְמַהּ דּוֹד אֶל אִישׁ חֵיקִי

בְּתִתּוֹ לִי סֵמֶל הַקִּנְאָה הַמַּקְנֶה

וְאֵלַי מִתְאַנֶּה

וְלֹא לָחַקְתִּי רַק בָּשְׁנָה

וְלֹא הָיְתָה לִי עֶדְנָה

אָמַרְתִּי אַךְ דְּבַר שְׂפָתָים

לֹא תָּקוּם צָרָה פַעֲמָים

וָאֶרְאֶה רֶכֶב אֱלֹהִים רִבּוֹתַיִם

שְׁנַיִם אִישׁ וְאִשְׁתּוֹ

מִתְעַלְּסִים בָּאֳהָבִים אֶחַד בְּאֶחָד בְּגִשְׁתּוֹ

וַאֲנִי הַצְּבִיָּה נִלְכַּדְתִּי בְּרִשְׁתּוֹ

וַתִּשָׂא מְשָׁלָהּ וַתֹּאמַר:

אֵיךְ אֶעֱצָר-כֹּחַ וְגָדַל עִצְּבוֹן

לִבִּי וְאֵין דָּבָר לְנַפְשִׁי יֶעֱרָב

מִי יִתְּנָה אָלִין בְּיַעַר בַּעֲרָב

אִם אֶשְׁכְּנָה מִזְרָח אֲנִי הוּא מַעֲרָב

יִשְׁכֹּן וְעָלַי לֹא יְחַשֵׁב חִשְׁבוֹן

אָאֵר זְמָן חִבֵּר שְׁנֵינוּ וַאֲנִי

עֶגְלָה מְלֻמָּדָה וְלָדוּשׁ אֹהֲבָה

וּלְחֹשְקִים נַפְשִׁי תְסֹעֵר כָּאֳנִי

עַד כִּי פְרָת אֵין דַּי לְכַבּוֹת לַהֲבָה

כִּי אֶזְכְּרָה בִּהְיוֹת בְּקִרְבָּם מַחֲנִי

וּבְחֵיק צְבָאִים אָז שְׁנָתִי עָרְבָה

וַיְהִי כִּשְׁמֹעַ הַשּׂר אֶת שֵׁמַע רָחֵל וְאֶת שִׁירָהּ

חֲמָתוֹ בּוֹ בָּעֲרָה

וַיֹּאמַר זֶה הַשִּׁיר בָּא בַּעֲבֵרָה

אֲרוּרָה הָאֲדָמָה בַּעֲבוּרָה

וַיִּשְׁאָלֵנִי לָדַעַת מֵאוֹדוֹתֶיהָ

אַחֲרֵי אֲשֶׁר שָׁמַע חִידוֹתֶיהָ

וַיֹּאמֶר אֵלָי הַאִם רָחֵל תְּהַתֵּל בְּחֹשְקֶיהָ

וְאִם בָּהּ אֱמוּנָה בָּאִיש אֲשֶׁר שְׁדֵי אַהֲבָה יְנִיקֶהָ

וָאַעַן וָאֹמַר דַּע אֲדֹנִי כִּי בַּעֲלֵי גוֹיִים הָלְמוּ שְׂרוּקֶיהָ

וְכָל רֶגֶל תְּזוּרֶהָ

וְחַיַּת הַשָּׂדֶה תְדוּשֶׁהָ

וְכָל יַד נָבָל תְּמֻשּׁהָ

וּכְטִיט חוּצוֹת יְרִיקֶנָּה

וְכָל פִּי חֲזִיר יִשּׁקֶנָּה

וַיֹּאמֶר הַשָּר אֱלֹהִים נַקֵּה לֹא יְנַקֶּנָּה

וְאֵל הַכָּבוֹד יַרְעִים

וִישַלַּח בָּהּ מִשְׁלַחַת מַלְאֲכֵי רָעִים

וַיִּשָׂא מְשָׁלוֹ וַיֹּאמַר:

אֱלֹהִים בְּרָחֵל בְּרָא חֵיל מְהוּמוֹת

וּמַשְׂאַת כְּלִימוֹת תְּמוּרַת יְקָרָהּ

בְּיַעַן בְּחֹשֵק תְּהַתֵּל וְהִיא בַת

אֲשֶׁר לֹא אֱמוּן בָּהּ וּמַעְלָה יְקָרָה

וְאֵיךְ תֶּהֱרֶה עוֹד וְתֵלֵד אֱמוּנִים

וְהִיא מֵאֱמוּנִים וּמֵחֵן עֲקָרָה

הֲיֵשׁ בַּזְּמָן עוֹד אֱמוּנָה וְשָׁב כֹּל

אֲשֶׁר בַּזְּמָן רַע וְרָחֵל עֲקָרָה

וַיֹּאמֶר הַשָּר אֵלָי הַאִם חָשַׁקְתָּ לְעוֹלָם בְּרָחֵל

וָאַעַן וָאֹמַר לְאַהֲבָתָהּ כָּלוּ עֵיַני מְיַחֵל

וְזֶה כַמָּה אָמַרְתִּי לַסּוֹלֵחַ וּמוֹחֵל

אֶעֱבָדְךָ שֶׁבַע שָׁנִים בְּרָחֵל

עַד אֲשֶׁר רָאִיתִי בָהּ עֶרְוַת דָּבָר

וְנִחַמְתִּי עַל אֲשֶׁר אֲהַבְתִּיהָ בְּשֶכְּבָר

וַיֹּאמֶר הַשּׂר וּמֶה הָיְתָה הַסִבָּה

אֲשֶׁר הֵבִיאָה אוֹתְךָ לָקוּץ בָּהּ

וָאַעַן וָאֹמַר רְאִיתִיהָ יוֹם אֶחָד עַל שָׂפָם עֹטְיָה

וּמְקַלֶּלֶת לַאֲשֶׁר נָשַךְ לֶחְיָּהּ

וָאֶשְׁאַל אוֹתָהּ וָאֹמַר מִי יָעַץ זֹאת עָלַיךְ הַצְּבִיָּה

וַתֹּאמַר עַל אֵלֶּה אֲנִי בּוֹכִיָּה

כִּי זְבוּבֵי מָוֶת נְשָׁכוּנִי בִצְדִיָּה

וַאֲנִי הַפֶּתִי הַמְּפֻתֶּה הֶאֱמַנְתִּי

עַד אֲשֶׁר שִׁנֵּי רְשָעִים בְּלֶחְיָּהּ הִבְחַנְתִּי

וּמֵאָז קַצְתִּי בְאַהֲבָתָהּ

וְאָמַרְתִּי אֶרְאֶה מָה אַחֲרִיתָהּ

וָאֶשָּא מְשָׁלִי וָאֹמַר:

יוֹם אֵרָרָה עָפְרָה לְנֹשֵך לֶחֱיָהּ

נָמֵס לְבָבִי אָז וְהָיָה מָיִם

וָאֹמְרָה הַבַּת יְרוּשָׁלַיִם

מִי זֶה אֲשֶׁר חָשַׁב עֲשׂוֹת כֵּן רַעְיָה

אָז קָפְצָה עָפְרָה וְנִשְׁבָּעָה בְיָהּ

עֶלְיוֹן אֲשֶׁר נָטָה כְּדֹק שָׁמָיִם

לִזְבוּב אֲשֶׁר בִּקְצֵה יְאוֹר מִצְרַיִם

אָאֹר נְשָׁכַנִי וְשַׁבְתִּי עֹטְיָה.

אָז הֶאֱמַנְתִּי עַד אֲשֶׁר רָאִיתִי

שִׁנֵּי רְשָׁעִים שֹׁקְדִים עַל הַלְּחִי

אָז אָמְרָה נַפְשִׁי אֲנִי עָנִיתִי

קַצְתִּי בְיָפְיָהּ אָז וְשַׂמְתִּיהָ סְחִי

הִיא שָנְתָה גַּם כֵּן אֲנִי שָׁנִיתִי

לוּ קוֹמְמִיוּת הֹלְכָה אָמַר שְׁחִי

וַיֹּאמֶר אֵלַי הַשּׂר אֱמָר-נָא בְּחַסְדָּךְ

אֵיךְ הָיָה הַמַּעֲשֶׂה אֲשֶׁר קָרָה לְךָ עִם אֱוִיל מְרֹדַךְ

הַמְדַקְדֵּק הַגָּדוֹל מֵאָרֶץ חַדְרָךְ

כַּאֲשֶׁר רָקַק בְּשִׁירָךְ

וְאִם יָכֹלְתָּ עַל אִשְׁתּוֹ בְחָכְמָתָךְ

וְאִם נִלְכְּדָה בִמְצוּדָתָךְ

כִּי שָׁמַעְתִּי אוֹמְרִים

כִּי הָיְתָה מְלֵאָה הִדּוּרִים

וָאַעַן וָאֹמַר דַּע אֲדֹנִי כִּי נַפְתוּלֵי אֱלֹהִים נִפְתַּלְתִּי

וְקֶרֶן הַצְּבִיּוֹת בֶּעָפָר עֹלַלְתִּי

וְכָל עֵץ גָּבֹהַ הִשְׁפַּלְתִּי

וְחֵל וְחוֹמָה הֶאֱבַלְתִּי

וְעִם אֵשֶׁת אֱוִיל מְרוֹדַךְ תִּירוֹשִׁי הֶחְדַלְתִּי

וְסַבּוֹתִי אֲנִי לְיָאֵשׁ אֶת לִבִּי בֶעָמָל שֶׁעָמַלְתִּי

כִּי בַּעְלָהּ הָיָה הוֹלֵךְ עִמָּהּ יָד לְיָד

וְעַל כֵּן נִמְלְטָה מִיַּד הַצַּיָּד

וּבְבוֹאוֹ אֵלֵינוּ נוֹדֵד מִמְקוֹמוֹ

הוּא וְאֵשֶׁת חֵיקוֹ עִמּוֹ

הָלַכְתִּי אֵלָיו וְשָׁאַלְתִּי בִשְׁלוֹמוֹ

וְנִפְלֵאתִי עַל יְפִי אִשְׁתּוֹ וְיִפְעָתָהּ

בִּרְאוֹתִי אֶת יְקַר תִּפְאֶרֶת גְּדוּלָתָהּ

וּשְׁאֵלַנִי הָאִישׁ וַיֹּאמַר מִי הוּא בָעִיר הַזֶּה הָרִאשׁוֹן

בְחָכְמַת הַשִּׁיר וְהַמְּלִיצָה וּבְדִקְדּוּק הַלָּשׁוֹן

כִּי לִי שְׁאֵלוֹת נִפְלָאוֹת

וְחֲדָשׁוֹת נוֹרָאוֹת

לֹא תִּשְׂבַּע עַיִן לִרְאוֹת

וּבְיָדִי הַמּוֹפֵת וְתִרְאֶה הָאוֹת

וָאֹמְרָה אֶל לִבִּי אֵין לִי טוֹב כִּי אִם לְהִתְעָרֵב עִמּוֹ בִּמְלַאכְתּוֹ

וּבֵין כָּך אֶצְטַיֵּד מִיפִי אִשְׁתּוֹ

וָאֹמַר אֶל הָאִישׁ דַּע אֲדֹנִי כִּי אֲנִי רִאשׁוֹן לַמְּדַקְדְּקִים

וְלַשִּירִים הַחֲזָקִים יְרוּצוּן כַּבְּרָקִים

חוּדָה חִידָתְךָ וְאֶשְׁמָעֶנָּה

אוֹ אָחוּד חִידָתִי וְאוֹדִיעֶנָּה

וְהִנֵּה קָמָה אֲלֻמָּתִי וְגַם נִצָּבָה

יַחְדָּו לַמִּשְׁפָּט נִקְרָבָה

אִם תּוּכַל הֲשִׁיבֵנִי עָרְכָה לְפָנַי הִתְיַצָּבָה

וָאֶשָּא מְשָׁלִי וָאֹמַר:

רַגְלִי בְרִצְפַת שִׁיר בְּמִישׁוֹר עָמְדָה

קָמָה אֲלֻמָּתִי וְגַם נִצָּבָה

רַבָּה צְבָא שִׁירִים וְשִׁירוֹת הַצְּבִי

עָרְכָה לְמוּל פָנַי וְהִתְיַצָּבָה

וַיַּעַן הָאִישׁ וַיֹּאמַר חַיֶּיךָ גַּם לִי לֵבָב כָּמוֹךָ אַמִּיץ וְחָזָק

רָצוֹא וָשׁוֹב בְּמַעֲרֶכֶת הַשִּיר כְּמַרְאֵה הַבָּזָק

וְאִם דּוּדָאֵי מְשֹׁרְרֵי הַזְּמָן נָתְנוּ רֵיח

וְעָלוּ בְמַעֲלָתָם עַד גַּלְגַּל הַיָּרֵחַ

עָלִיתִי אֲנִי עַד גַּלְגַּל עֲרָבוֹת

וּבְעֵת נָתְנוּ בְנוֹת הַשִּיר לַמְשֹׁרְרִים הָאֲלָפִים לִי נָתְנוּ הָרְבָבוֹת

וַיִּשָּא מְשָׁלוֹ וַיֹּאמַר:

לְקוֹל שִׁירַי הֲכִי יִשָּמְעוּ לִי

בְּנֵי הַשִּׁיר וְכָל יֹעֵץ נְדִיבוֹת

לְשָׂרִים נָתְנוּ אֶת הָאֲלָפִים

בְּנוֹת הַשִּׁיר וְלִי נָתְנוּ רְבָבוֹת

וַיֵּאוֹת הָאִישׁ עִמִּי לְהִתְוַכֵּחַ

וַאֲנִי נָתַתִּי לוֹ מָקוֹם לִהְיוֹת נֹצֵחַ

כִּי יְפִי אִשְׁתּוֹ עָמַד עָלַי כְּרֹצֵחַ

וְהוּא לְהַרְבּוֹת שְׁאֵלוֹתָיו רַעְיוֹנָיו יָחוּשׁוּ

וְעֵינַי מִנֶּגֶד עֵינֵי אִשְׁתּוֹ לֹא יָמוּשׁוּ

וְנֶעְתַּקְנוּ מִמַּאֲמָר אֶל מַאֲמָר

וֶאֱוִיל מְרוֹדַךְ לֹא נִשְׁמָר עַד אֲשֶׁר נָפַל כְּתוֹא מִכְמָר

בְּאָמְרוֹ כִּי לֹא נִמְצָא שַׁדְרַךְ רַק בְּדָנִיֵאל בִּלְבָד

וּכְשָׁמְעִי אָמַרְתִּי בְּלִבִּי אַךְ עָשַׁרְתִּי מָצָאתִי אוֹן נִכְבָּד

וָאֹמְרָה אֵלָיו חַיֶּיךָ הַמְדַקְדֵּק לֹא כֵן דִּבַּרְתָּ

וְלֹא יָקָם לְךָ אֹמֶר גָּזַרְתָּ

כִּי דַּע אֶל נָכוֹן כִּי דְּבָרִים בְּגוֹ

וְחֵי הַחֵשֶׁק הַמְצָרֵף לִבִּי וּמְבָרֵר סִיגוֹ

וְגַלְגַּלֵי אֲמָרַי סוֹבְבִים עַל חוּגוֹ

כִּי לוּ דִקְדַּקְתָּ מָצָאתָ בְחֵיק רַעְיָתְךָ שַׁדְרַךְ מֵישַׁךְ וַעֲבֵד נְגוֹ

אֵלֶּה הָיוּ דְבָרַי עִם אֱוִיל מְרוֹדַךְ

וְהִנְנִי אֶשָׂא עָלָיו שִׁיר זֶה לְמַעַן כְּבוֹדָךָ

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

שְׁמָעֵנִי נְשִׂיא דוֹרָךָ

כְּסִיל מִיַעֲקֹב דָּרָךְ

בְּחִירִי רָצְתָה נַפְשִׁי

שְׁמַע מִלַּי וְתִתְבָּרָךְ

תְּמוֹל חָנָה בְעִירֵנוּ

מְדַקְדֵּק מִבְּנֵי חַדְרָךְ

וְלוֹ עָפְרָה מְהֻלָּלָה

יְפַת תֹּאַר וְגוּף נֶעְרָךְ

וְשָׁדֶיהָ כְרִמּוֹנִים

לְרִיבִי הַזְּמָן עָרָךְ

וְאָמַרְתִּי לְלִבִּי זֹאת

בְּצָרָתָךְ תְהִי עֶזְרָךְ

שְׂעַר רֹאשָׁהּ בְּחַרְתִּיהוּ

לְתִתּוֹ צִיץ עֲלֵי נִזְרָךְ

וְאוֹר לֶחְיָהּ לְאוֹר יוֹמָךְ

וְלַיְלָה תִהְיֶה נֵרָךְ

אֲנִי מַבִּיט בְּחֵיק עָפְרָה

פְּלִיאָה מַעֲשֵׂה יֹצְרָךְ

וְהוּא יַרְבֶּה בְּבִלְבּוּלָיו

וְהוּא הִקְשָׁה וְהוּא פֵרָךְ

וְהִרְחִיב פֶּה בְּתוֹך עֵדָה

וְנָשָׂא קוֹל וּפֶה אָרָךְ

וְאָמַר כִּי בְדָנִיֵאל

לְבַדּוֹ נִמְצְאָה שַׁדְרָךְ

אֲזַי מִסַּעֲרַת חִשְׁקָהּ

אֲשֶׁר לִבִּי מְאֹד הֵרָךְ

עֲנִיתִיו דּוֹק וְתִשְׁכַּח כִּי

בְחֵיק עָפְרָתְךָ שַׁד-רָךְ.

וַיֹּאמֶר אֵלַי הַשָּׂר הוֹדַעְתַּנִי הַמַּעֲשֶׂה שֶׁהָיָה

וְלֹא הוֹדַעְתַּנִי תְשׁוּבַת אֱוִיל מְרֹדַךְ בַּעַל הַצְּבִיָּה

כַּאֲשֶׁר הִכִּיתָ אוֹתוֹ בִּצְדִיָּה

פֶּצַע וְחַבּוּרָה וּמַכָּה טְרִיָה

וְאֵיךְ הָיָה כַּאֲשֶׁר שִׁנָיו חָרַק

וּבְשִׁירָתְךָ יָרֹק יָרַק

וָאַעַן וָאֹמַר דַּע אֲדֹנִי הַמְבֹרָךְ

כִּי כַּאֲשֶׁר הִגַּדְתִּי אֶל הַמְדַקְדֵּק מֵחַדְרָךְ

כִּי בְחֵיק עָפְרָתוֹ שַׁד-רַךְ

הִתְקַצֵּף וְקִלְלַנִי קְלָלָה נִמְרֶצֶת

תִּהְיֶה בְרֹאשׁוֹ מְקַנֶּנֶת וְרוֹבֶצֶת

וַיֹּאמַר אָרוּר יִהְיֶה זֶה הָאִישׁ אֲשֶׁר לְנֶגְדִּי רִיב עָרָךְ

וּכְשָׁמְעִי אָמַרְתִּי לְלִבִּי חֶלְמָא לְשָׂנְאָךְ וּפִשְׁרֵהּ לְעָרָךְ

וְהַקְּלָלוֹת אֲשֶׁר קִלֵּל נֶגְדָּךְ

יָחוּלוּ עַל רֹאש אֱוִיל מְרֹדַךְ

וַיְהִי כַּאֲשֶׁר עָלָה עָשָׁן בְּאַפּוֹ

וְלֹא עָמַד יַם חֲרוֹנוֹ מִזַּעְפּוֹ

וְאֵין חֲזוֹן שִׁירָיו מֵאָז נִפְרָץ

נִפְרַדְתִּי מֵאִתּוֹ וָאֶהְיֶה אֶל בֵּיתִי רָץ

וָאֹמַר אֶל לִבִּי זֶה הַכֶּלֶב אֲשֶׁר עָלַי לָשׁוֹן חָרָץ

אַחֲרֵי אֲשֶׁר רוּחַ עִוְעִים מְבַעְתּוֹ

אוֹסִיף תּוּגָה עַל תּוּגָתוֹ

וַאֲדַבֵּר רַק מֵאֹדוֹת אִשְׁתּוֹ וּבְכֵן יָשֹׁם עַל עֵקֶב בָּשְתּוֹ

וָאָקוּם וָאחַבֵּר שִׁיר נִמְרָץ מִלּוּחַ לִבִּי חֲקַקְתִּיו

וְעַל יַד נַעַר אֶחָד אֶל הַמְדַקְדֵּק שְׁלַחְתִּיו

וְאֵלֶּה דִּבְרֵי שִׁירִי

אֲשֶׁר שָׁלַחְתִּי אֶל הַמְדַקְדֵּק עָרִי

לְקֹט שׁוֹשַׁן לְחִי עָפְרָה צְבִי חֵן

וְשׁוּר שֶׁמֶש בְּפָנֶיהָ מְצַחֵק

וְאִם תִּתְאַו מְצֹא שַׁד-רַךְ וּמֵישַׁךְ

וְעִמָּם בַּלְטְשַׁאצַר קוּם מְצָא חֵק

וַיְהִי כִּרְאוֹת הַמְדַקְדֵּק אֶת הַשִּיר וּמַעֲנָיו

יָרֹק יָרַק בְּפָנָיו

וַיֹּאמֶר אֶל הַנָּעַר כֹּה תֹאמַר לַאֲדֹנֶיךָ הַנָּבָל

אֲשֶׁר לִקְבָרוֹת יוּבָל

כִּי לוּ הָיָה בַמַּעֲמָד שִׁלַּחְתִּי רָזוֹן בְּמִשְׁמַנָּיו

וְכַאֲשֶׁר יָרַקְתִּי בְשִׁירָתוֹ יָרַקְתִּי בְפָנָיו

וַיָּבֹא הַנַּעַר אֵלַי בּוֹש וְנִכְלָם

וַיֵּשֶׁב דּוּמָם וְנֶאֱלָם

וְהָיְתָה לוֹ חֶרְפַּת עוֹלָם

וָאֹמַר אֵלָיו מַה לְּךָ נִרְדָּם בֵּן מַחְפִּיר וּמַבְאִישׁ

וַיֹּאמֶר אֵלַי הַנַּעַר כֹּה דִּבֵּר אֵלַי הָאִישׁ

וְאֵיךְ לֹא אֶעֱמֹד מַשְׁמִים

יָרֹק יָרַק בְּפָנֶיהָ הֲלֹא אִכָּלֵם שִׁבְעַת יָמִים

וָאָקוּם בַּחֲרוֹן אַפִּי וָאֲחַבֵּר שִׁירָה

לֹא יִשּׁקֵל כֶּסֶף מְחִירָהּ

וָאֶשְׁלַח אֶל הַמְדַקְדֵּק שֵׁנִית

לְהָסֵב אֶת לִבּוֹ אֲחוֹרָנִית

וְאֵלֶּה דִּבְרֵי שִׁירָתִי הַשְּׁלוּחָה אֶל המדקדק מֵאִתִּי

עוּרָה כְבוֹדִי מַה לְךָ נִרְדָּם

מוּל בֹּעֲרִים בָּעָם בְּרֹק בָּרָק

וּפְגַע בְּמִשְׁתּוֹלֵל אֲשֶׁר חִלֵּל

יִפְעַת מְלִיצָתִי בְּרֹק בָּה רָק

לוּ סִיסְרָא נֶגְדִּי אֲנִי יָעֵל

יָבִין וְעַמּוֹ אַךְ בְּרֹק בָּרָק

וַיְהִי כִּרְאוֹת אֱוִיל מְרֹדַךְ

כִּי הָיִיתִי בְעִנְיַן הַשִּׁיר לוֹ רֵעַ

וְהָיִיתִי בְעֵינָיו כִּמְתַעְתֵּעַ

וְעַל יְפִי אִשְׁתּוֹ מִשְׁתַּגֵּעַ

כִּסְּתָה פָנָיו כְּלִמָּה

וַיִּפֶן וַיֵּלֶךְ בְּחֵמָה

וְלִבִּי אַחֲרֵי אִשְׁתּוֹ נִמְשָׁךְ

וְאֵשׁ אַהֲבָתָהּ בְּקִרְבִּי כְּנָחָשֹ יִשָּךְ

וַיֹּאמֶר הַשָּׂר וּמֶה הָיָה עִנְיַן אִשְׁתּוֹ

כַּאֲשֶׁר שָׁמְעָה חֶרְפַּת בַּעֲלָהּ וּבָשְׁתּוֹ

וָאֹמַר חַיֶּיךָ הַשָּׂר כִּי כַאֲשֶׁר קָצַף עָלַי בַּעֲלָהּ

אָז הַצְּבִיָּה הַמְהֻלָּלָה

אֲשֶׁר אֵין עַל עָפָר מָשְׁלָהּ

נָתְנָה עָלַי בְּקוֹלָהּ

וְקָרְאָה אַחֲרַי הֵידָד

וּכְשָׁמְעִי אוֹתָה אָמַרְתִּי אָהֳלִי שֻׁדָּד

אַחֲרֵי אֲשֵׁר לֵב הַצְּבִיָּה מִדֵּי דַבְּרָהּ בִּי הִתְנוֹדָד

וָאֶשָׂא עֵינַי לִרְאוֹת הַחֲשׁוּקָה

וָאֵרֶא וְהִנֵּה יָדָהּ אֵל תּוֹךְ חֵיקָהּ

תִּתְנַדֵּב לִי בְּשַׁד-רַךְ אֲשֶׁר שִׁבַּחְתִּי

וְנֶפֶשׁ בְּעָלֶיהָ הִפָּחְתִּי

וָאֶקֹד וָאֶשְׁתַּחֲוֶה אֶל הַצְּבִיָּה

בְּנֶפֶשׁ כּוֹרַעַת וּמְדַבֶּרֶת וְלָשׁוֹן דֻּמִיָּה

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

נָסוֹג אָחוֹר לִבִּי בִּרְאוֹת

עָפְרָה תִקְרָא אַחְרַי הֵידָד

עַד רָאִיתִי יָדָהּ אֶל תּוֹך

חֵיקָהּ תֹּאמַר לִי דּוֹד הֵא דָד

וַיִּתְמַהּ הַשּׂר אֶל הַשִּׁירָה הָאֲיֻמָּה

וַיֹּאמֶר חַי אֲנִי כִּי הִגְדַּלְתָּ וְהוֹסַפְתָּ חָכְמָה

עַל כָּל הַמְשֹׁרְרִים אֲשֶׁר קְדָמוּךָ

וּמְלִיצֵי הַזְּמָן לֹא עֲמָמוּךָ

בָּרוּךְ אֲשֶׁר חָלַק לְךָ מִכְּבוֹדוֹ

וְאָצַל עָלֶיךָ מֵהוֹדוֹ

וָאֵשֵׁב עִם הַשּׂר בְּחַדְרֵי מַשְׂכִּיתוֹ

וַאֲרֻחָתִי אֲרֻחַת תָּמִיד נִתְּנָה לִי מֵאִתּוֹ.

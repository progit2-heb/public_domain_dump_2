מחברות עמנואל: מחברת שבע-עשרה

אָמַר הַמְּחַבֵּר: כַּאֲּשֶׁר שַׁבְתִי עִם הַשָּׂר מִבֵּית הַגְבִירָה הַהֲדוּרָה / אֲשֶׁר לֹא יֻתַּן סְגוֹר תַּחְתֶּיהָ וְלֹא יִשָּׁקֵל כֶּסֶף מְחִירָהּ / הָיָה לְבַב הַשָּׂר כְּמוֹ תַּנּוּר בּוֹעֵרָה / נִלְהָב מֵחִשְׁקָהּ וּבוֹעֵר / זַעֲקַת שֶׁבֶר יְעוֹעֵר / וּכְמוֹץ מִגֹּרֶן יְסוֹעֵר / וּבְכָל מַחֲמַדֵּי הַזְּמָן גּוֹעֵר / וַיְהִי הַיּוֹם וַיֹּאמֶר אֵלַי הַשָּׂר אַשְׁבִּיעֲךָ בִּבְרִית הָאַהֲבָה אֲשֶׁר בֵּינֵינוּ / לָלֶכֶת אֶל בֵּית הַגְּבִירָה שְׁנֵינוּ / כִּי מֵחִשְׁקָהּ נִלְהַבְתִּי / וְאַהֲבָה עַזָּה אוֹתָהּ אָהַבְתִּי / וְשִׁירֶיהָ מָצְאוּ חֵן בְּעֵינַי / וְשַׂמְתִּי בְלִבִּי לְשַׁנֵּס מָתְנַי / וּלְעוֹרֵר רַעְיוֹנַי / לְרֶגֶל הַמְּלָאכָה אֲשֶׁר לְפָנַי / וּלְחַבֵּר שִׁירִים עַל אַהֲבָתָהּ אוּלַי תִּשָּׂא פָנָי / וְגַם אַתָּה הִכּוֹן וְהָכֵן לְךָ שִׁירִים צְרוּפִים בַּעֲלִיל הַיֹּפִי / וּמְזֻקָּקִים שִׁבְעָתַיִּם מֵאֵין דֹּפִי / יִהְיוּ שְׁקוּלִים בְּמֹאזְנַיִם / נֶחְמָדִים לְהַשְׂכִּיל וְתַּאֲוָה הֵם לָעֵינָיִם / יְבַקְּעוּ לֶב-הַגְּבִירָה לִשְׁנָיִם /וּנְגַלֶּה אֵלֶיהָ מַצְפּוּן לִבֵּנוּ / וְאֵשׁ הַחֵשֶׁק הַטָּמוּן בְּחֻבֵּנוּ / אוּלַי תָּקֵּל מֵעָצְבֵּנוּ /

אָז אָמַרְתִּי אֶל הַשָּׂר שִׂים לַדֶּרֶך פְּעָמֶיךָ / וְהַט שָׁמֶיךָ / וְהוֹצֵא פְנִינֶיךָ וּלְשָׁמֶיךָ / וְגִנְזֵי בְרוֹמֶיךָ / מִבֵּין אֲסָמֶיךָ / וְאַחֲרֶיךָ לְבָבִי בַעֲמָמֶיךָ / וְעַל הָרֵי הַצְּבִיָּה תָּרִים נִסֶּיךָ /כְּעַמִּי כְעַמֶּךָ כְּסוּסַי כְּסוּסֶיךָ /

אָז קַמְנוּ וְהָלַכְנוּ אֶל בֵּית הַצְּבִיָּה הַיָּפָה / פּוֹרִיָּה וַעֲנֵפָה / לֶאֱרוֹת יַעְרַת דִּבְשָהּ וְצוּפָהּ / וּמְצָאנוּהָ בְעַד הַחַלּוֹן נִשְׁקָפָה / תַּחְפִּיר הַשֶּׁמֶש בְּצֵאתוֹ / וּמֵאֵן הַשֶָּׁמֶש לְהַרְאוֹת בְּפָנֶיהָ יִפְעַתוֹ / וּבַעֲלָהּ חֶרְפַּת אָדָם וּבְזוּי עָם / הָיָה נִצָּב עַל אֵם הַדֶּרֶך כְּאִישׁ נִפְעָם / וְעִמָּהּ דִּבְרֵי רִיבוֹת מְדַבֵּר / אָבֵל וַחֲפוּי רֹאשׁ כְּאִישׁ אָבִיו מְקַבֵּר/

וָאֹמַר אֶל הַשָּׂר אָרוּר הָאִישׁ אֲשֶׁר זֶה הַחִבּוּר חִבֵּר / זֶה הוּא בַעַל הַיָּפָה פֶתַע יִשָּׁבֵר / וְעַתָּה חֲיָלִים נְגַבֵּר / וְדַלְתוֹת נְחוּשָׁה נְשַׁבֵּר / וְעָרֹם נַעֲרִים / וְנַעֲשֶׂה עַצְמֵנוּ כְּמִתְנַכְּרַים / וּכְאִלּוּ לֹא נַכִּירֵהוּ נִהְיֶה / מֵאֹדוֹת הַגְּבִירָה בְפָנָיו מְדַבְּרִים / וְנַחְשְׁבֵהוּ לְחֹשֶׁק / וּלְאִישׁ אֵשֶׁת אֲחֵרִים גּוֹזֵל וְעוֹשֵׁק / וּנְאַשְרֵהוּ אֲשֶׁר יַחְשֹׁק גְּבִירָה כָזֹאת/ וּנְדַבֵּר לְפָנָיו דִּבְרֵי חֵשֶׁק בַּחֲרוּזוֹת / וְנַעֲשֶׂה אֶל הַגְּבִירָה רְמִיזוֹת / לִרְאוֹת אִם הוּא כוֹסֶה קָלוֹן אוֹ יַעֲנֶה עַזּוֹת /

וַיֹאמֶר הַשָּׂר עֵת לְנַסּוֹת / טוֹב הַדָּבָר אֲשֶׁר דִּבַּרְתָּ לַעֲשׂוֹת /

וַיְהִי כַּאֲשֶׁר קָרַבְנוּ אֲלֵיהֶם וַיֹאמֶר הַשָֹּר שָׁלוֹם לְשַׁר הַחֹשְקִים / וְשָׁלוֹם לְשׁוֹשַָנַּת הָעֲמָקִים /

וַתִּשְׂחַק הַגְּבִירָה וַתֹּאמַר / וַעֲלֵיכֶם שָׁלוֹם הַחֹשְקִים הַחֲשׁוּקִים / אֲשֶׁר בְּלֵב הַיְּעֵלוֹת חֲקוּקִים /

וַיַּעַן בַּעַל הָאִשָּׁה הַנִּרְצָחָה / אַחֲרֵי אֲשֶׁר הִתְאַנַּחַ מִלִּבּוֹ אֲנָחָה / וַיֹּאמַר רָאִיתִי הַיּוֹם כִּי חֲשַׁבְתּוּנִי לְחֹשֵׁק / וּלְאֵשֶׁת אֲחֵרִים גּוֹזֵל וְעוֹשֵׁק / אֵינֶנִּי כַאֲשֶׁר חֲשַׁבְתּוּנִי / וְזֶה לִי הָאוֹת כִּי לֹא הִכַּרְתּוּנִי/

וַיֹאמֶר הַשָּׂר אַל תֶּחֱרַד הַדּוֹד וְאַל תִּשֹּׁם / יְדַעְנוּךָ לְחֹשֶׁק מִתְּמוֹל שִׁלְשֹׁם / וְאֵין לִתְּמֹהַּ אִם חָשַׁקְתָּ בִּמְלֶכֶת הַשָׁמַיִם / כִּי גַּם אֲנַחְנוּ חֲשַׁקְנוּהָ לֹא יוֹם אֶחָד וְלֹא יוֹמָיִם / וְלוּלֵא הִתְמַהְמָהנוּ כִּי אַתָּה שַׁבְנוּ זֶה פַּעֲמַיִם / וְעַתָּה1 הִנֵּה אֲנַחְנוּ פֹה שְׁלֹשָׁה חֹשְׁקִים / בִּכְנַף הַיְּעֵלָה מַחֲזִיקִים / בָּזֹאת נִבָּחֵן / אִם בְּעֵיֵני הַצְּבִיָּה נִמְצָא חֵן / יְחַבֵּר כָּל אֶחָד מִמֶּנּוּ עַל חֵשֶׁק הַגְּבִירָה שִׁיר נֶחְמָד / אֲשֶׁר כָּל הַבָּא אֵלַי וְעָמָד / וְהָיָה הָאִישׁ אֲשֶׁר יִהְיֶה שִׁירוֹ כָלִיל בְּיָפְיוֹ וּבַהֲדָרוֹ / נַנִּיחַ לוֹ הַגְּבִירָה / כִּי לוֹ מִשְׁפַּט הַבְּכוֹרָה / וַאֲנִי אָשִׂים פָּנַי בָּרִאשׁוֹנָה לְהַלֵּל יִפְעָתָהּ / אוּלַי יִישָׁר בְּעֵינֶיהָ וְאֶשְׁכּוֹן עַל הָרֶיהָ וְעַל גִּבְעָתָהּ /

וַיִּשָּׂא מְשָׁלוֹ וָיֹאמַר

לֹא הֶאֱמַנְתִּי אוֹמְרִים לִי אֶמֶשׁ

יֵש יַעֲלָה תִּלְעַג עֲלֵי יָרֵחַ

עַד אֶחֱזֶה הָאוֹמְרָה לַשֶּׁמֶשׁ

עֵת יַעֲלֶה מִזְרָח עֲלֵה קֵרֵחַ.

וַיְהִי בְּהַשְׁלִימוֹ הַשִּׁירָה / וַיֹאמַר הִנֵּה הִלַּלְתִּי הַגְּבִירָה הַהֲדוּרָה / וְעַתָּה יּאמַר כָּל אֶחָד מִכֶּם שִׁירוֹ / וְנִרְאֶה לְמִי תִפְאֶרֶת הַשִּׁיר וַהֲדָרוֹ / וָאַעַן וָאֹמַר חַי אָנִי אִם בְּשִׁיר נְרַצֵּה הַגְּבִירָה קָרוֹב צִדְקִי / אֶעֱנֶה אַף אֲנִי חֶלְקִי / כִּי גַם לִי שִׁירִים כַּבְּרָקִים יְרוֹצֵצוּ / וּכְעֵין נְחֹשֶׁת קָלָל יְנוֹצֵצוּ / וְלִבּוֹת הָאֶבֶן בִּנְקַלָּה יְפוֹצֵצוּ /

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

עָפְרָה תַשְׁקִיף מִן הָאֶשְׁנָב

וּבְעֵינֶיהָ יוֹנָה אוֹ תוֹר

לִבִּי תַצִּיב כַּמַּטָּרָה

תַּרְעִישׁ סִפָּיו הַךְ הַכַּפְתּוֹר

כָּעֵת חִצָּהּ תוֹרֶה צִדָּהּ

תֹּאמַר תִּשְׁמַע הַחְלוֹם לִפְתוֹר.

וַיְהִי בְּהַשְׁלִימִי הַשִּׁירָה / וָאֹמַר אֶל בַּעַל הַגְּבִירָה / אֱמֹר נָא הַחֹשֵׁק גַּם אַתָּה שִׁירָךְ / הִשָּמֵר וְהַשְׁקֵט אַל תִּירָא וּלְבָבְךָ אַל יֵרַךְ / כִּי אִם תִּגְבַּר יָדְךָ עָלֵינוּ בַּשִּׁירָה / נַנִּיחַ לְךָ חֵשֶׁק הַגְּבִירָה/

וַיֹאמֶר הָאִישׁ חַיֵיכֶם הַחֹשְׁקִים הַנְּעִימִים / בּוֹקֵר אָנֹכִי וּבוֹלֵס שִׁקְמִים / אֵינֵנִּי חֹשֵׁק וְלֹא בֶן חֹשֵׁק / וְלֹא אֵשֶׁת אֲחֵרִים גּוֹזֵל וְעוֹשֵׁק / אָמְנָם דְּבַר סֵתֶר הָיָה לִי אֶל הַגְּבִירָה / וְחָלִילָה לִי מִדְּבַר חֵשֶׁק וְדִבּוּר סָרָה / כִּי הַחֵשֶׁק גִּנּוּהוּ הַחֲכָמִים וְאָמְרוּ בַעֲדוֹ שֶׁהוּא מַפְתֵּחַ כָּל הָאֲשָׁמִים / וְרֶסֶן מַתְעֶה עַל לֶחְיֵי עַמִּים / וּלִבְעָלָיו לְשׁוֹן סֵתֶר וּפָנִים נִזְעָמִים / וְהַדּוֹרְכִים דַּרְכּוֹ יֵלְכוּ קְדוֹרַנִּית /יְגַלּוּ עֶרְוַת אֲבִיהֶם וּפְנֵיהֶם אֲחוֹרַנִּית / וְהִיא הַגְּדוֹלָה שֶׁבַּמַּחֲלוֹת וְנִקְרָאָה מַחֲלָה רוּחָנִית / וְשֹׁרְשָהּ בַּמֹּחַ נִטַּעַת / וְהִיא מִכּוֹס הַתַּרְעֵלָה קֻבַּעַת / אֵין טוֹב בְּדַעַת לַחֲשׁוֹק אָכֵן טוֹב לַחֲשׁוֹק בַּדָּעַת /

וְאָמַר הַמְּשׁוֹרֵר:

הַחֵשֶׁק מַחְלָה רוּחָנִית

שֹׁרְשָׁהּ בַּמֹּחַ נִטָּעַת

אֵין טוֹב לָךְ בְּדַעַת לַחְשׁוֹק

אָכֵן טוֹב לַחְשׁוֹק בַּדַָעַת.

וָיֹאמֶר הַשָּׂר וְלָמָּה תַּעְלִים / חִשְׁקְךָ זַרְזִיר מָתְנַיִם / וְאַתָּה חֹשֵׁק מִמֶּנוּ פִי שְׁנַיִם / וּמִי לֹא יַחְשֹׁוק הַבַּת יְרוּשָׁלַיִם / אֲשֶׁר גָּזְלָה נוֹגַהּ וָצֶדֶק מִבֵּין כּוֹכְבֵי הַשָׁמַיִם / וְהָיוּ בָּהּ לְעֵינָיִם / וְזֶה כַמָּה עָמַדְתִּי מִשְׁתָּאֶה / מִמַּרְאֵה עֵינַי אֲשֶׁר בָּהּ אֶרְאֶה / כִּי אַבִּיט הַשָּׁמַיִם וְהִנֵּה נוֹגַהּ וָצֶדֶק אֵינָם / לֹא יָדַעְתִּי אִם גָלוּ בַעֲוֹנָם / וְשָׁאַלְתִּי הוֹבְרֵי שָׁמַיִם אִם יֵדְעוּ מֵאֹדוֹתָם / וַיֹּאמְרוּ כִּי בִּפְנֵי הַגְּבִירָה רָאוּ אוֹתָם / מְפֻתָּחוֹת פִּתּוּחֵי חוֹתָם / וּמִשְׁבְּצוֹת זָהָב בְּמִלֻּאוֹתָם /

וַיִּשַׂא מְשָלוֹ וַיֹאמַר:

אֶפְקֹד צְבָא מָרוֹם וְאֶרְאֶה כִּי

נִפְקַד מְקוֹם צֶדֶק וְאֹור נֹגַהּ

וָאֶשְׁאֲלָה כָּל הוֹבְרֵי שַׁחַק

מִי זֶה וְאֵיזֶה הוּא אֲשֶׁר שָׁגָה

לִגְנֹב כְּהַיּוֹם כֹּכְבֵי מָרוֹם

וּלְהַעֲמִיד אָרָץ עֲלֵי תוּגָה

וַיֹאמְרוּ לִי דוֹד שְׁאֵלָתָךְ

לִמְאֹד חֲקַרְנוּהָ לְהַשִׂיגָהּ

לֹא נֶעֶדְּרוּ צֶדֶק וְנֹגַהּ כִּי

עַל חוּג אֲדָמָה זָהֳרָם יִגַהּ

אַך יָרְדּו הֵמָּה וְשָׁכְנוּ רֹאשׁ

עָפְרָהּ בְּשׁוֹשַׁן הוֹד וְנוֹי סוּגָהּ

שוּר עֵין יְמִינָהּ תֶּחֱזֶה צֶדֶק

שוּר עֵין שְׂמֹאלָהּ תֶחֱזֶה נֹגַהּ.

וְהָיָה כַאֲשֶׁר הִשְׁלִים הַשָּׂר שִׁירָיו / בָּאתִּי אַחֲרָיו וּמִלֵּאתִי אֶת דְּבָרָיו / וָאֹמְרָה חַיֵיכֶם הַחֹשְׁקִים גַּם אֲנִי כָּכֶם אֲדַבֵּר / וּדְבַר חֵשֶׁק עַל אַהֲבַת הַגְּבִירָה אֲחַבֵּר / כִּי הַגְּבִירָה יְדִידוּת נַפְשִׁי / וְאָשִׂים חֵשֶׁק אַהֲבָתָהּ / עֲטֶרֶת לְרֹאשִׁי / וְלוּ תִהְיֶה שְׂמֹאלִי תַּחַת לְרֹאשָׁהּ / וּבִימִינִי אֲחַבְּקֶנָּהּ וְתִהְיֶה נַפְשִׁי קְשׁוּרָה בְנַפְשָׁהּ / תִּהְיֶה לִי כָּל הַנְּעִימוּת לְמֹורָשָׁה / אִינַק וְאֶשְׂבַּע מִשֹּׁד תַּנְחוּמֶיהָ / וְאֶלְחַם בְּמַנְעַמֶּיהָ / וָאָאֳרֶה מוֹרָהּ וּבְשָׂמֶיהָ / וְאִם גַּנָּבִים יְבוֹאוּנִי וְשׁוֹדְדֵי לַיְלָה לָרִיק לֹא יִגָעוּ / יִגְנְבוּ דַּיָּם וְיִשְׂבָּעוּ / וְלֹא אֹמַר סוּרוּ סוּרוּ אַל תִּגָּעוּ /

וָאֶשָּׂא מְשָלִי וָאֹמַר:

לוּ יַד שְׂמֹאלִי תִּהְיֶה תַחַת

רֹאשֵׁךְ צְבִיַת חֵן וּפִי אֶל דַּד

אִינַק וְאִם שׁוֹדֵד יְבוֹאֵנִי

לֹא אֹמְרָה כִּי אָהֳלִי שֻׁדָּד.

וַיְהִי כְּהַשְׁלִימִי הַשִּׁירָה / וַיִחַר אַף בַּעַל הַגְּבִירָה / וַיֹאמַר אוֹי לָכֶם שְׂעִירֵי עִזִּים / אֲנָשִׁים רֵיקִים וּפוֹחֲזִים / וְשׁוֹכְבִים וְחוֹזִים / אֲשֶׁר אָרְחוֹתֵיהֶם עִקְּשִׁים וּבְמַעַגְּלוֹתֵיהֶם נְלוֹזִים / רוֹאֶה אֲנִי אֶת פְּנֵיכֶם לַחְשׁוֹק נֶחְפָּזִים / וְדַרְכֵי הַנְּדִיבוּת בּוֹזִים / כְּאִלּוּ לֹא עָמְדוּ אֲבוֹתֵיכֶם עַל הַר גְּרִיזִים / הֲלֹא יְדַעְתֶּם כִּי לַגְּבִירָה גּוֹאֲלִים וְאַחִים / וְהִיא כְשׁוֹשַֹנָּה בֵּין הַחוֹחִים / וְאֵיך תֶּחֱזוּ לָהּ מַשְׂאוֹת שָׁוא וּמַדּוּחִים / וְלָהּ בַּעַל חוֹתַם תָּכְנִית / וְאִישׁ יִגַּע בָּהֶם יִמְצָא בַּרְזֶל וְעֵץ חֲנִית / יַחַרְשׁוּ עַל גַּב הַחֹשְׁקִים וְיַאֲרִיכוּ מַעֲנִית / וְיָשִׁיבוּ צֵל הַמַּעֲלוֹת אֲחוֹרַנִּית / וּמַה לָּכֶם לְרַחַם רַחֲמָתָיִם / וְהָעַזּוּת עַל פְּנֵיכֶם כְּאַמָּתָיִם / שׁוּבוּ לְאָהֳלֵיכֶם וְתִשְׁכְּבוּן בֵּין שְׁפַתָּיִם / אָמַרְתִּי אַךְ דְּבַר שְׂפָתָיִם / וּמַדּוּעַ נִגַּשְׂתֶּם אֶל הַחוֹמָה/ וְעַתָּה עֵצָה וּגְבוּרָה לַמִִּלְחָמָה / וּמִן הַתֵּמַהּ עֲלֵיכֶם בְּנֵי עוֹנְנָה / כִּי יַחְשְׁבוּ אֶתְכֶם הָרוֹאִים לְאַנְשֵׁי שֵׂכֶל וּתְבוּנָה / וְאֵיךְ תִּתְפַּתּוּ אַחֲרֵי אִשָּׁה תֵרָאֶה יָפָה כַלְּבָנָה / וְעַל דֶּרֶך הָאֱמֶת תִּשְׁכֹּן עָלֶיהָ עֲנָנָה / הֲלֹא הֻגַּד מֵרֹאש לָכֶם כִּי בְּלִבְנַת הַיְּעֵלָה אֵין לִבְטֹחַ / כִּי כַמָּה שְֹחוֹרוֹת שָׁבוּ לְבָנוֹת אַחֲרֵי הִטֹּחַ / וְהֵם לִמְצוֹדִים וַחֲרָמִים מִשְׁטוֹחַ /

וְאָמַר הַמְּשׁוֹרֵר:

דּוֹדִים בְּלִבְנַת יַעֲלָה אִם לֹא

יְדַעְתּוּהָ תְמוֹל אַל תִּבְטָחוּ בָטֹחַ

כִּי רָאֲתָה עֵינִי שְׁחוֹרָה מִ-

שְּׁחוֹר שָׁבָה לְבָנָה אַחֲרֵי הִטֹּחַ.

אֲשֶׁר עוֹד רָאִיתִי בְמַחֲשָבִי / וְנָתוֹן אֶל לִבִּי / כִּי כַמָּה מִן הַכְּעוּרוֹת הָיוּ לְאַדְמַת דֹמֶן / יְיַפֶּה אוֹתָם הַשֹּׁמֶן / וְתַחַת הֱיוֹתָם מֵרוּחַ קָדִים שְׁדוּפוֹת / וּשְׁפָחוֹת חֲרוּפוֹת / תֵּחָשַׁבְנָה בְעֵת יַשְׁמִינוּ מִן הַיָּפוֹת / וְתִהְיֵיןָ כַאֲרָיוֹת טוֹרְפוֹת / וַעֲפַר אֶרֶץ בְּרֹאש הַחֹשְׁקִים שׁוֹאֲפוֹת / וְאָמַר אֶחָד מֵאַנְשֵׁי הַמּוֹעֵצָה / בְּדֶרֶך מָשָׁל וּמְלִיצָה / כִּי בְּלֶכֶת אָנֶה וֶאָנָה אֲרִיסְטוֹ מָצָא הַשֹּׁמֶן מִתְהַלֵּך

לְאִטּוֹ / וַיִשְׁאָלֵהוּ לָדַעַת מִשְׁפָּטוֹ / וַיַעַן כִּי כֹה מֶבָּטוֹ/לְיַפּוֹת הַכְּעוּרוֹת / וּלְהַלְבִּין הַשְּׁחוֹרוֹת / וְכַאֲשֶׁר יַחְשְׁבוּ הַחֹשְקִים כִּי חָשְׁקוּ צוּרוֹת הֲדוּרוֹת / אָז הֵם רוֹכְבֵי אֲתוֹנוֹת צְחוֹרוֹת /

וְאָמַר בְּשִׁירוֹ:

תְּמוֹל שִׁלְשׁוֹם שְֹמָנִים וַחֲלָבִים

פְּגַשְׁתִּימוֹ בְנַחַל הַבְּשׂוֹרוֹת

שְׁאַלְתִּימוֹ יְדִידִים אַל פְּשַׁטְתֶּם

הֱשִׁיבוּנִי לְקַשֵּׁט הַכְּעוּרוֹת.

וְיֵש לְכָל חָכָם לֵב לְהִנָּצֵל מֵעָרְמָתָן / וּלְהוֹשִׁיב קַשְׁתּוֹ בְּאֵיתָן / פֶּן חֲמַס הַנָּשִׁים יְכַסֵּם וְשֹׁד בְּהֵמוֹת יְחִיתָן / כִּי חַמָּתָן מְרֻבָּה מִצִּלָּתָן / וּכְבָר אָמְרוּ הַחֲכָמִים / וְגָלוּי לְכָל הָעַמִּים / כִּי הָאִשָּׁה מְצוֹדִים וַחֲרָמִים / וְהִיא סִבַּת כָּל מְרִי נֶפֶשׁ וְאֵיבָה וּמְרִיבָה / וְלַמָּוֶת וּלְכָל תְּלָאָה עִלָּה וְסִבָּה / טוֹב לִפְנֵי הָאֱלֹהִים יִמָּלֵט מִמֶּנָּהּ וְחוֹטֵא יִלָּכֵד בָּהּ / וְאָמַר הֶחָכָם הָאִשָּׁה סִבַּת כָּל רָע לְמִגָדוֹל וְעַד קָטָן / וְהִיא מְצוּדָה פֵּרֵש אוֹתָהּ הַשָּׂטָן /

וְאָמַר הַמְּשׁוֹרֵר:

מוֹצֵא אָנִי מַר מִמָּוֶת

הַנָּשִׁים הוֹבִישׁ מַבָּטָן

נַחְנוּ צֵידָהּ הֵם כִּמְצוּדָה

פֵּרֵשׁ אוֹתָהּ יַד הַשָּׂטָן.

וְלָכֵן הַיָּרֵא אֶת דְּבַר אֲדֹנָי מֵעֲבָדָיו / וְיִפְחַד מֵאֳשּׁוֹ וְלַפִּידָיו / יִשָּׁמֵר מֵחִצֵּי עֵינֶיהָ / וְיֶעֱצַם עֵינָיו מֵרְאוֹת פָּנֶיהָ / וְאִם יַבְהִילוּהוּ אוּרֶיהָ וְתֻמֶּיהָ / יַהֲפֹך אוֹתָהּ וְיִתְגַּלּוּ אֵלָיו מוּמֶיהָ /

וְאָמַר הַמְּשׁוֹרֵר:

חֵמָה וָקֶצֶף אַף צְבִי הַחֵן שְׁפֹךְ

עַל יַעֲלָה וּגְעַר בְּמַנְעַמֶּיהָ

אִם לִבְּךָ יִטֶּה עֲלֵי חִנָּהּ הֲפֹךְ

אוֹתָה וְיִתְגַּלּוּ לְךָ מוּמֶיהָ.

וְכַמָּה מִן הַחֲכָמִים הַמְּחֻכָּמִים / וּמִזֶּרַע הַמְּלוּכָה וּמִן הַפַּרְתְּמִים / אֲשֶׁר הָיוּ עַל לוּחַ לֵב הַזְּמָן חוֹתָם / וְהַחֵשֶַׁק עָצַם רְאוֹתָם / עַד טַח מֵרְאוֹת עֵינֵיהֶם מֵהַשְׂכִּיל לִבּוֹתָם / וְלֹא הֵבִינוּ לְאַחֲרִיתָם / וַיָּמִירוּ בְקָלוֹן אֶת כְּבוֹדָם / וּמִשְׁנֶה קֶצֶף לָקְחוּ בְיָדָם / וְהִנִּיחוּ עַל קֶרֶן הַצְּבִי צְבִי אֲשֶׁר מֵאַיִן לְיֵש הָעֳלָם / וּבְאַהֲבָתוֹ וּבְחֶמְלָתוֹ הוּא גְאָלָם / וַיְנַטְּלֵם וַיְנַשְּׂאֵם כָּל יְמֵי עוֹלָם / רוֹכְבֵי אֲתוֹנוֹת צְחוֹרוֹת הָיוּ לָמוֹ לְבָרוֹת / וְנָסְעוּ בְסִבָּתָם מֵהַר שֶׁפֶר וּמֵחֲצֵרוֹת / וַיַּחֲנוּ בְקִבְרוֹת הַתַּאֲוָה וּבְמוֹסֵרוֹת / וְאָמַר אֶחָד בְּשִׁירוֹ / בִּרְאוֹתוֹ כִּי הַחֵשֶׁק בְּמַאֲסָרוֹ אֲסָרוֹ /

אֲנִי בֶּן חֲכָמִים וּמֵבִין עֲלוּמִים

וּבֵן גּוּר אֲרָיוֹת וְאִמִּי לְבִיָּה

וְהֵשַע וְעִצֵּם מְאוֹרֵי לְבָבִי

לְבִלְתִּי רְאוֹת הוֹד צְבִי יָהּ צְבִיָה.

וְאָמַר חָכָם אֶחָד שְׁאֵלוּהוּ אַנְשֵׁי עֲדָתוֹ / עַל אֹדוֹת חָכְמָתוֹ / וּמַעֲשֶׂה תָקְפּוֹ וּגְבוּרָתוֹ / דְּעוּ אֵפוֹא אַחַי / כִּי סִבַּת חָכְמָתִי וְעִלַּת שְׁבָחַי / וַאֲשֶׁר הַחֲכָמִים יֲשְׁקְדוּ עַל דַּלְתוֹתַי / לִשְׁמוֹר מְזּוזוּת פְּתָחַי / וַתְּהִי לָהֶם לְאַזְכָּרָה רֵיחַ נִיחוֹחָי / וְהָיָה כָּל הַנָּשׁוּךְ וְרָאָה אוֹתִי וָחָי / הוּא תַּחַת הֱיוֹת אִישׁ חֹשֵׁק לְנֶגְדּוֹ הַצְּבִיָה מַעֲמִיד / שִׁוִּיתִי אֲדֹנָי לְנֶגְדִּי תָמִיד /

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

לְשֹׁאֵל בְּמָה נַעֲלֵיתָ עֲדֵי כָל

נְשׁוּךְ הַזְּמָן יֶחֱזֶךָ וְהָיָה

עֲנִיתִיו תְּמוּרַת לְהַצִּיב לְנֶגְדִּי

רְאוּת מִן צְבִיָה הֲצִיבִי צְבִי יָהּ.

וְאֵין לִתְמֹהַּ אִם הָאִישׁ יֶחְסַר הוֹדוֹ וְיִפְעָתוֹ / וְיִהְיֶה רוּחַ אֱלֹהִים רָעָה מַבְעִיתוֹ / בְּהִתְחַבְּרוֹ עִם הָאִשָּׁה לְרָעָתוֹ / וּשְׁחִין אֲבַעְבֻּעוֹת עָלָיו פּוֹרֵחַ / וִיהִי מִכְּאֵב לִבּוֹ מַר צוֹרֵחַ / כִּי הִנֵּה הַשֶּׁמֶש הַזּוֹרֵחַ/ הַגָּדוֹל עַל כָּל צְבָא הַשָּׁמַיִם בְהֶכְרֵחַ / לֹא יִלְקֶה רַק בְּהִתְחַבְּרוֹ עִם הַיָּרֵחַ /

וְאָמַר הַמְּשׁוֹרֵר:

אַל תִּתְמַהּ עַל אִישׁ דּוּדָאֵי

חִשְׁקוֹ כַּיּוֹם נָתְנוּ רֵיחַ

אִם עֵת יִתְחַבֵּר עִם עָפְרָה

זִיו הוֹדוֹ מֶנּוּ בוֹרֵחַ

לֹא יִלְקֶה הַשֶּׁמֶש רַק עֵת

יִתְחַבֵּר עִם הַיָּרֵחַ.

וְלָכֵן חִדְּלוּ לָכֶם מִן הַחֵשֶׁק אֲשֶׁר מִפִּיכֶם יוֹצֵא וָשָׁב / כִּי בַמֶּה הוּא נֶחְשָׁב / וּתְנּוּ לָכֶם מַרְגּוֹעַ וָנֹפֶשׁ / וְלִבְשׁוּ לָכֶם מֲן הַשֵּׂכֶל בִּגְדֵי חֹפֶשׁ / פֶּן יִפְגְּעוּ בָכֶם אֲנָשִׁים מָרֵי נָפֶשׁ / וַאֲסֲפְתֶּם אֶת נַפְשְׁכֶם / וְנֶפֶשׁ בֵּית אֲבִיכֶם /

וַיֹאמֶר הַשָּׂר הֲזֹאת חָשַׁבְתָּ לְמִשְׁפָּט עָלֵינוּ / לְיָרְאֵנוּ וּלְהַבְהִילֵנוּ / בַּאֲמָרְךָ כִּי יֵשׁ לַגְּבִירָה בַּעַל חוֹתָם תָּכְנִית / חַי אָנִי הִנְנִי מִתְנַדֵּב לִגְזֹל מִיָּדוֹ הַחֲנִית / וְיֵּלֵך מִפָּנַי קְדוֹרַנִּית / וְגַם לַגְּבִירָה אַחִים שִׁבְעָה / אֶרְדֹּף אוֹתָם מַהֵר אֶל הַר וּמִגִּבְעָה אֶל גִּבְעָה / וָאֶקַּח לִי הַגְּבִירָה עַל אַפָּם וְעַל חֲמָתָם / וַאֲשִׂימֶנָּה עַל לִבִּי חוּתָם / וּבְּחֵפֶץ גָּדוֹל הָיִיתִי בַּעֲלָהּ מַכִּיר / כִּי הָיִיתִי מַכֶּה אוֹתוֹ בַחֲנִית וּבַקּיִר / אוֹ חִצַּי מִדָּמוֹ אַשְׁכִּיר /

וַיִּפֶן הַשָּׂר אֶל הַגְּבִירָה / וַיֹּאמַר הִנְנִי נִשְׁבָּע בְּאַהֲבָתֵךְ הַיְּקָרָה / וּבְיִפְעָתֵךְ הַטְּהוֹרָה / וּבְנֹגַהּ זַרְחֵךְ הַנָּטוּי עַל רָאשֵׁי הַחֹשְקִים כְּעֵין הַקֶּרַח הַנּוֹרָא / וּבְעֵינַיִךְ אֲשֶׁר כַּבְּרָקִים יָהֲלוֹכוּ / וּבְגַבּוֹתָם אֲשֶׁר קַשְׁתוֹתָם יִדְרוֹכוּ / וְלִבּוֹת הַחֹשְׁקִים יִמְשׁוֹכוּ /כִּי לֹא אֶפָּרֵד מִמֶּךְ עַד אֲשֶׁר אֶל חֲדַר הוֹרָתֵךְ תְּבִיאִינִי / אוֹ בַּעֲלֵךְ חוֹתָם תָּכְנִית תַּרְאִינִי /

וַתִּצְחַק הַגְּבִירָה וַתֹּאמַר חַיֶּיךָ הַשָּׂר אִם תִּרְאֶנּוּ תִּתְחָרֵט / כִּי הוּא אִישׁ צָרוּעַ וְרֹאשׁוֹ יִמָּרֵט / וְכָל עוֹמְסָיו שָׂרוֹט יִשָּׂרֵט /וְהוּא נִתְעָב וְנֶאֱלָח / כְּעוֹף נוֹדֵד קֵן מְשֻׁלָּח / חֵץ כְּבֵדוֹ יִפְלַחַ / וְאִם תַּחְפֹּץ לָדַעַת אֵיךְ נָסַעְתִּי מִיָּטְבָתָה אֶל אֶדְרֶעִי / הוּא הָאִישׁ הַמְּדַבֵּר עִמָּכֶם זֶה דּוֹדִי וְזֶה רֵעִי /

וַיִּקְצֹף הָאִישׁ עַל הַגְּבִירָה / וַחֲמַתוֹ בוֹ בָעֲרָה / וַיֹּאמַר אִי פָרָה סוֹרֵרָה / אֵשֶׁת כְּסִילּות וְאֵשֶׁת מֶרִי / אֲשֶׁר תֵּלְכִי עִמִּי קֶרִי / כְּדַבֵּר אַחַת הַנְּבָלוֹת תְּדַבֵּרִי / יַכֵּךְ אֲדֹנָי בִּשְׁחִין רָע וְלֹא יָמִישׁ מֵעֲשׂוֹת פֶּרִי / אִמְרִי נָא עוֹנְנָה מֶה חָזִית / כַּאֲשֶׁר קָדְשִׁי בָזִית / וְאָהַבְתְּ מִשְׁכָּבָם יָד חָזִית / חַי אָנִי כִּי לוּלֵא אֲשֶׁר עָלָה עַל פְּרִי הֲדָרֵךְ חֲזָזִית / וְאַחֲרֵי מַרְאֵה עֵינַיִךְ הִתְפַּתִּית / אָבִיךְ אֱמוֹרִי וְאִמֵּך חִתִּית / אָז רָאִית צוּרָתִי הָאֲמִיתִית /

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

בָּזִית צוּרָתִי שִׁית זֹונָה

מַרְאֵה עֵינַיִךְ הִתְפַּתִּית

לוּלֵי כִּעוּר צוּרָתֵךְ אָז

רָאִית צוּרָתִי הָאֲמִתִּית.

וַיֹּאמֶר הַשָּׂר וְהוּא מַבִּיט אֶל הַגְּבִירָה /אֲשֶׁר נַפְשׁוֹ בְנַפְשָׁהּ קְשׁוּרָה/ אִי צוּרָה נְעִימָה וַהֲדוּרָה / זֶה הוּא אֲשֶׁר קוֹרֵא עָצְמוֹ חוֹתַם תָּכְנִית / אֲשֶׁר פָּנָיו כִּפְנֵי הַתַּעֲנִית / וְעַל גַּבֵּךְ יַחֲרֹשׁ וְיַאֲרִיךְ מַעֲנִית / אִי הֵילֵל בֶּן שָׁחַר אֵיךְ נָפַלְתָּ מִשָׁמַיִם/ וַתִּהְיִי לְאִישׁ כְּמַר מִדְּלִי וּכְשַׁחַק מֹאזְנַיִם / חַי אָנִי לוּ הָיִית דַּלְתִּי דּוֹפֶקֶת / וְעַל דֹּדַי מִתְרַפֶּקֶת / אָז נוֹדַע לָךְ כִּי בְהֵמָה עַל צוּרַת אִישׁ אַתְּ חוֹבֶקֶת /

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֶת מִי תַחְבֹּקִי עָפְרַת חֵן

עַל מִי זֶה אַתְּ מִתְרַפֶּקֶת

לוּ הָיִית לִי נוֹדַע לָךְ כִּי

כָּעֵת “חַיָּה” אַתְּ חוֹבֶקֶת.

וַיְהִי כְּהַשְׁלִימוֹ הַשִּׁירָה / חֲמָתוֹ בוֹ בָּעֲרָה / וַיִּצְעַק צְעָקָה גְדוֹלָה וּמָרָה /וְנָהָה נְהִי וְנִהְיָּה / אוֹי וַאֲבוֹי וְאוֹיָה /אָמַר שָׁדוֹד נְשַׁדֻּנּוּ / חֶבְלֵי שְׁאוֹל סַבּוּנוּ / וְעוֹד יָדָם נְטוּיָּה / בְּבֵית יִשְׂרָאֵל רָאִיתִי שַׁעֲרוּרִיָּה / כִּי פָּרָה אֲדֻמָּה תְמִימָה בְשִׂכְלָהּ / אֲשֶׁר עַל עָלֶיהָ לֹא עָלָה / וְאֵין עַל עָפָר מָשְׁלָהּ / נָפְלָה כִּנְפֹל לִפְנֵי בְּנֵי עַוְלָה / תְּסַלֵּד בְּחִילָה / וְנַפְשָׁהּ מָרָה לָהּ / וְלֹא חָס עֵין אֱנוֹש עָלֶיהָ לְחֻמְלָה / וּבַעְלָהּ אֲרִי בַּמִּסְתָּרִים וְדֹב אוֹרֵב הוּא לָהּ /

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

אֵיךְ טָחוּ מֵהְשְׂכִּיל לִבּוֹת

אֵיךְ עֵינֵי רוֹאִים תִּשְׁעֶינָה

כִּי רָאִיתִי שַׁעְרוּרִיָּה

כִּי פָרָה וְדֹב תִּרְעֶינָה.

וַיְהִי טֶרֶם כִּלָּה הַשָּׂר שִׁירוֹ / וַאֲנִי בָּאתִי בְּדַבְּרוֹ / וְאֹמַר אַחֲלַי לְפָנֶיךָ הַשָּׂר אַל תִּמְנַע מִמֶּנִּי לֶחֶם חֻקִּי / וְאֶעֱנֶה אַף אָנִי חֶלְקִי /

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

אֶתְמַהּ עַל כָּל חֹשְׁקֵי חֵשֶׁק

אֵיךְ בַּחַיִים לֹא יָקוּצוּ

כִּי רָאִיתִי שַׁעְרוּרִיָּה

דָּבָר עִם הָפְכוֹ נִקְבָּצוּ

כִּי רָאִיתִי פָּרָה וָדֹב

תִּרְעֶינָה יַחְדָו יִרְבָּצוּ.

וַיְהִי כִּרְאוֹת הָאִישׁ כִּי הִרְחַבְנוּ עָלָיו פֶּה / וּלְמַחֲלָתוֹ אֵין אֲרֻכָה וּמַרְפֵּא / הָיָה כְּאִישׁ נִדְהָם וְעֻלְפֶּה / כְּאִלּוּ נִדְקַר בְּחֶרֶב וְנִסְפֶּה / לֹא יִפְתַח פִּיו כְּאִלֵּם / וּמִפְּנֵי קִצְפִּי וַחֲרוֹן אַפִּי מִתְעַלֵּם / וְהַגְּבִירָה בְּצָרַת בַּעֲלָהּ שְׂבֵעָה / וְאָז יִמָּלֵא שְׂחוֹק פִּיהָ וּשְׂפָתֶיהָ תְרוּעָה / וּבַעֲלָהּ מִצָּרָתוֹ צוֹעֵק חָמָס / רָשָׁע יִרְאֶה וְכָעַס שִׁנָּיו יַחֲרֹק וְנָמָס /

וַיְהִי אַחֲרֵי כֵן וַיִּשָּׂא הָאִישׁ אֶת עֵינָיו וַיָּשֵׂם אֶל הַשָּׂר פָּנָיו / וַיֹּאמַר נִפְלֵאתִי עָלֶיךָ הַשָּׂר /בַּעַל חָכְמָה וּמוּסָר /בְּיָדְעִי מַעֲלָתֶךָ / וְאֵת יְקָר תִּפְאֶרֶת גְּדוּלָתֶךָ / וְיָדַעְתִּי כִּי אֶת אֲדֹנָי אֱלֹהֶיךָ תִירָא וְאוֹתוֹ תַעֲבֹד / וּלְךָ אֵשֶׁת חֵן תִּתְמֹך כָּבוֹד / וְאֵיךְ בְּאֵשֶׁת אֲחֵרִים תַּחֲשֹׁק /הֲטוֹב לְךָ כִּי תַּעֲשֹק /וְהָיָה לְךָ לַחֲמֹל עַל יַחַס אֲבוֹתֶיךָ/ וְלִגְעוֹר בַּחֵשֶׁק אֲשֶׁר לֹא מִבְּרִיתֶךָ / לָמָּה תָּמוּת בְּלֹא עִתֶּךָ / שׁוּבָה הַשָּׂר מִזְּדוֹן לִבְּךָ וְעוּרָה מִשְּׁנָתֶךָ / וּפְקֹד בְּצֶדֶק הוֹן אוֹצְרוֹתֶיךָ / וְהָיָה לְךָ לְמֵשִׁיב נֶפֶשׁ וּלְכַלְכֵּל אֶת שֵׂיבָתֶךָ / פֶּן מַה תַּעֲשֶׂה בְאַחֲרִיתֶךָ /וְאָמַר הֶחָכָם לִבְנוֹ בְּנִי לְךָ יַחַס מְכֻבָּד / וְאִם תִּשְׁמְרֶנּוּ בְּטוּב הַמַּעֲשִׂים לֹא יֹאבַד / וּלְךָ הוֹן עָתָק בְּצֶדֶק תִּפְקְדֶנּוּ / וְדַע אִם לֹא תִפְקְדֶנּוּ תִּפְקְדֶנּוּ2 /
וַיִּשָּׂא מְשָלוֹ וַיֹאמַר:

בְּנִי יַחַסָךְ מְכֻבָּד תִּשְׁמְרֶנּוּ

בְּטוּב הַמַּעֲשִׂים פֶּן תֹּאבְדֶנּוּ

וְהוֹנְךָ רַב בְּצֶדֶק תִּפְקְדֶנּוּ

וְאִם לֹא תִּפְקְדֶנּוּ תִּפְקְדֶנּוּ.

וְיֵש לְךָ הַשָּׂר לְהָבִין לְאַחֲרִיתֶךָ / וּלְהִוָּסֵר בְּרָעַת זוּלָתֶךָ / וְלֹא יִוָּסְרוּ אֲחֵרִים בְּרָעָתֶךָ / וְאָמַר הֶחָכָם בְּסֵפֶר הַמּוּסָרִים / לֵץ תַּכֶּה וּפֶתִי יַעֲרִים / וְדַע כִּי הָאֱלֹהִים יִתְעָלֶּה גָּזְרָה חָכְמָתוֹ / אֲשֶׁר אִישׁ שֶׁלֹּא יִוָּסֵר בְּרָעַת זוּלָתוֹ / יִוָּסְרוּ אֲחֵרִים בְּרָעָתוֹ/

וְאָמַר הַמְּשׁוֹרֵר:

קְחָה מוּסָר וְהִוָּסֵר בְּזָּרִים

וְאַל יִוָּסְרוּ בְךָ הַבְּעָרִים

וְדַע כִּי אִישׁ בְּאַחֵר לֹא יְיֻסָּר

יְיַסֵּר בּוֹ אֱלֹהִים חַי אֲחֵרִים.

וְעַתָּה לֵךְ אִיעָצְךָ הַשָּׂר / הַעֲבֵר חֵשֶׁק מִקִּרְבֶּךָ וַעֲדֶה עֲדִי חָכְמָה וּמוּסָר / וּלְבַש תֻּמֶּיךָ וְאוּרֶיךָ / וְהַעֲבֵר רָעָה מִבְּשָׂרְךָ / וְהָלְכוּ גוֹיִים לְאוֹרֶךָ / וְאִם אֶת הַדָּבָר הַזֶּה תַּעֲשֶׂה וְחֻקּוֹתַי תִּלְמֹד / וְאֶת אֵשֶׁת עֲמִיתֶךָ לֹא תַּחְמֹד / וְצִוְּךָ אֱלֹהִים וְיָכָלְתָּ עֲמֹד / וְהַיָּמִים הָרִאשׁוֹנִים יִפְּלוּ כִּי מַה שֶּׁעָבַר חֲלוֹם / וְגַם כָּל הָעָם הָזֶּה עַל מְקוֹמוֹ יָבֹא בְשָׁלוֹם /

וַיַעַן הַשָּׂר וָיֹאמַר יָדַעְתִּי כִּי לִי אֵשֶׁת חִן וְכָבוֹד תִּתְמֹךְ / אָמְנָם בְּכָל יְכָלְתִּי עַל אֵשֶׁת אֲחֵרִים אֶסְמֹךְ / וְלֹא אָחוּשׁ בְּהַבִּיטִי אֶל מָקוֹם נָמֹך / וְזֹאת הַגְּבִירָה מִיּוֹם רְאִיתִיהָ אִוִּיתִיהָ חֲמַדְתִּיהָ / וְאַהֲבַת עוֹלָם אֲהַבְתִּיהָ / וְעַל כֵּן חֶסֶד מְשַׁכְתִּיהָ / וְעַל גַּרְגַּרוֹתַי עֲנָק עֲנַדְתִּיהָ / זֹאת מְנוּחָתִי עֲדֵי עָד פֹּה אֵשֵׁב כִּי אִוִּיתִיהָ / וַאֲשֶׁר תֹּאמַר אֵיךָ אֶחֱשֹׁק בְּאֵשֶׁת אַחֵר / דַּע כִּי אֲנִי מְבַקֵּש רָצוֹן וְטוֹב שׁוֹחֵר / אָמְנָם אֵשֶׁת בֶּן מַחְפִּיר וּמַבְאִישׁ / לֹא אֶחְשֹב לְאֵשֶׁת אִישׁ / וְאַתָּה אַחֲרֵי אֲשֶׁר עוֹנָתָהּ תִּגְרָע / וְהִיא לֹא תַצְמִיחַ וְלֹא תִזָּרַע / רָאוּי שֶׁיִּגַּע בְּךָ רָע / וַאֲשֶׁר יְעַצְתַּנִי בֶּן מְנַבֵּל / לְחַלֵּל בְּרִית הַחֵשֶׁק וּלְחַבֵּל / בָּזֶה עֲצָתְךָ לֹא אֲקַבֵּל / וְיָדַעְתִּי כִּי לִי יַחַס נִכְבָּד / וְלֹא אִירָא כִּי בְחֵשֶׁק הַגְּבִירָה יֹאבַד/ וְאִם תִּשְׁאָלֵנִי עָלָיו אֲנִי מִגֶּזַע יִשַׁי / וְאִם עֲצַתּךָ עַצַת אֲחִיתוֹפֶל אִתִּי עֲצַת חוּשַׁי /

וַיִּשָּא מְשָלוֹ וַיֹּאמַר:

דּוֹדִי שְׁאֵלַנִי עֲלֵי יַחַשִׂי

וָאֹמְרָה בֶּן עַבְדְּךָ יִשַׁי

אָמַר צְבִי קַח לְךָ עֲצַת לִבִּי

וֶאֶעֱנֶה טוֹבָה עֲצַת חוּשַׁי.

וְאֵיך לֹא תִהְיֶינָה יָדַי אְמוּנָה / לַחֲשֹׁק בַּיָּפָה כַּלְּבָנָה / וּמְקֻטֶּרֶת מוֹר וּלְבוֹנָה / חַי אָנִי כִּי מִיּוֹם הִכַּרְתִּי הָעֲדִינָה / וְרָאִיתִי דּוּדָאֵי חִשְׁקָהּ נָתְנוּ רֵיחַ / בֵּרַכְתִּי עַל אוֹרָהּ הַזּוֹרֵחַ / וְלֹא עַל חִדּוּשׁ הַיָּרֵחַ /

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

בְּיָדְעִי כִּי הַלְּבָנָה שֵׁם מְשֻׁתָּף

לְיָרֵחַ וְאֶל עָפְרָה לְבָנָה

אֲבָרֵך עַל צְבִיַת חֵן בְּחִדּוּשׁ

מְאוֹרָהּ וְלֹא בְחִדּוּשׁ הַלְּבָנָה.

וּמִן הַדִּין הָיָה לְךָ לְהַחֲזִיק אֵלַי טוֹבָה נֶאֱמָנָה / כַּאֲשֶׁר הָיִיתִי עִם הַגְּבִירָה בְאָמְנָה / כִּי אֶאֱהַב אֶת אֲשֶׁר תֶּאֱהַב וּמְשַׂנְאֶיךָ אֶשְׂנָא / וְהָיָה לְךָ לִשְׂנֹא אוֹתִי אִם אֶשְׂנָא אוֹהֲבֶיךָ / וְאֶגְעַל בִּקְרוֹבֶיךָ / וְהִנֵּה רְצֹונְךָ עִם רְצוֹנִי נִתְיַחָד / עוֹלוֹת בְּקַנֶה אֶחָד /

וַיֹאַמר הָאִישׁ אַל תַּחֲשֹׁב הַשָּׂר כִּי הַגְּבִירָה אֲהַבְתִיהָ / כִּי תַכְלִית שִׂנְאָה שְׂנֵאתִיהָ/ וְלוּלֵי חָמַלְתִּי עַל כְּבוֹדִי פֶּן יֵחָשֵׁב אֵלַי לִפְחִיתוּת / כָּתַבְתִּי לָהּ סֵפֶר כְּרִיתוּת / וְחַי אֲשֶׁר קָנָה לְבָבִי וְקָם לוֹ לְבַד לִצְמִיתוּת / בְּשִנְאָתִי אוֹתָהּ אֶתְאַו לִהְיוֹת מָעוּךְ וְכָתוּת /

וַיִּשָּׂא מְשָׁלוֹ וַיּאמַר:

בְּחֵי הָאֵל אֲשֶׁר קָנָה לְבָבִי

וְלִבִּי קָם לְבַד לוֹ לַצְּמִיתוּת

בְּשִׂנְאָתִי לְשׁוֹכֶבֶת בְּחֵיקִי

אֲנִי אֶתְאַו הֱיוֹת מָעוּך וְכָתוּת.

וַיֹאמֶר הַשָּׂר וְאִם יִגְזֹר הַמַּזָּל וְתָמוּת אִשְׁתֶּךָ / הֲתִלְכֹּד צְבִיָּה אַחֶרֶת בְּרִשְׁתֶּךָ /

וַיֹּאמֶר הָאִישׁ חַיֶיךָ הַשָּׂר אִם יֹאמְרוּ לִי בְנֵי עַמֵּנוּ / נְשָׂא אִשָּׁה וּמָלוֹך תִּמְלֹך עָלֶינוּ / לֹא חָפַצְתִּי לָקַחַת מַלְכָּה אוֹ נְסִיכָה / וְקֹדֶם חָפַצְתִּי לִהיוֹת כְּרוּת שָׁפְכָה /

וַיִּשָּׂא מְשָׁלוֹ וַיֹאמַר:

לוּ יֹאמְרוּ עַמִּי נְשָׂא אִשָּׁה

כַּיּוֹם וְעָלֵינוּ לְכָה מָלְכָה

אֶתְאַו הֱיוֹתִי נֶעֱבַד בִּלְתָּהּ

אוֹ לִהְיוֹת עִמָּהּ כְּרוּת שָׁפְכָה.

וַיֹאמַר הַשָּׂר הֲלֹא יָדַעְתָּ כִּי לֹא טוֹב הֱיוֹת הָאָדָם לְבַדּוֹ / וְלָכֵן נִבְרָאת הָאִשָּה לְצִדּוֹ / לִהְיוֹת לוֹ עֵזֶר כְּנֶגְדּוֹ / וְכַמָּה מִן הַנָּשִׁים/ יַעֲלִימוּ מוּמֵי הָאֲנָשִׁים / וְיַשְׂבִּיעוּ בְּצַחְצָחוֹת הַנְּפָשִׁים / וְתִהְיֶיןָ כָאֳנִיּוֹת סוֹחֵר וְיִשְׁמְרוּ הָרְכוּשִׁים / וְאָמַר הֶחָכָם כַּאֲשֶׁר שְׁאֵלוּהוּ / אֵיזֶה טוֹב לִבְנֵי הָאָדָם אֲשֶׁר יַעֲשׂוּהוּ / וַיֹאמֶר כִּי הָאָדָם יָחִיל טוּבוֹ / בְּשֵׂכֶל יִהְיֶה בוֹ / וְאִם אֵין זֶה יִתְהַדֵּר בְּמַסֶּכֶת זְהָבוֹ / וְאִם אֵין זֶה יִקַּח אִשָּׁה כְשֵׁרָה תַעֲלִים נֶגַע לְבָבוֹ / וְאִם אֵין זֶה יֳשְׁתֹּק וִיכֻסֶּה זְדוֹן לִבּוֹ בְקִרְבּוֹ / וְאִם אֵין זֶה הַקְּבוּרָה תִּהְיֶה סֻכָּה לְרֹאשׁוֹ וְרֻבּוֹ /

וְאָמַר בְּשִׁירוֹ:

שְׁאֵלוּנִי וּמַה זֶּה טוֹב בְּאָדָם

הֲשִׁיבוֹתִי בְּשֵׂכֶל יִהְיֶה בוֹ

וְאִם אֵין זֶה עֲנִיתִים הָעֲשִׁירוּת

לְהִתְהַדֵּר בְּמַּסֶּכֶת זְהָבוֹ

וְאִם אֵין זֶה עֲנִיתִים הַחֲבֵרָה

כְּשֵׁרָה תַעֲלִים מוּמֵי לְבָבוֹ

וְאִם אֵין זֶה עֲנִיתִים הַשְּׁתִיקָה

לְכַסּוֹת אֶת זְדוֹן לִבּוֹ בְקִרְבּוֹ

וְאִם אֵין זֶה עֲנִיתִים הַקְּבוּרָה

תְּהִי סֻכָּה עֲלֵי רֹאשׁוֹ וְרֻבּוֹ.

וַיֹאמֶר הָאִיש אָמְנָם יָדַעְתִּי כִּי הַנָּשִׁים לְעֵזֶר נִבְרָאוּ / אָמְנָם מְעַטּוֹת עַל זֹאת הַתְּכוּנָה נִמְצָאוּ / וְכַמָּה קָשֶׁה לִסְבֹּל חֶבְרַת הָרָעוֹת מֵהֶם / לֹא מֵהֶם וְלֹא מֵהֲמוֹנָם וְלֹא מֵהֱמֵהֶם / וְאָמַר הֶחָכָם בְּזֹאת יִבְחַן הָאִישׁ נַפְשׁוֹ הַחֲכָמָה / לָדַעַת כִּי הִגִּיעָה לְמַעֲלָה נִכְבֶּדֶת וְרָמָה / שֶׁיּוּכַל לִסְבֹּל חֶבְרַת אֵשֶּׁת הַזִּמָּה / כִּי הָאִשָּׁה הָרָעָה / תַּחֲלִיף שַׂעֲרָהּ וְלֹא טִבְעָהּ /

וָיִשָׂא מְשָׁלוֹ וַיֹּאמַר:

בָּזֶה יַבְחִין נַפְשׁוֹ אִיש שֶּ-

יִּסְבֹּל חֶבְרַת אִשָּׁה רָעָה

הִמְשִׁלוּהָ לִזְאֵב עֶרֶב

תַּחְלִיף שַׂעֲרָהּ אַךְ לֹא טִבְעָהּ.

וַיֹּאמֶר הַשָּׂר אֵין לִתְמֹהַּ אִם חָשַׁקְנוּ חֵשֶׁק עָצוּם הַיַּעֲלָה / כִּי שָׁמַעְנוּ אוֹמְרִים כִּי עוֹדֶנָה בְתוּלָה / וְרַבִּים חִשְׁקֵי בְּתוּלָה מֵחִשְׁקֵי בְעוּלָה / וּלְמַעֲלַת הַבְּתוּלָה נְתַנָּהּ הָאֵל בַּשָּׁמַיִם / בֵּין אֲרִי וּמֹאזְנָיִם /

וְאָמַר הַמְּשׁוֹרֵר:

בְּרַח דּוֹדִי בְּרַח מֵעִיר פְּרוּצָה

וְקַח נַעְרָה בְתוּלָה וַחֲרוּצָה

וְקַח מִפִּי לְךָ בַאוּר נַעֲרָה

וְנַעֲרָהּ מֵעֲפָרָהּ וַהֲקִיצָהּ.

וּמִן הַתֵּמַהּ מִן הַגְּבִירָה אֵיךְ בַּחַיִים עוֹדֶנָּה /כִּי נְשַׂאֲתָהּ וַתַּשְׁלִיכֶנָּה/ וְכִרְחוֹק מִזְרָח מִמַּעֲרָב תַּרְחִיקֶנָּה / וִישׁוּעָה רָחֲקָה מִמֶּנָּה / וְלָכֵן אִשָּׁה תְּאָרֵשׂ וְאִישׁ אַחֵר יִשְׁכָּבֶנָּהּ / וְלוּ לָמַדְתָּ וְהָיוּ תֹאֲמִים מִלְּמַטָּה / כָּל אַנְחָתָהּ הִשְׁבַּתָּה / הֲלֹא יָדַעְתָּ כִּי לֹא יִלְקֶה הַיָּרֵחַ אוֹ אוֹרוֹ לֹא יֶחֱצֶה / רַק בְּהִתְרַחֲקוֹ מִן הַשֶּׁמֶשׁ מִן הַקָּצֶה אֶל הַקָּצֶה /
וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

עָפְרָה עֵת עִם עֹפֶר תִּשְׁכֹּן

תָּאִיר תַּזְרִיחַ ותְרַצֶּה

אָכֵן עֵת נוֹד דּוֹד חִשְקָהּ עַל

נוּדוֹ גַּם לִבָּהּ יִמַָּצֶּה

תִּדְמֶה לַסַּהַר לֹא תֶּחְשָׁךְ

כֻּלָּהּ אוֹ אוֹרָהּ יֵחָצֶה

רַק עֵת תִּרְחַק מִן הַשֶּׁמֶשׁ

מִן הַקָּצֶה אֶל הַקָּצֶה.

וַיֹּאמֶר הָאִישׁ חָלִילָה לְךָ הַשָּׂר מֵחֲזוֹת מַהֲתַלּוֹת / וְלוֹמַר כִּי לְמַעֲלַת בְּתוּלָה הוּשְׂמָה בַּשָּׁמַיִם בֵּין הַמַּזָּלוֹת / וְהִנֵּה אִם תֹּאמַר כֵּן בַּבְּתוּלָה / מַה תֹּאמַר בְּסַרְטָן וְעַקְרָב אֲשֶׁר אֵצֶל גְּבוּלָהּ / הֲתֹאמַר כִּי בַעֲבוּר מַעֲלָתָם הוּשָׂמוּ / בֵּין רֹאש כּוֹכָבִים כִּי רָמוּ /

וַיִּשָׂא מְשָלוֹ וַיֹּאמַר:

בְּתוּלָה יַחְשְׁקוּ חֹשְׁקִים וְיֹאמְרוּ

לְמַעְלָתָהּ זְבוּל הָיָה זְבוּלָהּ

אֲשִׁיבֵמוֹ רְאוּ סַרְטָן וְעַקְרָב

בְּצִדָּהּ הֵם לְמַעְלָה לֹא מַעְלָהּ.

וַיֹּאמֶר הַשָּׂר חַי אֲנִי כִּי דְבָרֶיךָ לֹא יוֹעִילוּךָ / וּמְשָׁלֶיךָ וַהֲתֻּלֶּיךָ לֹא יַצִּילוּךָ / הַגְּבִירָה הַזֹּאת לֹא לְךָ תִהְיֶה / וַאֲשֶׁר יָקוּם אֵלֵינוּ אִם בְּהֵמָה אִם אִישׁ לֹא יִחְיֶה / מוֹת יוּמָת דָּמָיו בּוֹ יִהְיֶה /

וַיְהִי כִּרְאוֹת הַגְּבִירָה / כִּי בַּעְלָהּ יְדַבֵּר עָלֶיהָ סָרָה / כְּאִלּוּ הָיְתָּה לִזְנוּנִים הָרָה / וַתֹּאמֶר בְּלִבָּהּ אַחֲרֵי אֲשֶׁר בִּי יִמְעַל / וּבְחִשְׁקִי יִגְעָל / בּוֹר כָּרָה וַיַּחְפְּרֵהוּ וַיִּפֹּל בְּשַׁחַת יִפְעָל / וַתָּשֶׂם אֶל לִבָּהּ לְהַגְדִּיל עָצְבּוֹ / עַד יִהְיֶה גּוֹנֵחַ מִלִּבּוֹ / וַתֹּאמֶר אֵלֵינּוּ שְׁמָעוּנִי הַחֹשְׁקִים הַנְּעִימִים /כִּי אַתֶּם נֵזֶר הַיָּמִים /בַּעְלִי זָקֵן בָּא בַּיָּמִּים / בַּחֹשֶׁךְ בָּא וּבַחֹשֶׁךְ יִתְעַלָּם / וִיצוּרָיו כַּצֵּל כֻּלָּם / חָלְפוּ עִם אֳנִיּוֹת אֵבֶה / וְלַחֵשֶׁק לֹא יִשְׁמַע וְלֹא יֹאבֶה / וְלִי סְחוֹרוֹת יָפוֹת וְנִמְרָצוֹת /צְבִי הֵם לְכָל הָאֲרָצוֹת /גֻּלּוֹת עִלִּיּוֹת וְגֻלּוֹת תַּחְתִּיּוֹת/ עָבְרוּ מַשְׂכִּיּוֹת / וּבַעְלִי מָךְ מֵעֶרְכָּם / וְהוּא בֵּן לֹא חָכָם / וְאַתָּם שְׁנֵי חֹשְׁקִים / עַל דַּלְתּוֹתַי מִתְדַּפְּקִים / אֶחֱלֹּק גּוּפִי לָכֶם לִשְׁנֵי חֲלָקִים / הִנֵּה גוּפִי אֲשֶׁר דָּמְתָה לְתָּמָר קוּמָתוֹ / וְהוּא כְּחָתָן יוֹצֵא מֵחֻפָּתוֹ / הָאֵזוֹר יִגְבֹּל אוֹתוֹ / קַח אַתָּה הַשָּׂר בָּרִאשׁוֹנָה / וְיַד רֵעֲךָ תִּהְיֶה בָאֲחֲרוֹנָה /

וָיִשָּׁא הַשָּׂר עֵינָיו וַיַּרְא גּוּפָהּ נֶחְמָד לְמַבָּטָיו / וְכָל אֲשֶׁר בָּהּ שׁוֹכֵן לִשְׁבָטָיו / כְּלָלָיו וּפְרָטָיו / וְכִרְאוֹת הַשָּׂר נֶטַע נַעֲמָן / דַּלַת רֹאשָׁהּ כָּאַרְגָּמָן / וּשְׂפָתֶיהָ שְׂפַת כּוֹס פֶּרַח שׁוֹשָׁן מַעֲשֶׂה יְדֵי אָמָּן / וִיפִי הָעֵינַיִם / וּבְרַד הַשִּׁנַּיִם וְכַפְתּוֹרֶיהָ וּפְרָחֶיהָ / וַהִדּוּרֶיהָ וּשְׁבָחֶיהָ / וְצַוָּארָהּ כְמִגְדַּל הַשֵּׁן / וַיֹּאמַר אָרוּר הָאִישׁ אֲשֶׁר עִם זֹאת הַגְּבִירָה יָשֵׁן / וּבִרְאוֹתוֹ חֲזֶה הַתְּנוּפָה אֲשֶׁר הוּנָף וַאֲשֶׁר הוּרָם / וַיֹּאמַר זֹאת הָאָרֶץ אֲשֶׁר נִשְׁבַּע אֲדֹנָי לְאַבְרָם / וַיִּבְחַר לוֹ הַמָּנָה הָעֶלְיוֹנָה / וַיֹּאמֶר כִּי לוּלֵי הֱיוֹתָהּ הַמָּנָה הַנִּכְבֶּדֶת וְהָעֲדִינָה / לֹא הָיְתָה לְמֹשֶׁה לְמָנָה /

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

מָה אֹמְרָה עֵת אֶחֱזֶה מִמַּעַל

יִפְעַת תְּמוּנָתֵךְ אֲשֶׁר זוֹרַָחַת

אַבִּיט אֱלֵי אֶרֶץ וְשָׁמָּה תַאֲוַת

נַפְשִׁי כָּאֵש תִּבְעַר וּמִתְלַקָּחַת

אָמְנָם לְגֹדֶל מַעֲלַתֵךְ יַעְלָה

תִּרְגַז יְחִידָתִי שְׁאוֹל מִתָּחַת.

וַיְהִי הוּא כִלָּה לְדַבֵּר וָיַרְא בַּעַל הַיָּפָה / כִּי הַגְּבִירָה תִתְנַדֵּב לָנוּ בְּגוּפָהּ / וַיֹּאמַר אָבְדָה הָאֱמוּנָה / וְאֵין לָהּ שֹׁרֶשׁ וְעָנָף / כִּי הָאָרֶץ חָנֹף תֶּחֱנָף / וַיִּקְצֹף בְּלִבּוֹ וַיִּתְאַנָּף / וַיַּחֲשֹׁב בְּלִבּוֹ עָרְמוֹת/ לְהִנָּצֵל מִיַּד מְהוּמוֹת / וַיֹּאמֶר אֵלֵינוּ וְרַעְיוֹנָיו פּוֹחֲדִים וְרוֹהִים / שְׁמָעוּנִי וְיִשְׁמַע אֲלֵיכֶם אֱלֹהִים / הַיּוֹם הַזֶּה חֲשַׁבְתוּנִי לְחֹשֶׁק כְּמוֹכֶם / וַיֵּצֵא דְּבַר מַלְכוּת מִלִּפְנֵיכֶם / גְּזַרְתֶּם אֹמֶר וַיָקָם לָכֶם / אֲמַרְתֶּם יְחַבֵּר כָּל אֶחָד מִמֶּנּוּ שִׁירָה / עַל אֹדוֹת הַגְּבִירָה/ וְהָיָה הָאִישׁ אֲשֶׁר בִּמְלֶאכֶת הַשִּׁיר תִּגְבַּר יָדוֹ / נַנִּיחַ הַצְּבִיָּה לוֹ לְבַדּוֹ / וְעַתָּה אַחֲרֵי אֲשֶׁר הַמַּזָּל פֹּה הֱבִיאָנִי / הֱשִׁמָּנִי וְהֶלְאָנִי / וְלִרְאוֹת אֲשֶׁר לֹא דִמִּיתִי לִרְאוֹת הֶרְאָנִי / הִנְנִי לְמַלֹּאת מִצְוַתְכֶם אֶרְצֶה/ וְתִהְיוּ שְׁנֵיכֶם וְכָל הַחֹשְׁקִים מִקָּצֶה / וּנְחַבֵּר שִׁירִים מֵאֹדוֹת זֹאת הַקְּדֵשָׁה / וְאִם לֹא אֶגְבַּר עֲלֵיכֶם בְּשִׁירָה חֲדָשָׁה / בְּשָׂפָה בְרוּרָה בִּנְעִימָה וּבִקְדֻשָּׁה /תִּהְיֶה עַל לוּחַ לֵב הַזְּמָן חֲרוּשָׁה/ וְנָתַתִּי אוֹתָהּ לָכֶם לְמוֹרָשָה / אַךְ יָדֹעַ תֵּדְעוּ כִּי אֲנִי כְאָדָם עָשׁוּק וְרָצוּץ / אֲגַנֶּה הַבְּרוֹשׁ וַאֲפָאֵר הַנַּעֲצוּץ / כִּי חֵשֶׁק לֹא יָדַעְתִּי וְלֹא עָבַר מֵעוֹלָם בִּי / אַךְ הֲנִיפוֹתִי לְעוֹלָם עָלָיו חַרְבִּי/ וְהִכְרַתִּי אוֹתוֹ מִקִּרְבִּי / אֲשֶׁר לֹא צִוִּיתִי וְלֹא דִּבַּרְתִּי וְלֹא עָלְתָה עַל לִבִּי / אָמְנָם כְּאָדָם אָנוּס / מִן הַפַּחַד אֶת הַפַּחַת אָנוּס / וְעַתָּה הַגִּישׁוּ עַצֻמוֹתֵיכֶם אַנְשֵׁי הַזִּמָּה / וְקוּמוּ וְנָקוּמָה וְנַעֲלֶה עָלֶיהָ לַמִּלְחָמָה / וְיַעֲבֹר עָלַי מָה /

וַיְהִי כִּשְׁמֹעַ הַשָּׂר נִיבוֹ / עָלַז לִבּוֹ בְקִרְבּוֹ / בְּחָשְׁבוֹ כִּי יָשִׁיב צוּר חַרְבּוֹ/ וַיֹּאמַר זֶה הַיּוֹם עָשָׂה אֲדֹנָי נָגִילָה וְנִשְׂמְחָה בוֹ / וַיִּפֶן אֶל הַגְּבִירָה / וַיֹּאמַר רָנִּי וְשִׂמְחִי צוּרָה הַהֲדוּרָה / כִּי הַיּוֹם הַזֶּה יוֹם בְּשׂוֹרָה / כִּי יָדֵנוּ בִּמְלֶאכֶת הַשִּׁיר תִּגְבַּר / וְשִׁירָיו תָּעוּ בָּמִּדְבָּר / אֵין חֵפֶץ בָּם לֹא לִזְרוֹת וְלֹא לְהָבֵר / וְעַתָּה מַה לַתֶּבֶן אֶת הַבָּר / גִּילִי וְשִׂמְחִי כִּי נְשְׁבַּר שֵׁבֶט מַכֵּך / וְתֶחֱסִי תַּחַת כְּרוּב מִמְשַׁח הַסּוֹכֵךְ / יִהְיֶה פִרְיוֹ מָתוֹק לְחִכֵּךְ / כִּי הִנְנִי בָא וְשָכַנְתִּי בְּתוֹכֵךְ /

וַתֹּאמֶר הַגְּבִירָה חַיֶּיךָ הַשָּׂר אַף עַל פִּי שֶׁצּוּרָתוֹ כְעוּרָה / וּפָנָיו כְּשׁוּלֵי קְדֵרָה / יִשְׁלַח אֲדֹנָי בוֹ הַמְּאֵרָה / לוֹ שָׂפָה בְרוּרָה / וְשִׁירָתוֹ חֲזָקָה וְנִמְרֶצֶת / מִפְּנִינֵי הַמַּאֲמָרִים מְקֻבֶּצֶת / וְאִם לְאַשְמַת הָעָם יִגְבַּר עֲלֵיכֶם בְּשִׁירָתוֹ בֶן הַזּוֹנָה / תִּהְיֶה יָדִי עַל הַתַּחְתּוֹנָה / וּמַה תַּעֲשֶׂה נַפְשִׁי הָעֲנוּגָה / אִם תִּהְיֶה עַל זָקֵן מַמְרֶה לְמָנָה /

וַיֹּאמֶר הַשָּׂר אַל תִּרָאִי חֲבַצֶלֶת הַשָׂרוֹן / כִּי אֲחַבֵּר עָלַיִּךְ שִׁירִים יְהֻלְּלוּ בְּכָל פֶּה וְגָרוֹן / וְתַעַנְדִי אוֹתָם עֲנָקִים לְגַרְגְּרוֹתַיִךְ וְשַׂמְתָּם בָּאָרוֹן /

וַתֹּאמֶר הַגְּבִירָה אֱלֹהִים יַעֲנֶה אֶת שְׁלוֹמְכֶם / וְיִרְצֶה אֲדֹנָי לְתִתִּי לַהֲלֹך עִמָּכֶם /

וַיֹּאמֶר הָאִישׁ אֵיךְ יִהְיֶה עִנְיַן שִׁירֵינוּ בְּדִבְרֵי יְקָר אוֹ בְּזֻלּוּת/ בְּדִבְרֵי חָכְמָה אוֹ בְדִבְרֵי הוֹלֵלוּת /

וַיֹּאמֶר הַשָּׂר נַפְלִיג לְדַבֵּר בְּנַבְלוּת / עַד אֲשֶׁר יֵחָשֵׁב לָנוּ לְשִׁפְלוּת/ וְיִהְיוּ שִׁירֵינוּ הוֹלֵלוּת וְסִכְלוּת /

וַיֹּאמֶר הָאִישׁ וּמִי יִהְיֶה לְחַטָּאת רִאשׁוֹנָה /

וַיֹּאמֶר הַשָּׂר אֲנִי אֶגְאַל הַיּוֹם מִבֵּין הַחוֹחִים הַשּׁוֹשַׁנָּה /

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

עָפְרָה הָסֵבִּי עֵינַיִך

מֵעֵינֵי סֹעַר לֹא נֻחָם

לִבִּי תַּצִּיבִי מַטָּרָה

מֵחִצַּיִּך יוֹם יוֹם נִלְחָם

עָלַי נַפְשִׁי תִשְׁתּוֹחַח הִיא

מִתְלַקַּחַת תּוֹך אֵשׁ פֶּחָם

עַד אָן תַּסְתִּירִי פָנַיִך

תּוֹרִידִי לָאָרֶץ נִצְחָם

אַחְלַי עָפְרָה הָרִיקִי לִי

בִּרְכוֹתַיךָ אָז אֶתְנָחָם

בִּרְכוֹת אֵם רוֹבֶצֶת תָּחַת

בִּרְכוֹת שָׁדַיִם וָרָחַם.

וַיְהִי כְּכַלּוֹת הַשָּׂר שִׁירוֹ בָּאתִי אַחֲרָיו / וּמִלֵּאתִי אֶת דְּבָרָיו /

וָאֶשָׁא מְשָׁלִי וָאֹמַר:

עָפְרַת חֵן גֶּפֶן פּוֹרַחַת

מִפָּנַיִךְ לִבִּי נִחַת

הַרְאִינִי נָא אֵת מַרְאַיִך

כִּי עַל חִשְׁקֵךְ מַרְאִי מָשְׁחָת

נִדְמֶה חִשְׁקֵךְ אֶל תּוֹךְ קִרְבִּי

בָּרָד עִם אֵשׁ מִתְלַקָּחַת

הַשְׁמִעִינִי רַעְיָה קוֹלֵך

וּפְדִינִי מֵרֶדֶת שָׁחַת

הֲבִיאִינִי אֶל גַּן עֶדְנֵךְ

הַגְמִיאִינִי יֵין מֶרְקַחַת

מִמֶּגֶד שָׁדַיִם מֵעָל

וּמִתְּהוֹם רוֹבֶצֶת תָּחַת.

וַיְהִי טֶרֶם אֲכָלֶּה לְדַבֵּר וְשִׁירִי עוֹדֶנּוּ בְחִכִּי / וַיֹאמֶר הָאִישׁ אַל תְּאַחֲרוּ אוֹתִי וַאדֹנָי הִצְלִיחַ דַּרְכִּי /

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

מַה יָּפוּ דוֹדַיִך עָפְרָה

אוֹר לֶחְיֶךְ עַל כָּל אוֹר מָלָךְ

נָפַלְתִּי בִּמְצוֹד אַהְבָתֵךְ

כִּנְפֹל דָּגִים לִפְנֵי שֶׁלָךְ

מִיִּרְאַת סַעְרָתֵך אָנוּס

מֵחַמָּתֵךְ אֶל הוֹד צִלָּך

אֶרְחַץ סוּתִי עִם דַּם דִּמְעִי

אֶשְׁטָחֶנוּ אֶל אוֹר הִלָּךְ

בָּחַרְתִּי הִסְתּוֹפֵף כַּיּוֹם

לִהְיוֹת עִמָּךְ לִשְׁכַּב אֶצְלָךְ

רַעְיָּתִי אַל תַּעְצֵרִי לִי

לִרְכֹּב כִּי אִם אָמַרְתִּי לָךְ.

וַיְהִי כְּהַשְׁלִימוֹ הַשִּׁירָה הַמְּהֻלָּלָה /נָתַן עָלֵינוּ בְקוֹלוֹ קוֹל הֲמֻלָּה/ וַיֶּחֱרַד הַשָּׂר חֲרָדָה גְדוֹלָה / וַיֹּאמַר זֶה קֹול עֲנוֹת גְּבוּרָה / בַּפַּעַם הַזֹּאת לֹא נִצַּחְנוּ הַגְּבִירָה /

וַאַעַן וָאֹמַר אֶל הַשָּׂר אֵין לְנַסּוֹת עוֹד בְּשִׁירִים / כִּי לוֹ פְנִינֵי הַמַּאֲמָרִים / וְיַהֲלוֹמֵי הַדִּבּוּרִים / הָבָה נִתְחַכְּמָה לוֹ בְסוֹדוֹת / וְנָחוּד לוֹ חִידוֹת / כִּי אִם גָּבְּרָה יָדוֹ בְּשִׁיר קָצֵר שִׂכְלוֹ / אוּלַי יְפֻתֶּה וְנוּכְלָה לוֹ /

וַיֹּאמֶר הַשָּׂר מַה מְּאֹד הֵיטִיב לְדַבֵּר זֶה הַנָּבָל / אֲשֶׁר לִקְבָרוֹת יוּבָל / וּבְשָׂרוֹ עָלָיו יִכְאַב וְנַפְשׁוֹ עָלָיו תֶּאֱבָל / רוֹאֶה אֲנִי כִּי כָשַׁל כֹּחַ הַסַּבָּל / וְעַתָּה חוּדָה לִי חִידָתְךָ וְתַרְחִיב לִי בַצָּר / כִּי יָדִי מֵהֹושִׁיעַ תִקְצָר / וּמְזִמָּה מִמֶּנִּי תִבְצָר / וּלְבָבִי כֹּחַ לֹא עָצָר/

אָז לָבְשָׁה אוֹתִי רוּחַ קִנְאָה / וָאֹמַר אֶל בַּעַל הַנָּאָה / אָחוּדָה נָא לְךָ חִידָה / אִם יָכוֹל תוּכַל לְהַגִּידָהּ / נַנִּיחַ לְךָ הַגְּבִירָה בִּכְבוֹדָהּ / וְלֹא נְדַבֵּר עוֹד מֵאֹדוֹתֶיהָ / וְלֹא נִשְׂחַק עוֹד עַל מִשְׁבַּתֶּיהָ /

וַיֹּאמֶר הָאִישׁ הִשָּׁבְעוּ לִי כַיּוֹם / וַנִּשָּבַע לוֹ וַיִּשְׂמַח הָאִישׁ כִּי רַב חֵילוֹ / וַיֹּאמַר שָׁמַע אֱלֹהִים בְּקוֹלִי / אַךְ עָשַׁרְתִּי מָצָאתִי אוֹן לִי/ הִנֵּה אִשְׁתִּי אֲשֶׁר תַּחְמְדֶנָּה / חוּדּה חִידַתְךָ וְנִשְׁמָעֶנָּה / וְאִם אָעֱמֹד רֶגַע וְלֹא אֶדַעֶנָּה / קְחוּ לָכֶם אִשְׁתִּי / אֲשֶׁר לָקַחְתִּי מִיֶּדְכֶם בְּחַרְבִּי וּבְקַשְׁתִּי / וְהִתְעָלְּלוּ בָהּ בְּפָנַי לְבָשְׁתִּי /

וַיִּשָּא מְשָלוֹ וַיֹּאמַר:

אֵין לִי יְרִיבוּן עִלְגֵי שָׂפָה

נִרְפִּים וְלֵצִים חָמְדוּ לָצוֹן

כִּי נֶחְשְׁבוּ לִי כָּל נְאוֹת הַשִׁיר

לִנְווֹת כְּרוֹת רוֹעִים וגִדְרוֹת צֹאן.

וָאַעַן וָאֹמַר אַל יִתְהַלֵּל חוֹגֵר כִּמְפַתֵּחַ / וְאוּלַי כִּי לְמִסְגֶּרֶת חִידָתִי לֹא תִמְצָא מַפְתֵּחַ / וְאָמַר הֶחָכָם אֵין לָאָדָם לְהִתְאַבֵּל עַל פִּיד יוֹם / וְלֹא לִשְׂמֹחַ בְּעֵת יִמְצָא פִּדְיוֹם / כִּי הַכֹּל בִּגְזֵרַת הַנּוֹרָא וְהָאָיוֹם / וְלֹא תֵדַּע מַה יֵּלֶד יוֹם /

וַאֶשָּא מְשָׁלִי וָאֹמַר:

אַל תִּתְעַצַּב דּוֹד אֶל פִּיד יוֹם

אַל תִּשְׂמַח יוֹם תִּמְצָא פִּדְיוֹם

עִם אֵל עוֹלָם הַסְכֵּן וּשְׁלָם

הוּא הַנּוֹרָא הוּא הָאָיוֹם

וּבְיוֹם מָחָר אַל תִּתְהַלָּל

כִּי לֹא תֵּדַע מַה יֵּלֵד יוֹם.

וַיֹּאמֶר הָאִישׁ עַד מָתַי חִידַתְךָ בְּתוֹך חִכְּךָ תִמְנַעֶנָּה / חוּדָה חִידָתְךָ וְאֶשְמָעֶנָּה /

וָאַעַן וָאֹמַר הִנֵּה הַחִידָה / שִׂים פָּנֶיךָ לְהַגִּידָהּ / אֱמֹר נָא מִי הָאִשָּׁה אֲשֶׁר פָּרְצָה עָלֶיהָ פֶּרֶץ / וְנֹולְדָה בִּבְרִיאַת הַשָּׁמַיִם וְהָאָרֶץ/ וּבַיּוֹם אֲשֶׁר נוֹלְדָה בְּבֵית נוֹאֲפִים שָׁקְדָה / וּבְכָל אֲשֶׁר עָגְבָה עֲלֵיהֶם בָּגְדָה / וּבְסוֹד עִירִין וְקַדִּישִׁין לֹא עָמְדָה / וְהִיא בְעוּלַת בָּעַל / וַתַּחֲשֹׁק בְּזָב וּבְבְלִיָּעַל / וּבְחֹשְׁקֶיהָ מַעֲלָה מָעַל / אֵינֶנָּה פוֹעֶלֶת דָּבָר אַךְ מִתְפָּעֶלֶת / תַחֲשֹׁק בְּזָרִים וּבַעְלָהּ גוֹעֶלֶת/ לֹא יָדְעָה אַלְמוֹן יַחֲרִידֶנָּה / רַבּוּ שְׁנוֹתֶיהָ וּזְקֵנָה אֵינֶנָּה / הִנֵּה זֹאת חֲקַרְנוּהָ כֵן הִיא שְׁמָעֶנָּה / כִּי לְמַעַן הַרְאוֹתְךָ הוּבָאת הֵנָּה /

וַיֹּאמֶר הָאִישׁ הֲזֹאת חָשַׁבְתָּ לְמִשְׁפָּט בַּעַל הַלָּשׁוֹן / הַחִידָה אֲשֶׁר חַדְתָּ הוּא הַחֹמֶר הָרִאשׁוֹן /

וַיִּשָּא מְשָלוֹ וַיֹּאמֶר:

אָמְרוּ וּמָה אִשָּׁה בַּעֻלַּת בַּעַל

יוֹם נוֹלְדוּ הָרִים הֲכִי יֻלָּדָה

יוֹם נוֹלדָה בֵּית נוֹאֲפִים שָׁקָדָה

וּבְחֲֹשְׁקֶיהָ מַעֲלָה הִיא מָעַל

הִיא חָשְׁקָה בֶּזָב וּבִבְלִיַּעַל

בְכָל אֲשֶׁר הִיא עָגְבָה בָגָדָה

לֹא יָדְעָה אַלְמוֹן וְלֹא עָמָדָה

לִרְאוֹת בְּסוֹד עִירִין וְאֵלֵי מָעַל

לֹא פָעֲלָה דָבָר וּמִתְפָּעֶלֶת

רַבּוּ שְׁנוֹתֶיהָ וְלֹא זָקָנָה

תַּחְשֹׁק בְּזָרִים בַּעֲלָהּ גֹעֶלֶת

אֶעֱנֶה חִידַתְכֶם צָפָנָה

סוֹד כִּי הָיוּ לִי נוֹאֲפָה אִוֶּלֶת

תִּקְרָא וְהוּא הַסּוֹד אֲשֶׁר טָמָנָה.

וַיְהִי כְּכַלּוֹת הָאִישׁ לְהַגִּיד הַחִידָה / וַיֶחֱרַד הַשָּׂר חֲרָדָה / וַיָּרַע הָאִישׁ עָלֵינוּ תְרֻעָה גְדוֹלָה / נִקְהֲלָה עָלֵינוּ כָּל הַקְּהִלָּה / וַתְהִי צָרַָתֵנוּ כְפוּלָה / כִּי אָמְרוּ הַשְּׁכֵנִים / לִרְגֹּם אוֹתָנוּ בַאֲבָנִים / וְהַגְּבִירָה אֲשֶׁר הָיְתָה בַחֲלוֹן נִשְׁקָפָה / כְּסוּפוֹת בַּנֶּגֶב חָלָפָה / וְעֵינֶיהָ כַמָּטָר יַעֲרֹפוּ / יָזּוּבוּ מַיִם וּנְחָלִים יִשְׁטֹפוּ / וְשָׂם גַּפְנָהּ לְשַׁמָּה / וּתְאֵנָתָהּ לִקָצָפָה / וְעָמְדָה עָלֵינוּ קְצוּפָה / וְלֹא רְאִינוּהָ שָׁנָה רְצוּפָה / וַאֲנָחְנוּ קָלוֹן מִכָּבוֹד שָׂבַעְנוּ / וְדֶרֶך שָׁלוֹם לֹא יָדַעְנוּ / וְשַׁבְנוּ אֶל בֵּיתֵנוּ בְבֹשֶׁת וּבִכְלִמָּה / כְּעַם הַנִּכְלָמִים בְּנוּסָם מִן הַמִּלְחָמָה / וַיְהִי כִּי בָאנוּ אֶל בֵּית הַשָּׂר כָּל רוֹאֵנוּ יֶחֱרַד וְיִשּׁוֹם / כִּי פְּנֵי הַשָּׂר אֵינָם כִּתְמוֹל שִׁלְשׁוֹם / וּבְכָל אֲשֶׁר יִרְאֶה יִגְעַר וְיִלָּחֵם / וַיָּקוּמוּ כָל בָּנָיו וְכָל בְּנוֹתָיו לְנַחֲמוֹ וַיְמָאֵן לְהִתְנַחֵם / וַתֵּתַצַּב אֲחוֹתוֹ מֵרָחוֹק / וּפָעֲרָה פִיהָ לִבְלִי חֹק / אוּלַי מִצָּרָתוֹ תִגְרַע וְתִמְחֹק / וַיֹּאמֶר הַשָּׂר אֳין עֵת לִשְׂחֹק / כִּי הַיָּמִים עָבְרוּ תוֹרָה חָלְפוּ חֹק / וַיּוּשַׂם לְפָנָיו לֶאֱכֹל וַיֹּאמַר לֹא אֹכַל / כִּי צָרוֹת לְבָבִי הִרְחִיבוּ וְשַׁבְתִּי לְחִצֵּי הַדְּאָגוֹת לְמַאֲכָל / וְנִלְאֵיתִי כַלְכֵּל וְלֹא אוּכָל /

וְכִרְאוֹתִי כִּי כָל מַעֲשֵינוּ אֶל הַשָּׂר לֹא לְרָצוֹן / הָלַכְתִּי בְּחֲמַת רוּחִי וְלָקַחְתִּי גְדִי עִזִּים מִן הַצֹּאן / רַךְ וָטוֹב זָכָר תָּמִים / וַתַּעַשׂ אִמּוֹ מַטְעַמִּים / וְכִרְאוֹתִי כִּי חֲמָתוֹ כִמְעַט סָפָתָה / לָקַחְתִּי בְיָדִי הַמַּטְעַמִּים וְהַלֶּחֶם אֲשֶׁר עָשָׂתָה / וָאֹמַר יָקוּם אָבִי שַׂר הַתְּעוּדָה / וְיֹאכַל כִּי חָלַף עֵת הַסְּעוּדָה / וַיִּשָּׁבַע וַיֹּאמַר לְתוֹרָה וְלִתְעוּדָה / אִם תָּקוּם תָּמָר לֹא אֶבְרֶה מִיָּדָהּ /

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

לִבִּי יִרְעַד רַגְלִי יִמְעַד

אֵיכָה אֶסְעַד דּוֹד שֶׂה בִּרְיָּה

חַי אָנִי לוּ תָקוּם תָּמָר

לֹא אֶבְרֶה מִיָּדָהּ בִּרְיָה.

וַיִּשְׁאָלֵנִי הַשָּׂר הֲרָאִיתָ מֶה עָשְׂתָה הַגְּבִירָה הַנֶּחְמָדָה / כַּאֲשֶׁר הִגִּיד בֶּן נַעֲוַת הַמַּרְדּוּת הַחִידָה / וָאַעַן וָאֹמַר / רָאִיתִי וְהָיוּ עֵינֶיהָ יוֹרְדוֹת מָיִם / כְּאִלּוּ נִבְקְעוּ מַעְיְנוֹת תְּהוֹם רַבָּה וְנִפְתְּחוּ אֲרֻבּוֹת הַשָּמָיִם /

וַאֶשָּׂא מְשָלִי וָאֹמַר:

לֹא יָדְעָה נַפְשִׁי בְּשׁוּרִי עֲנַנֵי

עֵינֵי צְבִיָה בַּבְַּכִי יִדְמָעוּ

אִם הֵם אֲרֻבּוֹת הַשְּׁחָקִים נִפְתָּחוּ

אוֹ מֵי תְהוֹם רַבָּה אֲשֶׁר נִבְקָעוּ.

וַיֵבְךְּ הַשָׂר עַל צָרַת הַגְּבִירָה / וַיֹּאמַר הָאֵל יַצִּילֶנָּה מִכָּל צָרָה / כִּי יָדַעְתִּי כִּי נַפְשָׁהּ דָלְפָה מִתּוּגָה / וַיִּתְפַּלֵּל עַל הַיּוֹנָה מִמְּעֵי הַדְּאָגָה /

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

אֱלֹהִים בּעֻזָּךְ נְחֵה אֶל מְחוֹזָךְ

עְנִיָּה שְׁבִיָּה נְהַר צַר בְּזָאָהּ

אֲכָלָהּ הֲמָמָהּ זְמָן עַד לְכַמָּה

תְּלוּאַת תְּלָאָה טְלָאָהּ תְלָאָהּ

וְעֵת הִיא רְמוּסָה בְרָגְלָיו תְּהַלֵּך

לְמֻלָך לְדִמְיוֹן זְנַב הַלְּטָאָה

וְלֹא חָלְתָה יוֹם זְמַן רָע הֲרָגָהּ

וְשָׁבָה בְמוֹתָהּ בְּרִיאָה בְּרִיאָה

כְּבוֹדָהּ אֲשֶׁר נֶחֱלָתָה וְשָבָה

תְּמוּרַת רְפָאָהּ רְפָאָה רְפָאָה.

וַיְהִי כִּרְאוֹתִי כִּי הַשָׂר לֹא יִתְנַחֵם / וְכָל רוֹאָיו יַחֲרִיד וּבְעֵינָיו יִקָּחֵם / בָּרַחְתִּי מִבֵּית הַשָּׂר בְּנֶפֶשׁ נִבְהָלָה / אֶל הַמָּקוֹם אֲשֶׁר הָיָה שָׁם אָהֳלִי בַתְחִלָּה.


  1. כך במקור. כנראה שנפלה טעות דפוס במקור וצ"ל ‘ועתה’. (הערת פרויקט בן–יהודה)  ↩

  2. החזרה במקור, ויש כאן משחק מלים מכוון בשלוש משמעויות של השורש פ.ק.ד (הערת פרויקט בן–יהודה)  ↩

מחברות עמנואל – מחברת שלוש-עשרה / עמנואל הרומי


אָמַר עִמָּנוּאֵל הַמְּחַבֵּר: כִּשְׁמוֹעַ הַשָּׂר הַשִּׁירִים וְהַמְּלִיצוֹת

הָעֲלִיצִים וְהָעֲלִיצוֹת

אֲשֶׁר חִבַּרְתִּי אֶל יְהוּדָה גּוּר אַרְיֵה

חָשַׁב הֱיוֹתָם מַעֲמַד שֵׂכֶל וְכָהֶם לֹא הָיָה וְלֹא יִהְיֶה

וַיִשְׁאָלֵנִי וַיַּשְׁבִּיעֵנִי בֵאלֹהֵי אַבְרָהָם וֵאלֹהֵי נָחוֹר

אִם פָּגַעְתִּי בְיָמַי מְשׁוֹרֵר הֱשִׁיבַנִי אָחוֹר

וָאַעַן וָאֹמַר חַיֶּיךָ אֲדֹנִי פָּגַעְתִּי וּפָגָעְתִּי

וְקָלוֹן מִכָּבוֹד שָׂבַעְתִּי

וַיִּשְׂחַק הַשָּׂר אֲדֹנִי

וַיֹּאמַר אֵיךְ הָיָה הַדָּבָר בְּנִי

וָאַעַן וָאֹמַר הָיִיתִי בִקְצָת זְמָנִי

בְּמַחַץ הָעֳנִי מָחוּץ

וּבְמִשְׁעוֹל הַדְּאָגוֹת לָחוּץ

וּבְדַם דְּמָעַי מִתְגּוֹלֵל וְרָחוּץ

וְלִגְלוֹת בְּרֹאשׁ גּוֹלִים נָחוּץ

רַגְלִי אַחַת בִּפְנִים וְרַגְלִי אַחַת בַּחוּץ

הָיִיתִי מִסְתַּתֵּר מִן הַשֶּׁמֶש וּמִן הַיָּרֵחַ

כִּי דּוּדָאֵי הַיְּגוֹנִים נָתְנוּ רֵיחַ

וְהָיִיתִי כְגֵר בָּאָרֶץ וּכְאֹרֵחַ

מִלִּפְנֵי נוֹשַי בֹּרֵחַ

שׁוֹאֵג מִנַהֲמַת לִבִּי וּמַר צֹרֵחַ

וְשָׁכַחְתִּי שִׁירִים וּמְלִיצוֹת

כִּי הַיָּגוֹן פָּרַץ לִי פְּרָצוֹת

וַיְהִי הַיּוֹם וַאֲנִי הָיִיתִי בְעֹמֶק הַסְּעָרָה וְהַבִּלְבּוּל

וְלִבִּי כְּלֵב נֹחַ בְּבוֹא הַמַּבּוּל

וְהִנֵה נַעַר אֶחַד לִקְרָאתִי רָץ

וּבְיָדוֹ שִׁיר נִמְרָץ

בְּמֶלְקָחַיִם לָקַח לִבִּי

וּנְשָׁמָה לֹא נוֹתְרָה בִי

כִּי הָיִיתִי לִבְרוֹחַ נֶחְפָּז

וְהַשִּׁיר הָיָה נֶחְמָד מִזָּהָב וּמִפָּז

וְלֹא הָיִיתִי יָכוֹל לְקָרְאוֹ בְּמָתוֹן

וּפִי יְהַלֵּךְ עַל טוּרָיו כְּפִי הָאָתוֹן

וְלִבִּי יִסֹּב אָחוֹר וַיֵּלֶךְ עֲקַלָּתוֹן

וְהַנַּעַר יִקְרָאֵנִי בְחָזְקָה לָמָּה תַעֲמֹד מִשְׁתָאֶה

רְאֵה וַעֲשֵׂה בְּתַבְנִיתָם אֲשֶׁר אַתָּה מָרְאֶה

וָאֹמַר בְּקוֹל נֶעֱצָב

אֶל הַנַּעַר הַמַּחֲזִיק מַצָּב

וְלִרְאוֹת מָה אֲדַבֵּר בּוֹ וּמָה אָשִׁיב עַל תּוֹכַחְתּוֹ יִתְיַצָּב

מִי הוּא בַעַל הַשִּׁיר אֲשֶׁר כָּזֶה הֶרְאָנוּ

וַיַעַן הַנַּעַר וַיֹּאמַר הוּא יְהוּדָה צִיצִילְיָנוּ

הוּא שׁוֹלֵחַ אֵלֶיךָ הַשִּׁירָה אֲשֶׁר תִּרְאֵנָה

וְאִם מְעַט יוֹסִיף לְךָ כָהֵנָה וְכָהֵנָה

עַד אֲשֶׁר יִרְאֶה בְעֵינָיו וְהַנִּסָּיוֹן יִרְאֶנּוּ

מִמִּי יִתְפָּאֵר הַשִּׁיר מִמְּךָ אוֹ מִמֶּנּוּ

כְּשָמְעִי פַּלְגֵי מַיִּם יָרְדוּ עֵינָי

וּשְׂעִפַּי וְרַעְיוֹנַי

לֹא מָצְאוּ חָזוֹן מֵאֲדֹנַי

הָיִיתִי מִתְהַפֵּך עַל חֲלָצָי

וְכִלְיוֹתַי מֵאַנוּ לִהְיוֹת יוֹעֲצָי

וְהַיְּגוֹנִים עָלוּ כְּשׁוּעָלִים עַל פְּרָצָי

וָאֹמַר אֶל הַנָּעַר תְּשׁוּבַת הַשִּׁיר לֹא אוּכַל הַיּוֹם לִתֵּן

לֵךְ וָשׁוּב וּמָחָר אֶתֵּן

וַיֵּלֶךְ הַנַּעַר לְדַרְכּוֹ

וָאֶתְבּוֹנַן עַל הַשִּׁיר לָדַעַת חִין עֶרְכּוֹ

וָאֶרְאֶה וְהִנֵּה הַשִּׁיר נֶחְמָד לְעֵין רֹאָיו

שָׁלֵם בְּכָל תְּנָאָיו

וּמֻפְלָג בְּכָל עִנְיָנָיו

וְלֹא יָכְלוּ שִׁירַי לַעֲנוֹת אוֹתוֹ כִּי נִבְהֲלוּ מִפָּנָיו

הָיִיתִי כָּל הַלַּיְלָה מִתְגּוֹלֵל עַל יְצוּעָי

וּמִתְבּוֹסֵס בִּדְמֵי דְמָעָי

הוֹמֶה וְגוֹעֶה לְרוֹב עִצְבוֹנָי

וַתִּדַּד שְׁנָתִי מֵעֵינָי

הַבֹּקַר אוֹר וְהַנַּעַר בָּא

וְלִבִּי חָלָל בְּקִרְבִּי מֵת אוֹ נִשְׁבַּר אוֹ נִשְׁבָּה

וַיֹּאמֶר אֵלָי מַה לְךָ נֶחְבָּא

נִבָּא וְלֹא יָדַע מַה נִבָּא

וָאַעַן וָאֹמַר כָּל הַלַּיְלָה שָׂבַעְתִּי נְדוּדִים

כִּי גַנָּבִים בָּאוּ לִי הַלַּיְלָה וְשׁוֹדְדִים

וְהִנֵּה צָרַת לְבָבִי כְּמֵאָה עֵדִים

וַיָּבֶן הַנַּעַר כִּי אֲנִי כִּמְהָתֵל בּוֹ וּכְמוֹצֵא עֲלִילוֹת

וְתוֹלֶה בְכָל דָּבָר שֶׁאוּכַל לִתְלוֹת

וַיֹּאמֶר חַי יוֹצֵר הָרִים וּבוֹרֵא רוּחַ

כִּי אַתָּה בֵין הַמְּשֹׁרְרִים יַבְרוּחַ

וְלֹא אָסוּר מֵהֵנָּה אוֹ אֶהְיֶה רוֹצֵחַ אוֹ רָצוּחַ

עַד תְּשִׁיבֵנִי אוֹ תוֹדֶה כִּי אַתָּה נָצוּחַ

וּבִרְאוֹתִי כִּי כָלָה שׁוֹד וְאָפֵס הַמֵּץ

וְהַנַּעַר לְהַחֲזִיק מַצָּב יִתְאַמֵּץ

כִּמְעַוֵּל וְחוֹמֵץ

לִבִּי בְקִרְבִּי יִתְחַמֵּץ

וָאֹמְרָה אֶל הַנַּעָר בְּפַתַח וּבְקָמֵץ

לֵךְ אֱמֹר אֶל הַשּׁלֵחַ אוֹתְךָ כִּי שִׁירָתוֹ עָרְבָה לִי כְּרֵיחַ נִיחוֹחַ

וּמַה לָאֶרֶז אֲשֶׁר בַּלְבָנוֹן לִשְׁלֹחַ אֶל הַחוֹחַ

וְעִתִּי לֹא עֵת דֹּדִים אַךְ עֵת לִשְׁכֹּחַ

וְלִפְנֵי שִׁירָיו לֹא עָצַרְתִּי כֹחַ

הָיִיתִי כְּגֶבֶר אֵין אֱיָל וּכְאִישׁ רָצוּחַ

וּמֵהַיּוֹם הִנֵּנִי מַחֲזִיק עַצְמִי נָצוּחַ

וַיֵּלֶךְ הַנַּעַר וּבְפִיו בְּשׂוֹרָה

וַיְסַפֵּר כָּל דְּבָרַי לְבַעַל הַשִּׁירָה

וַיּוֹסֶף עוֹד יְהוּדָה שְׁלֹחַ שִׁירִים רַבִּים וְנִכְבָּדִים

חִכָּם מַמְתָּקִים וְכֻלָּם מַחְמָדִים

לֹא רָאִיתִי כָהֵם בְּצַחוּת לְשׁוֹנָם

וּבְתֹקֶף מִלֵּיהֶם וּבְטוּב עִנְיָנָם

וְאוֹתִי יוֹם יוֹם יִדְרְשׁוּ

כָּל הַיּוֹם וְכָל הַלַּיְלָה תָּמִיד לֹא יֶחֱשׁוּ

וַאֲנִי הָיִיתִי כְּחַיַּת הַשָּׂדֶה נִרְדָם וְעֻלְפֶּה

כְּחֵרֵש אֲשֶׁר לֹא שׁוֹמֵעַ וּכְאִלֵּם לֹא יִפְתַּח פֶּה

כִּי שָׂבְעָה נַפְשִׁי בָרָעוֹת

וְעֵינַי כָּלוּ בַדְּמָעוֹת

וְעָמַדְתִּי כְמוֹ שָׁנָה וּשְׁנָתַיִם וְלֹא עֲנִיתִיו

וּבֹשְתִּי וְגַם נִכְלַמְתִּי בָּעִתִּים אֲשֶׁר רְאִיתִיו

עַד אֲשֶׁר נִסְכְּרוּ מַעְיְנוֹת הַצָּרוֹת וְהָאֲנָחוֹת

וְנִפְתְּחוּ אֲרֻבּוֹת הַגִּיל וְשׂבַע שְׂמָחוֹת

וְאָז לָבְשָׁה אוֹתִי רוּחַ קְנָאוֹת

וְחִבַּרְתִּי אֵלָיו שִׁירוֹת בַּפָּז מְסֻלָּאוֹת

הָיוּ קְצָתָם דִּבְרֵי נְבוּאוֹת

יְפוֹת הַמַּרְאֶה וְהַבְּרִיאוֹת

וְהִנֵּה הָיוּ עַל כַּנְפֵי נִשְׁרֵי הַמַּהֲלָלִים נְשׂוּאוֹת

יָשְׁרוּ בְעֵינָיו וְהִלְּלָם

וְעַד רוּם הַשְּׁחָקִים הֶעֱלָם

וַיֹּאמֶר כִּי כָהֵם לֹא רָאָה מִן הָעוֹלָם וְעַד הָעוֹלָם

וַיֹּאמֶר הַשָּׂר הֲתִזְכֹּר הַשִּׁירִים אֲשֶׁר שָׁלַח אֵלֶיךָ יְהוּדָה

וָאַעַן וָאֹמְרָה חַיֶּיךָ שַׂר הַתְּעוּדָה

הִפְסַדְתִּים לְרֹעַ זְמַנִּי

וְעֲלֵיהֶם גָדַל יְגוֹנִי

וּפַלְגֵי מַיִם תֵּרֵד עֵינִי

אָמְנָם אִם תֹּאבֶה לִשְׁמֹעַ הַשִּׁירִים אֲשֶׁר אֵלַיו שָׁלַחְתִּי

אָז תֹּאמַר אֵלַי סָלָחְתִּי

וַיֹּאמֶר אֵלַי הַשָּׂר הַגֶּד נָא שִׁירֶיךָ

פְּתַח פִּיךָ וְיָאִירוּ דְבָרֶיךָ

וָאֹמַר הַשִּׁיר הָרִאשׁוֹן אֲשֶׁר אֵלָיו שָׁלַחְתִּי הוּא זֶה

וְלֹא תְשַקֵּץ עֱנוּתוֹ וְלֹא תִבְזֶה

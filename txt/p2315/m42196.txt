


	
	תּוֹכַחַת מְגֻלָּה: חֵלֶק שְׁלִישִׁי: הֶבֶל בְּנֵי אָדָם
	




– – – – – – ͝ – – – –



דרדע: דּוֹד גַּן הַזֶּה אָכֵן יְחַוֶּה דַעַת

אֶל־הַבָּאִים תּוֹכוֹ וְרֹב שִׂכְלֶךָ.

אִילָנוֹת וּנְטָעִים אֲשֶׁר בּוֹ רַבּוּ

הַצִּצִּים וּפְרָחִים אֲשֶׁר נָטַעְתָּ

אֵל מֵאֵלֶּה שׁוֹנִים יְרַוּוּ עָיִן

וּבְשָׂמָיו נִזָּלִים יְשִׁיבוּן נָפֶשׁ.

פַּלְגֵי־מַיִם הָעוֹֽבְרִים בֵּינֵימוֹ

כִּי יִגָּחוּ שָׁרְשָׁם יְעַדְּנוּ אֹזֶן

בִּדְמָמָה הַדַּקָּה אֲשֶׁר נִשְׁמַעַת

מִנִּי דָּכְיָם הַקָּט וּמִשְׁבָּרֵמוֹ.

אַךְ מָה אוֹסִיף הַמַּהֲלָל וָשֶׁבַח

אִם פֹּעַל זֶה רַק כּוֹנֲנוּ יָדֶיךָ.

כלכל: אוֹדֶה דַּרְדַּע כִּי פֹה לְבַד יַצְמִיחַ

כָּל־יִשְׁעִי בִלְבָבִי וְחֶפְצִי סֶלָה.

מֵהֵד תֵּבֵל פֹּה אֶמְצְאָה מָנוֹחַ

וּבְהַשְׁקֵט יָשׁוּט לֵב בְּיַם הַדָּעַת

בַּל יִבָּהֵל מִשֹּׁד זְמָן וָנָגֶף.

אָמְנָם זוּלָתִי זֹאת בְּגַנִּי אַחַז

הַבְלֵי תֵבֵל וּכְמוֹ בְמַרְאֵה טֹהַר

אֶגְלֶה סִכְלוּת הַיּוֹֽשְׁבִים עָלֶיהָ.

דרדע: הֵן בַּדָּבָר הַזֶּה פְּלִיאָה דָּעַת

מִמֶּנִּי, לֹא אָבִין אֲשֶׁר אָמָרְתָּ

אִם לֹא תַרְאֵנִי מִפְלְאוֹת מִלֶּיךָ.

כלכל: נָקֵל לִי הַדָּבָר בְּעֵת נַחְקֹרָה

יַחַד הַבְלֵי תֵבֵל וְצֶאְצָאֶיהָ.

תַּחַת עַרְמוֹן הַלַּז גְּשָׁה וָשֵׁבָה

כִּי מֵחֹם שֶׁמֶשׁ צִלֲּלוֹ מַחְסֵנוּ.

דרדע: לֵישֵׁב אִתְּךָ כִּרְצוֹֽנְךָ חָפָצְתִּי

דַּבֵּר עַתָּה אֵלַי וְאַטֶּה אֹזֶן.

כלכל: אַשְׁפִּיל אַרְצָה עֵינַי וְשָׁם אַבִּיטָה

כַּר נִרְחָב מַה־נֶּחְמָד לְמַרְאֶה יֶהִי

כִּי בִּפְרָחָיו שׁוֹנִים גְּוָנָיו רָבוּ

עַל־הַמַּרְאֶה לָבָן אֲדַמְדָּם שָׁמָּה

אַרְגָּמָן וִירַקְרַק וְצָהֹב תֵּרֶא

בִּתְכֵלֶת אוֹ שָׁנִי אֲחוּזִים יָחַד

כִּי בָם עַיִן תִּשְׂמַח וְתַעְלֹז רוּחַ.

וּבְהַבִּיטִי אֶל־מִפְלְאוֹת הַטֶּבַע

אֶגְמֹר כִּי אֵין צַיָּר כְּמוֹ צוּרֵנוּ.

בִּתְלֻנּוֹת עַל־תֵּבֵל אֲזַי אֵצֵאָה

כִּרְאוֹת הָמוֹן רַבָּה בְחוּצוֹתֶיהָ

תָּמִיד לָלֶכֶת אַחֲרֵי עֵינֵימוֹ

שׁוֹנִים אַחַר צִיּוּר צְבָעִים אָלֶף.

זֶה יִבְנֶה בֵית מִדּוֹת עֲלִיּוֹת רֶוַח

מָשֻׁחַ בַּשָּׁשַׁר בְּסִפּוּן אָרֶז

וִיחַקֶּה עַל־הַקִּיר נְטַע שַׁעְשׁוּעַ

אִילָנוֹת וּדְשָׁאִים וְצִיץ וָפֶרַח

אוֹ עַל־בַּיִת אָרוּג שְׁתִי וָעֵרֶב

רָבוּעַ בִּסְבִיבוֹ כְּמוֹ מִקְלָעַת

אוֹ לִפְקָעִים עָשׂוּי בְּצִפּוּי בֶּצֶר

בִּתְבוּנָה וּבְחָכְמָה יְצַיֵּר צֶבַע.

שֶׁבַע עֵינַיִם בִּמְעִיל וּכְתֹנֶת

יִבְחַר אַחֵר לִרְאוֹת, וּבִלְבוּשׁ־אֵשֶׁת

לָכֵן חוּר כַּרְפָּס יֶאֱחֹז וּתְכֵלֶת

עִם חַבְלֵי־בוּץ בָּהֶם לְרִקְמָתָיִם,

פֶּרַח יֵצֵא מִפֹּה וּמִפֹּה פֶרִי

מִזֶּה כָּל־צִיץ שָׂדֶה וְגַם עֵץ־יָעַר.

בִּדְבָרִים אֵל מוֹצָא יִהְיֶה לָכָסֶף

וּבְמָקוֹם זֶה זָהָב הֲכִי יָזֹקּוּ.

אִם זֶה לֹא הֶבֶל הוּא, אֱמֹר לִי עָתָּה

מַה־יּוֹעִילוּ לָהֶם וּמַה־יַּצִּילוּ

אִם אָז בִּלְבַד יִתְהַלְּכוּ בַצֶּלֶם.

אֹמַר אָז אֶל־לִבִּי: פְּתָיִים אֵלֶּה

יָסוּרוּ פֹה אִם חוֹֽמְדִים הַצֶּבַע.

דרדע: מָתוֹק אֶל־פִּי מִנִּי דְבַשׁ נִיב פִּיךָ

הַזָּקוּק וּמְיֻסָּד בְּאַדְנֵי־צֶדֶק

לָכֵן אָנָּא הוֹסֵף שְׂאֵת נִיבֶךָ.

כלכל: אָנֹכִי הָרוֹאֶה בְּהַבְלֵי־חָלֶד

אֶרְאֶה רַבִּים תּוֹעִים בְּשִׁיר וָזָמֶר

כִּי הַשֵּׁנָה הֵן נִגְזְלָה מֵהֵמָּה

כִּי אֶת־אָכְלָם הֵם יִמְנְעוּ מִפִּימוֹ

גַּם יַעַזְבוּ מִכָּל־חֲפָצֵי־בָיִת

אִם לֹא מִטֶּרֶם יִשְׁמְעוּ אוֹתָמוֹ.

אִשָּׁה נוֹלָדָה מִקְּטַנֵּי־אָרֶץ

קָלוֹן וּכְלִמַּת עָם וּבוּז מִשְׁפָּחַת

אִם קוֹלָהּ קוֹל עָרֵב וְשִׁיר מַשְׁמָעַת

תוּבַל בֵּיתָה אִישִׁים נְטִילֵי־כָסֶף

כִּגְבֶרֶת תֵּחָשֵׁב וְשָׁם שַׁלָּטֶת.

גַּם אִישׁ כִּי גִידֵי פַחֲדָיו נִתָּקוּ

אוֹ מָעוּךְ יֶהִי אוֹ מְרוֹחַ אָשֶׁךְ

גַּם הוּא לָשִׁיר יֻקַּח בְּרֹב תִּפְאָרֶת

אֵלָיו יָרִיצוּ מַתְּנוֹת הָעֶרֶךְ

זוּלָתִי הַשָּׂכָר אֲשֶׁר לוֹ יֵרֶב

יִטֶּה אַרְצָה מִנְלָם וְהוּא הָעֹשֶׁר

מִיּוֹם אֶל־יוֹם יַרְבֶּה עֲדֵי אֵין־חֵקֶר.

אֵלֶּה כִי אַחַר שִׁיר וְקוֹל יֶהְבָּלוּ

כִּי רֻבָּם אוֹ כֻלָּם עֲגָבִים הֵמָּה

מַפְרִיעִים מֵחָכְמָה וּמוּסָר נָפֶשׁ.

לָמָּה יִשְׁגוּ בָהֶם, הֲלֹא טוֹב לָמוֹ

לָסוּר אֶל־צֵל גַּנִּים וְכַרְמֵי־חֶמֶד

כִּי שָׁם לָאֶלֶף צִפֳּרִים קִנֵּנוּ

וּבְצִפְצוּפָם קוֹלוֹת עֲרֵבִים יַחַד

רַק מֶתֶק וּנְעִימוּת בְּלֵב יִתֵּנוּ

בִּלְתִּי זָהָב וּבְלֹא מְחִיר וָכָסֶף

וּבְטֹהַר רַב כִּי לֹא יְהַבֵּל נָפֶשׁ.

אַף זֶה יִבְחַן לִבִּי בְּשִׁבְתִּי הֵנָּה.

דרדע: גַּם בָּזֶה עַל־נָכוֹן רְאוֹת הֵטָבְתָּ

לֹא אֶשְׁנֶה אָנִי אַחֲרֵי מִלֶּיךָ

כִּי טֹבוּ לִי בִמְאֹד וְכִי נָעֵמוּ.

כלכל: עוֹד אָשׁוּבָה אֶרְאֶה בְגַנִּי הֵנָּה

כָּל־שִׂיחַ הַשָּׂדֶה אֲשֶׁר הַפֶּרִי

יִתֵּן אֵלַי בִּלְבַד בְּעִתּוֹ סֶלָה

כַּאְשֶׁר חָק לוֹ צוּרִי בְּיָצְרוֹ אָרֶץ.

מִזֶּה אַבִּיט יָמִין פְּתַיּוּת־גָּבֶר

הוֹלֵךְ אַחַר יִצְרוֹ לְבַקֵּשׁ אֹכֶל

הַיָּקָר בַּנִּמְצָא וְעוֹד אֵינֶנּוּ

עַד־כִּי יוּבַל לוֹ עַל־יְדֵי הַכָּסֶף

מִמֶּרְחַקִּים כָּל־מַאֲוַיֵּי נָפֶשׁ

וִישַׁלֵּם עַל־אֶחָד בְּשִׁבְעָתָיִם

עִם כִּי נָמַר רֵיחוֹ וְאָבַד טָעַם

אוֹ בַבָּכִיר אוֹ בַאֲפִיל בִּפְרֹחַ,

אָמְנָם אֶל־חֵךְ יִטְעַם לְבַד מֵהֶבֶל.

אָכֵן הַבְלֵי הַחֵךְ מְאֹד נִפְלָאוּ

וּלְעֵין שִׂכְלִי הִנֵּה אֲזַי יָבוֹאוּ.

יִמָּלֵא לוֹ שִׂפְקוֹ וְעוֹד לוֹ יֵצַר

וּתְבוֹאֶנּוּ עוֹד תַּאֲוָה שׁוֹקָקֶת

כִּי תִתְאַוֶּה נַפְשׁוֹ אֲשֶׁר אֵין שָׁמָּה

וּלְמֵרָחוֹק עֵינָיו הֲלֹא יַבִּיטוּ.

הָאוֹצָרוֹת יִהְיוּ מְלֵאֵי יָיִן

כִּי לוֹ הַגֶּפֶן נָתְנָה לָשׂבַע

אַךְ יֵין חֶלְבּוֹן נִדְרָךְ בְּעִיר דַּמֶשֶׂק

אוֹ בִיקָבֵי כַרְמֶל יְהִי לוֹ נַחַת

עַל־שֻׁלְחָנוֹ, כִּי בוֹ יְשַׂמַּח חֵבֶר.

בָּז כָּל־בָּא מֵאַרְצוֹ, וְיָבוּס נֹפֶת

אִם נִמְצָא שָׁם, גַּם זִהֲמַתּוּ לָחֶם

וּבְשַׂר פָּר הַשָּׁמֵן, אֲשֶׁר שָׂדֵהוּ

יִתֵּן אֵלָיו, יַעַן בְּכָל־עֵת יֶהִי.

וִיבַקֵּשׁ לוֹ תַחְתָּם בְּחַיְתוֹ־יָעַר

בִּדְגֵי הַיָּם וּבְצִפֳּרִי שָׁמָיִם

הַיְקָרִים מִנִּי־פָז לְפִיו הַטָּרֶף.

הוֹן מֵהֶבֶל יִמְעַט בְּבֵיתוֹ כָכָה.



דרדע: יִהְיֶה חִזּוּק וָאוֹת לְאִמְרוֹתֶיךָ:



הֶבֶל כָּזֶה הָיָה בְקֵיסָר רוֹמִי

כִּי נַחַת שֻׁלְחָנוֹ בְּהָרֵי־גֹבַהּ

הָיָה לָשׂבַע מִדְּגֵי הַמָּיִם

וּבְשִׁבְתּוֹ בִלְבַב־יָם בְּאַוַּת־סֶכֶל

חָפֵץ חַיַּת שָׂדֶה וְעוֹף שָׁמָיִם.

כָּכָה מַהֵר כָּל־אוֹֽצְרוֹתָיו תָּמּוּ.

אִם עַל־כָּל־עֵץ עוֹד אַעֲמִיד הָעָיִן

אָשׁוּב אֶרְאֶה עוֹד מֵאֱנוֹשׁ הֶהָבֶל.

הֵא עַרְמוֹן זֶה, כִּי רַעֲנָן עָלֵהוּ

כִּי צֵל דָּלִיּוֹתָיו לְמַחֲסֶה לָנוּ

כֻּלָּם כַּאֲשֶׁר תִּרְאֶה עֲמוּסֵי פֶרִי.

הַגֶּפֶן שׁוּר הַמְשַׂמְּחָה עֵינַיִם

בִּירַק פֹּארוֹת וּבְאַשְׁכְּלוֹת הִבְשִׁילוּ,

שׁוּר בִּתְאֵנָה כִּי בִּשְּׁלָה פַגֶּיהָ

זָבַת חָלָב מֵהֶם וּמִבַּדֶּיהָ.

הִנֵּה יָבוֹא מַהֵר סְתָיו גַּם הֵנָּה

כִּי תָבוֹא אָז מִנִּי מְזָרִים רוּחַ

לִפְתֹּחַ דֶּלֶת אוֹֽצְרוֹת הַשָּׁלֶג

תָּבוֹא קָדִים עַזָּה וְשַׂעַר קָטֶב

וּבְנָשְׁבָם בָּם יָפְיָם אֲזַי יָסִירוּ

יִבֹּלוּ הֶעָלִים וּפִרְיָם יָחַד

וּלְבַד הָעֵץ יִשְׁאַר כְּעַל־הָר תֹּרֶן

עָרֹם מִכָּל־הוֹדוֹ וּמִכָּל־הֶדֶר.

אָכֵן אֶל־עֵץ שָׂדֶה יְדַמֶּה גָבֶר

כִּי כָל־יָפְיוֹ יֹאבַד בְּפִתְאֹם פֶּתַע.

הִנֵּה כִּי עַתָּה רַעֲנָן תִּרְאֵהוּ

מִתְעָרֶה כָאֶזְרָח וְכֻלּוֹ יֹפִי

לוֹ תֹאַר לוֹ הָדָר וְלוֹ הָעֹצֶם

וּבְשַׁחְרוּתוֹ נָעִים הֲכִי מַרְאֵהוּ,

אָמְנָם פִּתְאֹם מָוֶת יְסֹעֵר סָעַר

וִיחוֹלֵל עַל־רֹאשׁוֹ וְחָלַף תֵּאֵר

יִשְׁאַר נֵצֶר נִתְעָב כְּנֻס הַלֵּחַ

וָהוֹד מֶנּוּ יִמְעַט וְרַב הַפֶּגֶר

כִּי יָשׁוּב אֶל־עָפָר כְּרֵאשִׁיתֵהוּ.

הַנּוֹתָר מֵרִמָּה וּבוּז תּוֹלָעַת

כֵּן סוֹף אָדָם לִסְחִי וּמָאוֹס יֶהִי.

הָהּ כִּי לוּ גֶבֵר יַחֲשֹׁב כָּמוֹנִי

לֹא יִשְׁעֶה בִלְבָבוֹ בְּדִבְרֵי־שָׁקֶר

יַשְׁפִּיל כַּנְפֵי רוּמוֹ גְּבַהּ עֵינָים

כִּי יַבִּיט אֶת־סוֹפוֹ הֱיוֹתוֹ אָפֶס.

כִּרְנַן הַמִּתְעַלֵּס בְּנוֹצַת־הֶדֶר

כִּי אַרְצָה יַשְׁפִּילָהּ בְּהַבִּיט רֶגֶל

תּוֹעֶה אַחַר תֵּבֵל וְרוֹדֵף יֵצֶר

מוֹרָשֵׁי־לֵב יִנָּֽתְקוּ מִמֶּנּוּ.

יִרְאַת שַׁדַּי לֹא יַעֲזֹב הַגֶּבֶר

לִמָּנַע מֵרֵעוֹ גְמִילוּת־חָסֶד

אָמְנָם הוּא לֹא יָבִין בְּיוֹם הַמָּוֶת

כִּי לֹא יִקַּח כָּבוֹד וְלֹא הַבֶּצַע

יֵרֵד עִמּוֹ יוֹמָם אֲשֶׁר יַגִּיעַ

וּלְאַדְמָתוֹ עֵרֹם וְעֶרְיָה יֵלֶךְ

יִשְׂחַק רַק אֶל־יוֹם אַחֲרוֹן הַפֶּתִי

וּכְאִלּוּ הוּא כָרַת בְּרִית אֶת־מָוֶת

בֹּאוֹ פִּתְאֹם אַל־יַאֲמֵן וּבְרֶגַע

זֶה הוּא יוֹתֵר מִכָּל־הֲבָלִים הָבֶל.

דרדע: אֶת־דַּבֶּרְךָ כַּלְכֹּל אֱמֶת צָדָקְתָּ

וּבְשָׁפְטֶךָ רֵעִי הֲכִי זָכִיתָ

אַחַר הַהֶבֶל יֵֽלְכוּ יֶהְבָּלוּ

אִישִׁים כִּי יוֹם הַמָּֽוְתָה שָׁכָחוּ

אוּלָם תַּחְקֹר כִּי לֹא נְעָרִים הֵמָּה

בִּלְבַד כִּי בָמוֹ בוֹֽטְחִים וָבָאוּ

אַף גַּם יָשִׁישׁ וָשָׂב וּבָא עַד־כָּלַח

מִיּוֹם אֶל־יוֹם קָרֵב עֲלֵי־פִי־פַחַת

יַחְשֹׁב וִיִדַמֶּה כִּי רְחוֹקָה מֶנּוּ.

שׁוּר שָׁם זָקֵן כֻּלּוֹ רְתֵת וָרָעַד

כִּי נָס לֵחֹה כִּי כָהֲתָה לוֹ עָיִן

וּבְיַד יָמִין סוֹמֵךְ עֲלֵי מִשְׁעָנֶת

עוֹד כִּי תוֹכָהּ יַד הַשְּׂמֹאל סוֹגֶרֶת

הֵיטֵב מַפְתֵּחַ אוֹֽצְרוֹת הָרֶשַׁע

כִּי נִתֵּק מִבְטַחוֹ וְיַבְעִיתוּהוּ

בַּלָּהוֹת כִּי כַסְפוֹ יְגֻנַּב מֶנּוּ.

שׁוּר אִישׁ אַחֵר כִּי זָרְקָה עָלֵהוּ

שֵׂיבָה כִּי שַׂעְרַת רֹאשׁ מְאֹד הִלְבִּינָה

מַשְׁחִית אוֹתָהּ וּפְאַת זְקָנוֹ יָחַד

יַעַן זִמּוֹתָיו הֵם לְכַחֵד זֹקֶן

וּשְׂעַר צָהֹב אָחוּז בְּרֶשֶׁת מֶשִׁי

כִּי לִגְזֹז מִמֵּתִים אֲנָשִׁים זָדוּ

יַעַל תַּחַת שַׂעְרוֹ עֲלֵי רֹאשֵׁהוּ.

וּבְכֵן בִּלְבוּשׁ מִכְלוֹל יְפַת הַתֹּאַר

יִרְדֹּף וּבְאַהְבָתָהּ כְּבָחוּר חֶמֶד

יִשְׁגֶּה וִיאַבֵּד הוֹן וְאֵין בּוֹ דַעַת

לֵאמֹר הִנֵּה כֹחִי כְבָר כִּלֵּיתִי.

אֵין זֶה כִּי אִם רֹעַ לְבַב הַיֵּצֶר.

אִם מִישִׁישִׁים כֵּן הַתְּבוּנָה סָרָה

מַה־נּוֹחִיל בִּנְעָרִים, אֲהָהּ כִּי הֶבֶל

מָלֵא לֵב הָאָדָם וְכֵן דִּבָּרְתָּ.

כלכל: הִנֵּה רָפָה הַיּוֹם וְטֶרֶם עֶרֶב

יָבוֹא פֹה בִּצְלָלָיו לְכַסּוֹת אָרֶץ

בֹּאָה אִתִּי דּוֹדִי וּפֹה אֶרְאֶךָּ

כָּל־מַסְלוּלֵי הַגַּן לְךָ נִסְתָּרוּ

כָּל־מַטָּעָיו שׁוֹנִים וְכֶרֶם חֶמֶד

כִּי עַד־עַתָּה לֹא גִלְּתָה עֵינֶךָ

כִּלְבָבְךָ אָז תָּשׁוּב לְמוֹ בֵיתֶךָ.

דרדע: אִתְּךָ הִנֵּנִי כַּאֲשֶׁר דִּבָּרְתָּ.
